compress_bp = Blueprint("compress", __name__)

_RESTORE_EXCLUDE = frozenset({"src_exif", "dst_exif"})


@compress_bp.route("/api/compress", methods=["POST"])
def api_compress():
    ok, msg = _check_disk_space()
    if not ok: return jsonify({"error": msg}), 507
    enc     = request.form.get("encoder","libx265")
    if enc not in ALLOWED_ENCODERS:
        return jsonify({"error": "Invalid encoder!"}), 400
    try:
        crf = int(request.form.get("crf", 24))
    except ValueError:
        return jsonify({"error": "Invalid CRF"}), 400
    crf_max = 63 if enc in ("libsvtav1", "libsvtav1_p4", "libaom-av1") else 51
    if not (0 <= crf <= crf_max):
        return jsonify({"error": f"CRF must be 0–{crf_max}"}), 400
    out_ext = request.form.get("format","original")
    mute        = request.form.get("mute","0") == "1"
    preserve_dv = request.form.get("preserve_dv","1") == "1"  # default True — matches UI checkbox default
    crf_auto    = request.form.get("crf_auto","0") == "1"
    # DV output requires HEVC — silently ignore preserve_dv for other encoders
    if preserve_dv and enc not in {"libx265", "hevc_nvenc", "hevc_amf", "hevc_qsv"}:
        preserve_dv = False

    client_batch_id = request.form.get("client_batch_id", "")
    if client_batch_id and not re.match(r'^[a-z0-9]{6,16}$', client_batch_id):
        return jsonify({"error": "Invalid batch token!"}), 400

    job_id = uuid.uuid4().hex[:8]
    try:
        src_path, orig_name = _get_upload(request, job_id)
    except ValueError as e:
        return jsonify({"error": str(e)}), 400

    orig_ext = Path(orig_name).suffix.lower()
    if orig_ext not in VIDEO_EXTS:
        return jsonify({"error": "Only videos are supported!"}), 400
    if out_ext == "original":
        out_ext = orig_ext

    with _jobs_lock:
        has_active = any(j.get("type") == "compress" and j.get("status") in ["processing", "waiting"] for j in jobs.values())
        if has_active:
            if not client_batch_id or client_batch_id != _jobs_module.active_client_batch_id:
                try: src_path.unlink()
                except: pass
                return jsonify({"error": "System is busy processing another batch!"}), 429
        else:
            if client_batch_id != _jobs_module.active_client_batch_id:
                _jobs_module.current_compress_batch_id += 1
                _jobs_module.active_client_batch_id = client_batch_id
            _jobs_module.active_compress_batch_settings = {
                "encoder": enc,
                "format": request.form.get("format", "original"),
                "crf": crf,
                "mute": mute,
                "preserve_dv": preserve_dv,
                "crf_auto": crf_auto,
            }
        batch_id_snapshot = _jobs_module.current_compress_batch_id

    ui_meta = {}
    try:
        ui_meta_str = request.form.get("ui_meta", "{}")
        if ui_meta_str:
            ui_meta = json.loads(ui_meta_str)
    except:
        pass

    if not ui_meta and src_path and src_path.exists():
        ui_meta = _extract_ui_meta(src_path)

    with _jobs_lock:
        batch_index = sum(1 for j in jobs.values() if j.get("batch_id") == batch_id_snapshot)
        jobs[job_id] = {
            "status":"processing","progress":0,
            "message":"Preparing...",
            "output":None,"src_name":orig_name,"type":"compress",
            "batch_id": batch_id_snapshot,
            "file_index": batch_index,
            "ui_meta": ui_meta,
            "src_size": src_path.stat().st_size if src_path and src_path.exists() else 0,
            "created_at": time.time()
        }
    threading.Thread(
        target=_compress_job,
        args=(job_id, src_path, enc, crf, out_ext, mute, preserve_dv, crf_auto),
        daemon=True
    ).start()
    return jsonify({"job_id": job_id})


@compress_bp.route("/api/active_compress_batch", methods=["GET"])
def api_active_compress_batch():
    with _jobs_lock:
        has_active = any(j.get("type") == "compress" and j.get("status") in ["processing", "waiting"] for j in list(jobs.values()))
        if not has_active:
            return jsonify({"active": False, "jobs": {}})
        batch_jobs = {
            jid: {k: v for k, v in j.items() if k not in _RESTORE_EXCLUDE}
            for jid, j in jobs.items()
            if j.get("type") == "compress" and j.get("batch_id") == _jobs_module.current_compress_batch_id
        }
        settings_snapshot = dict(_jobs_module.active_compress_batch_settings)
    return jsonify({
        "active": True,
        "jobs": batch_jobs,
        "settings": settings_snapshot,
        "server_time": time.time()
    })

@files_bp.route("/api/download_file/<job_id>")
def api_download_file(job_id):
    with _jobs_lock:
        job = jobs.get(job_id)
        if not job or not job.get("output"):
            return jsonify({"error":"File not ready!"}), 404
        output   = job["output"]
        job_type = job.get("type", "")
        src_name = job.get("src_name", "video")

    p = Path(output)
    if not p.exists():
        return jsonify({"error":"File has been auto-deleted (7-day limit). Please re-process."}), 404

    if job_type == "compress":
        dl_name = p.name

    elif job_type in ("social", "audio"):
        src_stem   = Path(src_name).stem
        clean_stem = re.sub(r"[^\w\-. ]", "_", src_stem)
        suffix_map = {"social": "_social", "audio": "_audio"}
        dl_name = f"{clean_stem}{suffix_map.get(job_type, '')}{p.suffix}"

    else:
        dl_name = p.name

    return send_file(str(p), as_attachment=True, download_name=dl_name)


@files_bp.route("/api/log/<job_id>")
def api_log(job_id):
    with _jobs_lock:
        job = jobs.get(job_id, {})
        log_path_str = job.get("log_path")
        log_file_name = job.get("log_file", f"ffmpeg_err_{job_id}.log")
    if log_path_str and Path(log_path_str).exists():
        log_path = Path(log_path_str)
        log_name = log_path.name
    else:
        log_name = log_file_name
        log_path = Path(OUTPUT_DIR) / log_name

    if not log_path.exists():
        return jsonify({"error": "No error log available for this job."}), 404
    try:
        content = log_path.read_text(encoding="utf-8", errors="replace")
        return jsonify({"log": content, "log_file": log_name})
    except Exception as e:
        return jsonify({"error": str(e)})


@files_bp.route("/api/cleanup/<job_id>", methods=["DELETE"])
def api_cleanup(job_id):
    with _jobs_lock:
        job = jobs.pop(job_id, None)
    procs.pop(job_id, None)
    if job:
        if job.get("status") != "done":
            out = job.get("output")
            if out:
                p = Path(out)
                if p.is_file():  p.unlink(missing_ok=True)
                elif p.is_dir(): shutil.rmtree(p, ignore_errors=True)

        log_path_str = job.get("log_path")
        if log_path_str:
            try: Path(log_path_str).unlink(missing_ok=True)
            except: pass

        log_name = job.get("log_file")
        if log_name:
            try: (Path(OUTPUT_DIR) / log_name).unlink(missing_ok=True)
            except: pass
    return jsonify({"ok": True})


@files_bp.route("/api/probe", methods=["POST"])
def api_probe():
    ok, msg = _check_disk_space()
    if not ok: return jsonify({"error": msg}), 507
    f = request.files.get("file")
    if not f: return jsonify({"error": "No file"}), 400

    orig_name = _sanitize_upload_filename(f.filename)
    if not orig_name:
        return jsonify({"error": "Invalid filename"}), 400
    if Path(orig_name).suffix.lower() not in _ALLOWED_PROBE_EXTS:
        return jsonify({"error": "Unsupported file type"}), 400
    uid = uuid.uuid4().hex[:8]
    tmp_path = UPLOAD_DIR / f"{Path(orig_name).stem}_{uid}{Path(orig_name).suffix}"
    f.save(str(tmp_path))

    try:
        info = _extract_ui_meta(tmp_path)
        info["name"] = orig_name
        info["size"] = _fmt_size(str(tmp_path))
        info["server_file"] = tmp_path.name
        return jsonify(info)

    except Exception as e:
        try: tmp_path.unlink()
        except: pass
        return jsonify({"error": str(e)}), 500


def _res_label(w: int, h: int) -> str:
    long_side = max(w, h)
    if long_side >= 7680: return '8K'
    if long_side >= 5120: return '5K'
    if long_side >= 3840: return '4K'
    if long_side >= 2880: return '3K'
    if long_side >= 2700: return '2.7K'
    if long_side >= 2500: return '2.5K'
    if long_side >= 2048: return '2K'
    if long_side >= 1920: return 'FHD'
    if long_side >= 1280: return 'HD'
    return ''


def _quick_meta(path: str):
    """Returns (res, res_label, duration_str) via ffprobe. Returns ('', '', '') on failure."""
    try:
        info = _probe(path)
    except Exception:
        return '', '', ''
    res, lbl = '', ''
    for s in info.get('streams', []):
        if s.get('codec_type') == 'video':
            w, h = s.get('width'), s.get('height')
            if w and h:
                res = f'{w}×{h}'
                lbl = _res_label(w, h)
            break
    dur = ''
    try:
        d = float(info.get('format', {}).get('duration', 0))
        if d > 0:
            m, s = divmod(int(d), 60)
            h, m = divmod(m, 60)
            dur = f'{h}:{m:02d}:{s:02d}' if h else f'{m}:{s:02d}'
    except (ValueError, TypeError):
        pass

    fps = ''
    for s in info.get('streams', []):
        if s.get('codec_type') == 'video':
            fr = s.get('r_frame_rate') or s.get('avg_frame_rate')
            if fr and '/' in fr:
                try:
                    num, den = map(int, fr.split('/'))
                    if den > 0:
                        f = num / den
                        fps = f"{round(f)} FPS"
                except: pass
            break

    return res, lbl, fps, dur


@files_bp.route("/outputs")
def list_outputs():
    out_dir = Path(OUTPUT_DIR)
    entries = []
    for p in out_dir.iterdir():
        if p.is_file() and p.suffix.lower() in _OUTPUT_EXTS:
            try:
                st = p.stat()
                entries.append((p, p.name, st.st_size, st.st_mtime))
            except OSError:
                pass
    entries.sort(key=lambda x: x[3], reverse=True)

    # Probe all files in parallel (max 6 workers)
    meta_map = {}
    with ThreadPoolExecutor(max_workers=6) as ex:
        futs = {ex.submit(_quick_meta, str(p)): name for p, name, _, _ in entries}
        for fut in as_completed(futs):
            meta_map[futs[fut]] = fut.result()

    def _sz(n):
        for unit in ("B", "KB", "MB", "GB"):
            if n < 1024: return f"{n:.1f} {unit}"
            n /= 1024
        return f"{n:.1f} TB"

    def _fmt_dt(mtime):
        return datetime.datetime.fromtimestamp(mtime).strftime("%b %d, %H:%M")

    _AUDIO_EXTS = frozenset({'.mp3', '.m4a', '.aac', '.wav', '.flac', '.ogg'})

    def _card(i, p, name, size, mtime):
        res, lbl, fps, dur = meta_map.get(name, ('', '', '', ''))
        ext      = Path(name).suffix.lower()
        ext_text = ext.lstrip(".").upper()
        is_audio = ext in _AUDIO_EXTS
        ico_cls  = 'au' if is_audio else 'vi'
        
        # Text info (Duration · Date)
        text_parts = [x for x in [dur, _fmt_dt(mtime)] if x]
        detail_html = f'<div class="det">{" · ".join(text_parts)}</div>'
        
        # Chips info (Res, FPS, Size)
        chips = []
        if lbl:
            chips.append(f'<span class="badge {ico_cls}">{lbl}</span>')
        elif res:
            chips.append(f'<span class="badge {ico_cls}">{res}</span>')
            
        if fps:
            chips.append(f'<span class="badge {ico_cls}">{fps}</span>')
            
        chips.append(f'<span class="badge {ico_cls}">{_sz(size)}</span>')
        meta_html = ''.join(chips)
        
        href = f'/outputs/{urllib.parse.quote(name)}'
        stem = Path(name).stem
        return (
            f'<a href="{href}" class="row" style="--d:{i * 40}ms">'
            f'<div class="ic {ico_cls}">{ext_text}</div>'
            f'<div class="info">'

export function removeBatchFile(idx) {
  if (!S.compress.files) return;
  const f = S.compress.files[idx];
  if (!f || S.compress.files.some(fi => fi.status === 'probing')) return;
  S.compress.files.splice(idx, 1);
  S.compress.files.forEach((fi, i) => { fi.id = i; });
  if (S.compress.files.length === 0) {
    const drop  = $('drop-compress');  if (drop)  drop.style.display  = '';
    const vinfo = $('vinfo-compress'); if (vinfo) vinfo.style.display = 'none';
    return;
  }
  renderBatchList();
  _updateBatchStartBtn();
}

export async function probeBatchFile(index) {
  const fObj = S.compress.files[index];
  fObj.status = 'probing';
  renderBatchList();

  const loadEl = $('vinfo-loading-compress');
  loadEl.classList.remove('hidden');
  loadEl.textContent = `Uploading file ${index+1}/${S.compress.files.length} (0%)...`;

  return new Promise((resolve) => {
    const fd = new FormData();
    fd.append('file', fObj.file);
    const xhr = new XMLHttpRequest();
    xhr.open('POST', '/api/probe', true);
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) loadEl.textContent = `Uploading file ${index+1} (${pct}%)...`;
        else loadEl.textContent = `Reading metadata for file ${index+1}...`;
      }
    };
    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          if (data.server_file) {
            fObj.server_file = data.server_file;
            fObj.status = 'ready';
            fObj.meta = data;
          } else {
            fObj.status = 'error';
            fObj.errMsg = data.error || 'No server_file';
          }
        } catch(e) {
          fObj.status = 'error';
          fObj.errMsg = 'JSON Parse Error';
        }
      } else {
        fObj.status = 'error';
        try {
          const errData = JSON.parse(xhr.responseText);
          fObj.errMsg = errData.error || `HTTP ${xhr.status}`;
        } catch(e) {
          fObj.errMsg = `HTTP ${xhr.status}: ${xhr.statusText}`;
        }
      }
      loadEl.classList.add('hidden');
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.onerror = () => {
      fObj.status = 'error';
      fObj.errMsg = 'Network disconnected';
      loadEl.classList.add('hidden');
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.send(fd);
  });
}

export async function cancelSingleJob(jid, uid) {
  if (!(await customConfirm('Are you sure you want to cancel this video compression?'))) return;
  try {
    const cBtn = $('btn-cancel-' + uid);
    if (cBtn) cBtn.style.display = 'none';
    await fetch('/api/cancel/' + jid, { method: 'POST' });
  } catch(e) {}
}

export async function startCompress() {
  if (!S.compress.files || S.compress.files.length === 0) {
    showErr('compress', 'Please select video(s)!'); return;
  }

  const readyFiles = S.compress.files.filter(f => f.status === 'ready');

    showErr('compress', 'No files are ready or currently uploading!'); return;
  }

  $('err-compress').textContent = '';
  $('err-compress').classList.remove('show');

  S.compress.encLabel = $('cs-encoder-val').textContent;
  S.compress.crf      = $('crf-value').value;

  setBusy('compress', true);

  const enc = $('enc-compress').value;
  const fmt = $('fmt-compress').value;
  const crf = $('crf-value').value;

  S.compress.jobs = {};
  S.compress.batchStartTime = Date.now();
  const timerEl = $('batch-elapsed-timer');
  if (timerEl) timerEl.textContent = '00:00';

  const progCard = $('prog-compress');
  progCard.classList.remove('hidden');
  progCard.style.background = `linear-gradient(180deg, rgba(var(--purple-rgb), 0.18) 0%, rgba(var(--purple-rgb), 0.00) 100%)`;
  progCard.style.border     = `1px solid rgba(var(--purple-rgb), 0.35)`;
  progCard.style.boxShadow  = `0 16px 40px rgba(0, 0, 0, 0.45)`;
  $('res-compress').classList.add('hidden');
  $('btn-area-compress').classList.add('hidden');

  const btnChange = $('btn-change-compress');
  if (btnChange) btnChange.style.display = 'none';

  const progCountEl = $('batch-progress-count');
  if (progCountEl) progCountEl.textContent = `(0/${readyFiles.length} videos)`;

  const pList = $('batch-progress-list');
  pList.innerHTML = '';

  pList.innerHTML = readyFiles.map((f, i) => {
    const id = 'prog-item-' + i;
    return `
      <div id="${id}" style="padding:12px 0; border-bottom:1px solid rgba(255,255,255,0.05); background:transparent; position:relative;">
        <div style="font-size:13px; margin-bottom:6px; font-weight:500; padding-right:25px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis; color:var(--purple-l);" class="mob-title">
          <span style="margin-right:4px;">${i+1}.</span>${f.file.name}
        </div>
        <button id="btn-cancel-${id}" style="position:absolute;top:10px;right:10px;background:none;border:none;cursor:pointer;opacity:0.7;display:none;color:var(--text-3);padding:0;line-height:1" title="Cancel this process"><svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg></button>
        <div style="font-size:12px; color:var(--text-3); margin-bottom:4px;" id="msg-${id}">Initializing...</div>
        <div class="prog-track"><div class="prog-fill" id="bar-${id}"></div></div>
        <div class="prog-meta" style="margin-top:4px"><span id="pct-${id}">0%</span><span id="eta-${id}"></span></div>
      </div>
    `;
  }).join('');

  const clientBatchId = Date.now().toString(36) + Math.random().toString(36).substr(2, 5);
  for (let i = 0; i < readyFiles.length; i++) {
    const fObj = readyFiles[i];
    const id   = 'prog-item-' + i;

    const fd = new FormData();
    fd.append('client_batch_id', clientBatchId);
    fd.append('server_file', fObj.server_file);
    fd.append('original_name', fObj.file.name);
    fd.append('encoder', enc);
    fd.append('format', fmt);
    fd.append('crf', crf);
    if (muteAudio)          fd.append('mute', '1');
    fd.append('preserve_dv', preserveDv ? '1' : '0');  // always explicit — server default is now "1"
    if (crfPreset === 'auto') fd.append('crf_auto', '1');

    const ui_meta = fObj.meta ? {
      codec: fObj.meta.codec, res: fObj.meta.res, duration: fObj.meta.duration,
      fps: fObj.meta.fps, audio: fObj.meta.audio, size: fObj.size,
      is_dolby_vision: fObj.meta.is_dolby_vision || false,
      dv_profile: fObj.meta.dv_profile || 0,
      is_dual_camera: fObj.meta.is_dual_camera || false,
      is_action_mode: fObj.meta.is_action_mode || false,
    } : {};
    fd.append('ui_meta', JSON.stringify(ui_meta));

    try {
      const res  = await fetch('/api/compress', { method: 'POST', body: fd });
      const data = await res.json();
      const fileIdx = S.compress.files.indexOf(fObj);
      if (res.ok && data.job_id) {
        S.compress.jobs[data.job_id] = { ui_id: id, file: fObj.file.name, fileIdx, done: false };
        const cBtn = $('btn-cancel-' + id);
        if (cBtn) {
          cBtn.style.display = 'block';
          cBtn.onclick = () => cancelSingleJob(data.job_id, id);
        }
      } else {
        S.compress.jobs[id] = { ui_id: id, file: fObj.file.name, fileIdx, done: true, result: { status: 'error', message: data.error || 'Rejected' } };
        const progItem = $(id);
        if (progItem) {
          progItem.innerHTML = _buildJobResultHTML(id, fObj.file.name, S.compress.jobs[id].result, `<span style="margin-right:4px;">${i + 1}.</span>`);
        }
      }
    } catch(e) {
      sv('msg-' + id, 'Connection error');
    }
  }

}

// Called by the central sync loop with pre-fetched job data for a single job.
export function updateBatchJobFromData(jid, data) {
  const jobMeta = S.compress.jobs[jid]; if (!jobMeta) return;
  const uid = jobMeta.ui_id;
  const pct = data.progress || 0;

  const elPct = $('pct-' + uid);
  const elMsg = $('msg-' + uid);
  const elBar = $('bar-' + uid);
  const elEta = $('eta-' + uid);

  if (elPct) elPct.innerText = pct + '%';
  if (elMsg) elMsg.innerText = data.message || 'Processing...';
  if (elBar) elBar.style.width = pct + '%';

  if (pct > 0 && pct < 100) {
    // Prefer server-provided start time (survives page refresh).
    const serverT0 = data.encode_start_time;
    if (serverT0) jobMeta.t0 = serverT0;
    else if (!jobMeta.t0) jobMeta.t0 = Date.now() / 1000;
    if (pct >= 5 && jobMeta.t0) {
      const ela = (Date.now() / 1000) - jobMeta.t0;
      const etaSec = Math.round((ela / pct) * (100 - pct));
      let etaStr = '';
      if (etaSec >= 60) etaStr = `~${Math.floor(etaSec/60)}m ${etaSec%60}s left`;
      else if (etaSec > 0) etaStr = `~${etaSec}s left`;
      if (elEta) elEta.innerText = etaStr;
    } else {
      if (elEta) elEta.innerText = '';
    }
  } else {
    // pct dropped back to 0 (e.g. OOM retry) — reset t0 so ETA restarts clean
    if (pct === 0) jobMeta.t0 = null;
    if (elEta) elEta.innerText = '';
  }

  if (data.status === 'done' || data.status === 'error') {
    jobMeta.done   = true;
    jobMeta.result = data;
    const btnCancel = $('btn-cancel-' + uid);
    if (btnCancel) btnCancel.style.display = 'none';
    const progItem = $(uid);
    if (progItem) {
      progItem.style.background   = 'transparent';
      progItem.style.padding      = '12px 0';
      progItem.style.borderBottom = '1px solid rgba(255,255,255,0.05)';
      const idx    = jobMeta.fileIdx !== undefined ? jobMeta.fileIdx
                   : S.compress.files ? S.compress.files.findIndex(f => f.file.name === jobMeta.file) : -1;
      const idxStr = idx >= 0 ? `<span style="margin-right:4px;">${idx + 1}.</span>` : '';
      progItem.innerHTML = _buildJobResultHTML(jid, jobMeta.file, data, idxStr);
    }
  }
}

export function updateBatchProgressCount() {
  if (!S.compress.jobs) return;
  const allJobIds = Object.keys(S.compress.jobs);
  const doneCount = allJobIds.filter(jid => S.compress.jobs[jid].done).length;
  const progCountEl = $('batch-progress-count');
  if (progCountEl && allJobIds.length > 0)
    progCountEl.textContent = `(${doneCount}/${allJobIds.length} videos)`;
  const timerEl = $('batch-elapsed-timer');
  if (timerEl && S.compress.batchStartTime) {
    const totalSec = Math.round((Date.now() - S.compress.batchStartTime) / 1000);
    const m = Math.floor(totalSec / 60), s = totalSec % 60;
    timerEl.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  }
}

export function checkCompressBatchDone() {
  if (!S.compress.jobs) return;
  const allJobIds = Object.keys(S.compress.jobs);
  if (!allJobIds.length) return;
  if (!allJobIds.every(jid => S.compress.jobs[jid].done)) return;
  // All done — finalize
  setBusy('compress', false);
  updateBatchProgressCount();  // final timer snapshot
  renderBatchResult();
  S.compress.jobs = null;  // clear so sync loop stops watching
}

export function getActiveCompressJobIds() {
  if (!S.compress.jobs) return [];
  return Object.keys(S.compress.jobs).filter(jid => !S.compress.jobs[jid].done);
}

  const doneCount = Object.values(S.compress.jobs).filter(j => j.result && j.result.status === 'done').length;
  const errCount  = jobCount - doneCount;

  let theme = 'green', themeL = 'green-l';
  let titleHtml = `<i data-lucide="check-circle-2" style="width:16px;height:16px;vertical-align:-2px;color:var(--green)"></i> All tasks completed!`;

  if (doneCount === 0 && errCount > 0) {
    theme = 'red'; themeL = 'red-l';
    titleHtml = `<i data-lucide="x-circle" style="width:16px;height:16px;vertical-align:-2px;color:var(--red)"></i> Batch failed!`;
  } else if (errCount > 0) {
    theme = 'yellow'; themeL = 'yellow-l';
    titleHtml = `<i data-lucide="alert-triangle" style="width:16px;height:16px;vertical-align:-2px;color:var(--yellow)"></i> Batch finished with errors`;
  }

  const badge = $('res-compress').querySelector('.res-badge');

  resBox.style.background = `linear-gradient(180deg, rgba(var(--${theme}-rgb), 0.18) 0%, rgba(var(--${theme}-rgb), 0.00) 100%)`;
  resBox.style.border     = `1px solid rgba(var(--${theme}-rgb), 0.35)`;
  resBox.style.boxShadow  = `0 16px 40px rgba(0, 0, 0, 0.45)`;

  let totalTimeHtml = '';
  if (S.compress.batchStartTime) {
    const totalMs  = Date.now() - S.compress.batchStartTime;

    const mins     = Math.floor(totalSec / 60), secs = totalSec % 60;
    const timeStr  = mins > 0 ? `${mins}m ${secs}s` : `${totalSec}s`;
    totalTimeHtml  = `
      <div style="margin-top:10px; padding:10px 14px; background:rgba(var(--${theme}-rgb),0.06); border:1px solid rgba(var(--${theme}-rgb),0.2); border-radius:var(--radius-sm); display:flex; align-items:center; justify-content:space-between; flex-wrap:wrap; gap:8px;">

    totalEl = document.createElement('div');
    totalEl.id = 'batch-total-time';
    badge.insertAdjacentElement('afterend', totalEl);
  }
  totalEl.innerHTML = totalTimeHtml;
  if (window.lucide) lucide.createIcons();

  const itemStyle = 'padding:12px 0; border-bottom:1px solid rgba(255,255,255,0.05); font-size:13px;';
  $('batch-result-list').innerHTML = (S.compress.files || [])
    .map((f, idx) => {

      if (!jobEntry) return '';

      return `<div style="${itemStyle}">${_buildJobResultHTML(jid, job.file, job.result, idxStr)}</div>`;
    })
    .join('');
}

# Threads per concurrent CPU H265/H264 job.
# When _cpu_h_slots jobs run in parallel, each gets an equal share of cores.
# Example: 12 cores / 2 slots → 6 threads/job → 2 jobs × 6 = 12 threads (full CPU),
# but RAM per job drops from ~3.5 GB to ~1.5 GB → safe for concurrent 4K batches.
_cpu_h_per_job = max(4, _cpu_cores // max(1, _cpu_h_slots))
_X265_LOW_RAM_MB    = 4000   # threshold for 4K jobs
_X265_LOW_RAM_1080P = 1500   # threshold for 1080p jobs

def _build_x265_params(width: int = 0, height: int = 0) -> list:
    """
    Build -x265-params string for libx265.

    Concurrency strategy for 4K:
    - _cpu_h_slots >= 2: partition thread pool so each concurrent job uses
      _cpu_h_per_job threads instead of all cores. Total CPU remains 100%,
      but RAM per job halves → no OOM on concurrent 4K batches.
    - _cpu_h_slots == 1 (single job): use all cores freely; apply RAM-based
      limits only if the system is genuinely low on memory.

    This function is called AFTER acquiring the CPU semaphore (post-queue-wait),
    so free_ram_mb() reflects actual memory pressure at encode start time.
    """
    params  = "psy-rd=1.0:psy-rdoq=0.5:aq-mode=3:aq-strength=0.8:pme=1"
    is_4k   = (width >= 3840 or height >= 2160)
    ram_now = free_ram_mb()

    if is_4k and _cpu_h_slots >= 2:
        # Partition: each job gets _cpu_h_per_job threads.
        # frame-threads ≈ threads/4, capped at 4 (diminishing returns above).
        ft = max(1, min(4, _cpu_h_per_job // 4))
        params += (f":pools={_cpu_h_per_job}:frame-threads={ft}"
                   ":lookahead-threads=1:rc-lookahead=20:bframes=3")
    elif ram_now < _X265_LOW_RAM_MB:
        # Single-job or low-RAM fallback: apply conservative limits.
        if is_4k:
            params += ":pools=1:frame-threads=1:lookahead-threads=1:rc-lookahead=20:bframes=3"
        elif ram_now < _X265_LOW_RAM_1080P:
            params += ":pools=2:lookahead-threads=1:rc-lookahead=20"
    return ["-x265-params", params]

def _auto_crf(enc: str, width: int, height: int) -> int:
    """Return resolution-aware CRF for Auto mode.
    Uses Highest quality base (CRF_DEFAULTS[enc][0]) + resolution offset,
    so perceptual quality stays consistent across resolutions in a batch.
    av1_amf uses QVBR (inverted scale) → offset is subtracted.
    """
    base = CRF_DEFAULTS.get(enc, (24, 28, 32, 51))[0]
    px   = (width or 0) * (height or 0)
    for limit, off in CRF_AUTO_OFFSETS:
        if px < limit:
            return base - off if enc == "av1_amf" else base + off
    return base  # fallback — unreachable in practice (float('inf') always matches)

def _compress_dovi_pipeline(job_id, src, dst, enc, real_enc, enc_extra, crf,
                              mute, best_audio_idx, target_kbps, total_s, ppath,
                              meta_stream_indices=None, x265_params=None):
    """
    Dolby Vision preserving pipeline — 9 steps.

    Key insight: B-frame PTS/DTS live in the MOV container's CTTS table, not in
    the raw HEVC bitstream. By encoding to a full temp.mov first (so CTTS is
    written correctly), then binary-patching its video samples with RPU-injected
    data while keeping the CTTS/stts/stss tables intact, we preserve smooth
    B-frame playback on iOS AND keep Dolby Vision — without the -bf 0 penalty.
    """
    tmp = Path(TEMP_DIR) / f"dovi_{job_id}"
    tmp.mkdir(parents=True, exist_ok=True)
    try:
        raw_hevc  = tmp / "raw.hevc"     # copy of source video (for RPU extraction)
        rpu_bin   = tmp / "rpu.bin"      # extracted DV RPU metadata
        temp_mov  = tmp / "encoded.mov" # full encoded output (video+audio, correct timestamps)
        enc_hevc  = tmp / "enc.hevc"     # Annex-B HEVC extracted from temp_mov
        dv_hevc   = tmp / "dv.hevc"      # enc.hevc with RPU injected

        # Step 1: Extract raw HEVC from source (Annex-B, for RPU extraction)
        _job_msg(job_id, "Extracting Dolby Vision stream...", 2)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(src),
                            "-c:v", "copy", "-an", "-f", "hevc", str(raw_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 2: Extract DV RPU metadata
        _job_msg(job_id, "Extracting DV RPU metadata...", 5)
        r = subprocess.run([DOVI_TOOL_BIN, "extract-rpu",
                            "-i", str(raw_hevc), "-o", str(rpu_bin)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"RPU extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 3: Encode to temp.mov — full output with B-frames, audio, metadata.
        # B-frames are ALLOWED here: the container (temp.mov) correctly stores their
        # PTS/DTS in the CTTS box. We'll preserve this CTTS in the final output.
        _job_msg(job_id, f"Encoding DV base layer ({real_enc})...", 8)
        with _jobs_lock:
            jobs[job_id]["encode_start_time"] = time.time()
        x265_mem = (x265_params or []) if real_enc == "libx265" else []
        audio_map = [] if mute else ["-map", f"0:a:{best_audio_idx}"]
        audio_enc = [] if mute else ["-c:a", "aac", "-b:a", f"{target_kbps}k",
                                     "-ar", "48000", "-ac", "2"]
        encode_cmd = ([FFMPEG_BIN, "-noautorotate"] + _hwaccel_args(real_enc) +
                      ["-i", str(src), "-map", "0:v:0"] + audio_map +
                      ["-c:v", real_enc] +
                      _quality_args(enc, crf) + enc_extra + x265_mem +
                      ["-tag:v", "hvc1"] + audio_enc +
                      ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"] +
                      ["-movflags", "use_metadata_tags+faststart"] +
                      ["-y", "-progress", ppath, "-stats_period", "0.5", "-nostats",
                       str(temp_mov)])
        rc, is_oom = _run_progress(encode_cmd, job_id, total_s, ppath, pct_min=8)
        if rc != 0:
            return rc, is_oom

        if _dovi_cancelled(job_id): return -1, False

        # Step 4: Extract Annex-B HEVC from temp_mov (loses container timestamps,
        # but that's OK — we will restore them via binary patching in Step 7).
        _job_msg(job_id, "Extracting encoded HEVC stream...", 97)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(temp_mov),
                            "-c:v", "copy", "-bsf:v", "hevc_mp4toannexb",
                            "-an", "-f", "hevc", str(enc_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract from temp failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 5: Inject RPU NAL units into the encoded HEVC bitstream
        _job_msg(job_id, "Injecting Dolby Vision metadata...", 98)
        r = subprocess.run([DOVI_TOOL_BIN, "inject-rpu",
                            "-i", str(enc_hevc), "--rpu-in", str(rpu_bin),
                            "-o", str(dv_hevc)],
                           capture_output=True, timeout=300)
        if r.returncode != 0:
            raise RuntimeError(f"RPU inject failed: {r.stderr.decode(errors='replace')[-1000:]}")
        if not dv_hevc.exists() or dv_hevc.stat().st_size < 1024:
            raise RuntimeError(f"RPU inject produced empty output")

        # Step 6+7: Streaming DV mux — parse dv.hevc + rebuild container.
        # Peak RAM: O(moov_size + one_frame_size) regardless of video length.
        _job_msg(job_id, "Rebuilding Dolby Vision container...", 98)
        if not _dovi_mux_streaming(str(temp_mov), str(dv_hevc), str(dst)):
            raise RuntimeError("DV container mux failed")

        # Step 8: Transplant mebx (Core Media Metadata) tracks from source into dst.
        # These tracks carry dual-camera composition and Action Mode timed metadata
        # that iOS Photos uses to display feature icons and "Camera mặt trước" info.
        # FFmpeg corrupts mebx hdlr/stsd; _fix_mebx_tracks patches them back.
        if meta_stream_indices:
            _job_msg(job_id, "Copying metadata tracks...", 99)
            mebx_out = tmp / "with_mebx.mov"
            mebx_maps = []
            for idx in meta_stream_indices:
                mebx_maps += ["-map", f"1:{idx}"]
            mebx_cmd = ([FFMPEG_BIN, "-y",
                         "-i", str(dst), "-i", str(src),
                         "-map", "0", "-c", "copy"] +
                        mebx_maps +
                        ["-c:d", "copy",
                         "-tag:v", "hvc1",
                         "-movflags", "use_metadata_tags+faststart",
                         str(mebx_out)])
            try:
                r = subprocess.run(mebx_cmd, capture_output=True, timeout=120)
            except subprocess.TimeoutExpired:
                r = None  # graceful: keep dst as-is (core DV already written)
            if r is not None and r.returncode == 0 and mebx_out.exists() and mebx_out.stat().st_size > 1024:
                _fix_mebx_tracks(str(mebx_out))
                os.replace(str(mebx_out), str(dst))

        # Step 9: Copy dvvC/dvcC atom from source → output
        _job_msg(job_id, "Restoring Dolby Vision container signaling...", 99)
        _inject_dvcc(str(src), str(dst))

        return 0, False

    except RuntimeError as e:
        with _jobs_lock:
            jobs[job_id].update({"status": "error", "message": str(e)})
        return -1, False
    finally:
        shutil.rmtree(tmp, ignore_errors=True)


@job_guard
def _compress_job(job_id, src: Path, enc: str, crf: int,
                  out_ext: str, mute: bool = False, preserve_dv: bool = False,
                  crf_auto: bool = False):
    with _jobs_lock:
        if job_id not in jobs:
            return
        job = jobs[job_id]

    if not src.exists():
        with _jobs_lock:
            job["status"]  = "error"
            job["message"] = "Source file not found. Please reselect."
        return

    family = _get_codec_family(enc)
    t0     = None

    src_name = job.get("src_name", "video")
    src_stem = re.sub(r"[^\w\-. ]", "_", Path(src_name).stem)

    av1_warning = ""
    if family == "av1" and out_ext == ".mov":
        out_ext = ".mp4"
        av1_warning = " (auto-converted to MP4)"

    supported = CONTAINER_CODEC_SUPPORT.get(out_ext, ["h264"])
    if family not in supported:
        if out_ext == ".webm":
            enc = "libaom-av1"
            _msg = "⚠️ .WebM only supports AV1 → auto-switching to libaom-AV1"
        else:
            enc = "libx264"
            _msg = f"⚠️ {out_ext.upper()} does not support {family.upper()} → switching to H.264"
        family = _get_codec_family(enc)
        with _jobs_lock:
            job["message"] = _msg

    dst = _unique_output_path(src_stem, out_ext)
    if not _check_path_len_job(dst, job_id):
        try: dst.unlink()
        except: pass
        try: src.unlink()
        except: pass
        return  # ppath not yet created at this point

    real_enc, enc_extra = _get_real_enc_extra(enc)

    pfd, ppath = tempfile.mkstemp(suffix=".txt", prefix="vs_")
    os.close(pfd)

    probe_data     = _probe(str(src))
    total_s        = float(probe_data.get("format", {}).get("duration") or 0)

    # Extract video dimensions and rotation for resolution-aware params and metadata sync
    video_w = video_h = 0
    src_rotation = 0
    for _s in probe_data.get("streams", []):
        if _s.get("codec_type") == "video":
            video_w = int(_s.get("width",  0) or 0)
            video_h = int(_s.get("height", 0) or 0)
            src_rotation = _get_rotation(_s)
            break
    if crf_auto:
        crf = _auto_crf(enc, video_w, video_h)
        with _jobs_lock:
            if job_id in jobs:
                jobs[job_id]["effective_crf"] = crf
                jobs[job_id]["crf_auto"] = True

    # x265_params will be rebuilt inside the encode loop (after queue wait)
    # so it reads free RAM at the moment encoding actually starts — not when
    # multiple siblings may not have loaded yet.
    _is_x265   = (real_enc == "libx265")
    x265_params = []  # populated inside retry loop for libx265; empty for all others

    # Dolby Vision detection
    is_dv, dv_profile = _get_dv_info(probe_data)
    use_dovi = is_dv and _DOVI_AVAILABLE and preserve_dv

    # Store source metadata snapshot
    src_exif = _get_exif_snapshot(str(src))
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["src_exif"] = src_exif

    best_audio_idx = 0
    best_audio_br  = 0
    _apac_idx      = None  # last-resort fallback if apac is the only stream
    audio_streams_raw = [
        s for s in probe_data.get("streams", [])
        if s.get("codec_type") == "audio" and s.get("codec_name") and s.get("codec_name") != "none"
    ]
    audio_streams = list(enumerate(audio_streams_raw))
    for stream_idx, s in audio_streams:
        # Prefer decodable AAC (mp4a) over Apple Lossless/spatial (apac).
        # apac is a 3-4ch spatial stream: re-encoding to stereo AAC without proper
        # downmix causes iOS playback stutter (timing desync with mebx tracks).
        # Track apac separately — only use if no other decodable stream exists.
        codec_tag = s.get("codec_tag_string", "").lower()
        if codec_tag == "apac":
            if _apac_idx is None:
                _apac_idx = stream_idx
            continue
        br = 0
        raw_br = s.get("bit_rate")
        if raw_br:
            try: br = int(raw_br)
            except: pass
        if br < 10000:
            tags = s.get("tags", {})
            bps  = tags.get("BPS") or tags.get("BPS-eng")
            if bps:
                try: br = int(bps)
                except: pass
        if br < 10000:
            ch = s.get("channels", 0)
            sr = s.get("sample_rate", 0)
            if ch and sr:
                br = ch * int(sr) * 16
        if br > best_audio_br:
            best_audio_br  = br
            best_audio_idx = stream_idx

    # If no decodable stream found, fall back to apac.
    # best_audio_br stays 0 → target_kbps correctly falls back to 192k default.
    if best_audio_br == 0 and _apac_idx is not None:
        best_audio_idx = _apac_idx

    # Detect Apple metadata tracks (mebx, tmcd) — carry iPhone feature signaling
    # such as Action Mode, Dual Camera, GPS track. Copy them as-is to preserve
    # iOS Photos feature badges without re-encoding.
    meta_stream_indices = []
    for s in probe_data.get("streams", []):
        if s.get("codec_type") in ("data", "tmcd"):
            handler = s.get("tags", {}).get("handler_name", "")
            tag = s.get("codec_tag_string", "")
            # mebx = CoreMedia timed metadata; tmcd = timecode
            if tag in ("mebx", "tmcd") or any(h in handler for h in _META_HANDLER_TAGS):
                # tmcd with time-of-day timecode (hours > 0) causes DTS overflow in
                # movenc.c during faststart second pass because the tmcd track has its
                # own internal timebase that -video_track_timescale does NOT affect.
                # The timecode value is already preserved in the video stream's metadata
                # tag, so dropping the tmcd track here loses no meaningful information.
                # mebx tracks (iOS feature badges) are always kept.
                if tag == "tmcd" or "time code" in handler.lower():
                    tc = s.get("tags", {}).get("timecode", "00:00:00:00")
                    try:
                        if int(tc.split(":")[0]) > 0:
                            continue  # skip: time-of-day timecode risks INT32 overflow
                    except (ValueError, IndexError):
                        pass
                meta_stream_indices.append(s["index"])

    cmd = [FFMPEG_BIN, "-noautorotate"] + _hwaccel_args(real_enc) + ["-i", str(src), "-map", "0:v:0"]

    if not mute and audio_streams:
        audio_map = (f"0:a:{best_audio_idx}"
                     if len(audio_streams) > 1 else "0:a:0")
        cmd += ["-map", audio_map]

    for idx in meta_stream_indices:
        cmd += ["-map", f"0:{idx}"]

    cmd += ["-c:v", real_enc]
    cmd += _quality_args(enc, crf)
    cmd += enc_extra
    if family == "h265" and out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-tag:v", "hvc1"]

    if meta_stream_indices:
        cmd += ["-c:d", "copy"]

    target_kbps = 192
    if not mute and audio_streams:
        target_kbps = max(128, min(320, int(best_audio_br / 1000))) if best_audio_br > 0 else 192
        audio_br_str = f"{target_kbps}k"
        cmd += [
            "-c:a", "aac",
            "-b:a", audio_br_str,
            "-ar",  "48000",
            "-ac",  "2",
        ]

    cmd += ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"]
    if not mute and audio_streams:
        cmd += ["-map_metadata:s:a", "0:s:a"]

    if out_ext in (".mp4", ".mov", ".m4v"):
        if meta_stream_indices:
            # -video_track_timescale 600: keep output video timebase at 600
            # (matching Apple QuickTime standard) instead of FFmpeg's default 15360.
            # Prevents DTS overflow (INT32_MAX assertion) in movenc.c during
            # faststart second pass when tmcd timecode starts late in the day
            # (e.g., Blackmagic Camera time-of-day timecode like 18:34:52).
            # At 15360 tbn: 18:34:52 × 15360 = ~1.03B; with edit list offsets
            # can exceed INT32_MAX=2,147,483,647.  At 600 tbn: ~40M — safe.
            cmd += ["-video_track_timescale", "600"]
        cmd += ["-movflags", "use_metadata_tags+faststart"]
    elif out_ext != ".mkv":
        cmd += ["-movflags", "+faststart"]

    cmd += ["-y", "-progress", ppath,
            "-stats_period", "0.5", "-nostats"]
    _dst_arg = [str(dst)]

    oom_retries = 0
    while True:
        with _jobs_lock:
            job["status"]  = "waiting"
            job["message"] = "Waiting in queue..."
        with _pick_sem(real_enc):
            with _jobs_lock:
                if job.get("cancelled"): return
            t0 = time.time()
            # Rebuild x265 thread limits now (after queue wait) so free_ram_mb()
            # reflects the actual pressure from other running jobs.
            if _is_x265:
                x265_params = _build_x265_params(video_w, video_h)
            with _jobs_lock:
                job["status"]  = "processing"
                crf_note = f" — CRF {crf} (Auto)" if crf_auto else ""
                job["message"] = f"Encoding ({real_enc}){crf_note}{av1_warning}{' — no audio' if mute else ''}..."
                job["encode_start_time"] = t0
            try:
                if use_dovi:
                    rc, is_oom = _compress_dovi_pipeline(
                        job_id, src, dst, enc, real_enc, enc_extra, crf,
                        mute, best_audio_idx, target_kbps, total_s, ppath,
                        meta_stream_indices, x265_params=x265_params)
                else:
                    rc, is_oom = _run_progress(cmd + x265_params + _dst_arg, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            with _jobs_lock:
                if job.get("cancelled"): return
            oom_retries += 1
            if oom_retries >= 3:
                if use_dovi:
                    # DV pipeline OOM → fall back to regular encode (no DV)
                    use_dovi = False
                    oom_retries = 0
                    with _jobs_lock:
                        job["status"]  = "waiting"
                        job["message"] = "⚠️ DV preservation failed (low RAM) — encoding as HLG..."
                    try: dst.unlink()
                    except: pass
                    time.sleep(2)
                else:
                    with _jobs_lock:
                        job.update({"status": "error", "message": "Insufficient RAM — failed after 3 retries. Too many 4K jobs running at once."})
                    try: dst.unlink()
                    except: pass
                    return
            else:
                with _jobs_lock:
                    if job.get("cancelled"): return
                    # Distinguish GPU VRAM exhaustion from CPU RAM exhaustion
                    ram_msg = "RAM" if real_enc in ("libx265", "libx264", "libsvtav1", "libaom-av1") else "GPU"
                    job["status"]  = "waiting"
                    job["message"] = f"{ram_msg} full, retrying ({oom_retries}/3)..."
                time.sleep(3)
                with _jobs_lock:
                    if job.get("cancelled"): return

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        # _fix_mebx_tracks already runs inside _compress_dovi_pipeline (step 8)
        # for DV videos, so skip the redundant second pass.
        if meta_stream_indices and not use_dovi:
            _fix_mebx_tracks(str(dst))
        _sync_meta(str(src), str(dst), src_rotation)

        dst_exif = _get_exif_snapshot(str(dst))
        with _jobs_lock:
            if job_id in jobs:
                jobs[job_id]["dst_exif"] = dst_exif

        ela      = time.time() - t0 if t0 else 0
        enc_time = _fmt_time(ela)

        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = out_duration = "N/A"

        try:
            d = float(out_data.get("format", {}).get("duration", 0))
            if d > 0:
                out_duration = f"{int(d//60)}:{int(d%60):02d}"
        except: pass

        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                w, h = s.get('width', '?'), s.get('height', '?')
                rotation = _get_rotation(s)
                if rotation in (90, 270) and w != '?' and h != '?':
                    out_res = f"{h}x{w}"
                else:
                    out_res = f"{w}x{h}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break

        if mute:
            out_audio = "Muted"
        else:
            for s in out_data.get("streams", []):
                if s.get("codec_type") == "audio":
                    ac     = s.get("codec_name", "N/A").upper()
                    br     = s.get("bit_rate")
                    br_int = int(br) if br else 0
                    if br_int < 10000:
                        tags   = s.get("tags", {})
                        br_tag = tags.get("BPS") or tags.get("BPS-eng")
                        if br_tag:
                            try: br_int = int(br_tag)
                            except: pass
                    if br_int < 10000:
                        br_int = target_kbps * 1000
                    brs       = f" {br_int//1000}kbps" if br_int > 10000 else ""
                    out_audio = f"{ac}{brs}"
                    break

        result = {
            "status":        "done",
            "progress":      100,
            "message":       "Completed!",
            "output":        str(dst),
            "sz_before":     _fmt_size(str(src)),
            "sz_after":      _fmt_size(str(dst)),
            "encode_time":   enc_time,
            "out_res":       out_res,
            "out_fps":       out_fps,
            "out_audio":     out_audio,
            "out_codec":     real_enc,
            "out_duration":  out_duration,
            "src_is_dv":     is_dv,
            "src_dv_profile":dv_profile,
            "dst_is_dv":     _detect_dolby_vision(out_data),
        }
        if family == "av1" and av1_warning:
            result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 does not support MOV format — auto-converted to MP4"
        with _jobs_lock:
            job.update(result)
            log_name = job.get("log_file")
        if log_name:
            try: (Path(OUTPUT_DIR) / log_name).unlink(missing_ok=True)
            except: pass
    else:
        with _jobs_lock:
            if not job.get("cancelled") and job.get("status") != "error":
                job.update({"status": "error", "message": "Compression failed!"})
        try: dst.unlink()
        except: pass

    try: src.unlink()
    except: pass
    try: os.unlink(ppath)   # no-op if _run_progress already cleaned it
    except: pass
