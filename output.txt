class _AdaptiveCPUSemaphore:
    """
    Limits CPU-encoder job parallelism based on:
    1. Hard slot count (max concurrent jobs)
    2. Free system RAM (must keep > keep_ram_mb free before starting a new job)

    libx265 encoding 4K 10-bit 4:2:2 needs ~3 GB RAM per instance.
    Without this guard, a batch of 4K files causes concurrent malloc failures
    and "memory allocation failure, aborting encode" from x265.
    """
    def __init__(self, max_slots: int, keep_ram_mb: int = 2500):
        self._max        = max_slots
        self._keep_ram   = keep_ram_mb
        self._lock       = threading.Lock()
        self._count      = 0
        self._last_acq   = 0.0

    def acquire(self):
        first = True
        while True:
            if not first:
                time.sleep(2.0)
            first = False
            # Stagger concurrent acquires by 0.5 s to avoid thundering-herd
            with self._lock:
                elapsed = time.time() - self._last_acq
            if elapsed < 0.5:
                time.sleep(0.5 - elapsed)
            with self._lock:
                if self._count < self._max and free_ram_mb() > self._keep_ram:
                    self._count += 1
                    self._last_acq = time.time()
                    return

    def release(self):
        with self._lock:
            self._count = max(0, self._count - 1)

    def __enter__(self):    self.acquire()
    def __exit__(self, *_): self.release()

class _AdaptiveGPUSemaphore:
    """
    Limits GPU job parallelism based on:
    1. Hard slot count (max concurrent jobs)
    2. Free VRAM (> 10% total VRAM)
    3. System RAM (> 2GB reserved)
    """
    def __init__(self, max_slots=4, keep_vram_pct=0.10):
        self._max          = max_slots
        self._keep_vram_pct = keep_vram_pct
        self._lock         = threading.Lock()
        self._count        = 0
        self._last_acquire = 0.0

    def _vram_stats(self):
        try:
            r = subprocess.run(
                ["nvidia-smi",
                 "--query-gpu=memory.free,memory.total",
                 "--format=csv,noheader,nounits"],
                capture_output=True, text=True, timeout=2
            )
            if r.returncode == 0:
                free_l, total_l = [], []
                for line in r.stdout.strip().splitlines():
                    parts = line.split(',')
                    if len(parts) >= 2 and parts[0].strip().isdigit() and parts[1].strip().isdigit():
                        free_l.append(int(parts[0].strip()))
                        total_l.append(int(parts[1].strip()))
                if free_l and total_l:
                    return min(free_l), min(total_l)
        except Exception:
            pass
        return None, None

    def acquire(self):
        first = True
        while True:
            if not first:
                time.sleep(1.5)
            first = False
            with self._lock:
                elapsed = time.time() - self._last_acquire
            if elapsed < 0.5:
                time.sleep(0.5 - elapsed)

            vram_free, vram_total = self._vram_stats()

            with self._lock:
                req_vram  = vram_total * self._keep_vram_pct if vram_total else 800
                has_vram  = vram_free is None or vram_free > req_vram
                has_sys_ram = free_ram_mb() > 2000

                if self._count < self._max and has_vram and has_sys_ram:
                    self._count += 1
                    self._last_acquire = time.time()
                    return

    def release(self):
        with self._lock:
            self._count = max(0, self._count - 1)

    def __enter__(self):    self.acquire()
    def __exit__(self, *_): self.release()


def _auto_gpu_slots(vram_per_job_mb: int = 1800, min_slots: int = 1, max_slots: int = 8) -> int:
    try:
        r = subprocess.run(
            ["nvidia-smi", "--query-gpu=memory.total", "--format=csv,noheader,nounits"],
            capture_output=True, text=True, timeout=2
        )
        if r.returncode == 0:
            totals = [int(x.strip()) for x in r.stdout.strip().splitlines() if x.strip().isdigit()]
            if totals:
                total_mb   = min(totals)
                usable_mb  = int(total_mb * 0.85)
                slots      = max(min_slots, min(max_slots, usable_mb // vram_per_job_mb))
                print(f"  GPU VRAM detected: {total_mb} MB → {slots} concurrent encode slots")
                return slots
    except Exception:
        pass
    print("  GPU VRAM: cannot read, using default 2 slots")
    return 2

def _detect_gpu() -> dict:
    candidates = [
        ("av1_nvenc",  "NVIDIA", "AV1"),
        ("av1_amf",    "AMD",    "AV1"),
        ("av1_qsv",    "Intel",  "AV1"),
        ("hevc_nvenc", "NVIDIA", "H.265"),
        ("hevc_amf",   "AMD",    "H.265"),
        ("hevc_qsv",   "Intel",  "H.265"),
        ("h264_nvenc", "NVIDIA", "H.264"),
        ("h264_amf",   "AMD",    "H.264"),
        ("h264_qsv",   "Intel",  "H.264"),
    ]
    found, seen = {}, set()
    for codec, vendor, fmt in candidates:
        key = (vendor, fmt)
        if key not in seen and _test_enc(codec):
            seen.add(key)
            found[codec] = {"vendor": vendor, "format": fmt}
    return found

def _get_real_enc_extra(enc: str) -> tuple:
    # film-grain=6: SVT-AV1 analyses source grain, denoises, re-synthesises at decode.
    # On clean video: no grain detected → no synthesis → safe. On iPhone footage: 20-30% smaller.
    _svtav1_fg = f"lp={_cpu_cores}:film-grain=6:film-grain-denoise=1"
    if enc == "libsvtav1_p4": return "libsvtav1", ["-preset","4","-svtav1-params",_svtav1_fg]
    if enc == "libsvtav1":    return "libsvtav1", ["-preset","6","-svtav1-params",_svtav1_fg]
    if enc in ("libx265", "libx264"):
        return enc, ["-preset", "medium"]
    if enc == "libaom-av1":
        return enc, ["-cpu-used","6","-row-mt","1","-tiles","4x4"]
    return enc, []


def _quality_args(enc: str, crf: int) -> list:
    # h264/hevc nvenc: uncapped VBR + HQ tune + Adaptive Quantization.
    # av1_nvenc: -tune hq already enables AQ; -spatial_aq/-temporal_aq not supported on AV1 NVENC.
    if enc in ("h264_nvenc","hevc_nvenc"):
        return ["-rc","vbr","-cq",str(crf),"-b:v","0",
                "-tune","hq","-spatial_aq","1","-temporal_aq","1","-aq-strength","15"]
    if enc == "av1_nvenc":
        return ["-rc","vbr","-cq",str(crf),"-b:v","0","-tune","hq"]
    if enc in ("h264_amf","hevc_amf"):
        return ["-rc","cqp","-qp",str(crf)]
    if enc == "av1_amf":
        return ["-rc","qvbr","-qvbr_quality_level",str(crf)]
    if enc in ("h264_qsv","hevc_qsv","av1_qsv"):
        return ["-global_quality",str(crf),"-look_ahead","1","-b:v","0"]
    return ["-crf",str(crf),"-b:v","0"]

def _sanitize_upload_filename(raw: str) -> str:
    name = re.sub(r'[\\/*?:"<>|]', "_", raw)
    name = re.sub(r'[\x00-\x1f]', '', name).strip()
    return name

def _resolve_upload_file(server_file: str):
    p = (UPLOAD_DIR / server_file).resolve()
    try:
        p.relative_to(UPLOAD_DIR.resolve())
    except ValueError:
        return None
    return p

def _get_upload(request, job_id):
    """Resolve upload source. Returns (src_path, orig_name).
    For server_file: resolves existing temp path (from /api/probe).
    For request.files: saves uploaded file to UPLOAD_DIR.
    Raises ValueError with user-facing message on invalid input."""
    server_file = request.form.get("server_file")
    if server_file:
        orig_name = request.form.get("original_name") or ""
        tmp_p = _resolve_upload_file(server_file)
        if tmp_p is None:
            raise ValueError("Invalid file path!")
        if not tmp_p.exists():
            raise ValueError("Uploaded file was lost! Please reselect.")
        if not orig_name:
            raise ValueError("Missing original filename!")
        return tmp_p, orig_name
    f = request.files.get("file")
    if not f:
        raise ValueError("No file uploaded!")
    orig_name = _sanitize_upload_filename(f.filename)
    src_path = UPLOAD_DIR / f"{Path(orig_name).stem}_{job_id}{Path(orig_name).suffix}"
    f.save(str(src_path))
    return src_path, orig_name

def _check_disk_space():
    total, used, free = shutil.disk_usage(UPLOAD_DIR)
    free_gb = free / (1024 ** 3)
    if free_gb < MIN_FREE_DISK_GB:
        return False, f"Server storage is almost full ({free_gb:.1f}GB left). Please try again later!"
    return True, ""

def _parse_dl_size(text: str) -> int:
    m = re.search(r'of\s+~?\s*([\d.]+)\s*(GiB|MiB|KiB|B)\b', text)
    if not m: return 0
    mult = {"GiB": 1 << 30, "MiB": 1 << 20, "KiB": 1 << 10, "B": 1}
    return int(float(m.group(1)) * mult.get(m.group(2), 0))

def _get_rotation(stream: dict) -> int:
    """Return display rotation (0/90/180/270) from a video stream dict."""
    if "rotate" in stream.get("tags", {}):
        try: return abs(int(stream["tags"]["rotate"]))
        except: pass
    for sd in stream.get("side_data_list", []):
        if sd.get("side_data_type") == "Display Matrix":
            try: return abs(int(sd.get("rotation", 0)))
            except: pass
    return 0


def _extract_ui_meta(tmp_path):
    data = _probe(str(tmp_path))
    info = {"duration": "N/A", "codec": "N/A", "res": "N/A", "fps": "N/A", "audio": "N/A"}
    if not data: return info

    try:
        d = float(data["format"]["duration"])
        info["duration"] = f"{int(d//60)}:{int(d%60):02d}"
    except: pass

    for s in data.get("streams", []):
        if s.get("codec_type") == "video":
            raw = s.get("codec_name", "").lower()
            info["codec"] = CODEC_LABELS.get(raw, raw.upper())
            w, h = s.get('width', '?'), s.get('height', '?')
            rotation = _get_rotation(s)
            if rotation in (90, 270) and w != '?' and h != '?': info["res"] = f"{h}x{w}"
            else: info["res"] = f"{w}x{h}"
            r = s.get("r_frame_rate", "0/1")
            if "/" in r:
                n, d2 = r.split("/")
                if int(d2): info["fps"] = f"{round(int(n)/int(d2),1)}"
            break

    for s in data.get("streams", []):
        if s.get("codec_type") == "audio":
            raw = s.get("codec_name", "").lower()
            ac = AUDIO_LABELS.get(raw, raw.upper())
            br = s.get("bit_rate")
            br_int = int(br) if br else 0
            if br_int < 10000:
                br_tag = s.get("tags", {}).get("BPS") or s.get("tags", {}).get("BPS-eng")
                if br_tag:
                    try: br_int = int(br_tag)
                    except: pass
            brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
            info["audio"] = f"{ac}{brs}"
            break

    info["is_dolby_vision"], info["dv_profile"] = _get_dv_info(data)

    # Detect Apple iPhone-specific shooting modes from QuickTime format tags
    # These tags come from ffprobe format.tags — no extra ExifTool call needed.
    fmt_tags = data.get("format", {}).get("tags", {})

    # Dual Camera: front-and-back-camera-composition = "1" means iPhone filmed with
    # both front and rear cameras simultaneously (Dual Cam / FaceTime Camera feature)
    info["is_dual_camera"] = (
        fmt_tags.get("com.apple.quicktime.front-and-back-camera-composition") == "1"
    )

    # Action Mode: action-video.stabilization-strength present means EIS was active
    # (value 4 = max stabilization, any non-zero value = Action Mode on)
    stab = fmt_tags.get("com.apple.quicktime.action-video.stabilization-strength", "")
    try:
        info["is_action_mode"] = int(stab) > 0
    except (ValueError, TypeError):
        info["is_action_mode"] = False

    return info

def _read_progress(fpath):
    try:
        txt  = Path(fpath).read_text(encoding="utf-8", errors="ignore")
        done = "progress=end" in txt
        t    = 0.0
        for ln in reversed(txt.splitlines()):
            if ln.startswith("out_time_ms="):
                try:
                    ms = int(ln.split("=")[1])
                    if ms > 0: t = ms/1_000_000
                except: pass
                break
        return t, done
    except: return 0.0, False

def _inject_apple_keys(src: str, dst: str, key_names: list) -> bool:
    """
    Binary-inject specific Apple QuickTime Keys from src into dst.

    QuickTime Keys live in moov→udta→meta(FullBox)→keys+ilst.
    ExifTool rewrites these boxes and drops tags it doesn't know.
    This function appends the missing entries at the binary level.

    keys box:  [size][keys][ver/flags(4)][count(4)] [entry: size+namespace+'mdta'+name]*
    ilst box:  [size][ilst]              [item: size+1basedIdx+data_box]*
    """
    if not key_names:
        return False
    try:
        # Source: read only the moov box (source file may not be faststart → moov at end).
        # Seeking past non-moov boxes is O(1), avoiding loading 4 GB into RAM.
        src_b = None
        with open(src, 'rb') as f:
            while True:
                hdr = f.read(8)
                if len(hdr) < 8: break
                sz = struct.unpack_from('>I', hdr, 0)[0]
                fc = hdr[4:8]
                hdr_len = 8
                if sz == 1:
                    ext = f.read(8)
                    if len(ext) < 8: break
                    sz = struct.unpack_from('>Q', ext, 0)[0]; hdr_len = 16
                if sz < hdr_len: break
                body_len = sz - hdr_len
                if fc == b'moov':
                    src_b = hdr + (ext if hdr_len == 16 else b'') + f.read(body_len)
                    break
                f.seek(body_len, 1)
        if not src_b:
            return False
        with open(dst, 'rb') as f:
            dst_b = bytearray(f.read())
    except OSError:
        return False

    def find_meta_chain(data):
        ms, me = _mp4_find_in_range(data, 0, len(data), b'moov')
        if ms < 0: return None
        # Scan direct moov children only (no DFS) to find meta and udta separately.
        us = ue = direct_ms = direct_me = -1
        for fc, ds, be in _mp4_walk_top(data, ms + 8, me):
            if fc == b'udta' and us < 0: us, ue = ds - 8, be
            elif fc == b'meta' and direct_ms < 0: direct_ms, direct_me = ds - 8, be
        # Prefer moov→meta Regular Box (QuickTime native) — iOS reads this, not udta→meta.
        # Regular Box: fourcc of first child at +12 is a printable fourcc (hdlr/keys/ilst).
        # Full Box:    version+flags at +8, child size at +12 (high byte = 0x00, non-ASCII).
        mts = mte = -1
        if direct_ms >= 0:
            fc_probe = bytes(data[direct_ms + 12: direct_ms + 16]) if direct_ms + 16 <= direct_me else b''
            if fc_probe in (b'keys', b'ilst', b'hdlr'):
                mts, mte = direct_ms, direct_me  # QuickTime Regular Box
        if mts < 0 and us >= 0:
            mts, mte = _mp4_find_in_range(data, us + 8, ue, b'meta')  # fallback: udta→meta
        if mts < 0 and direct_ms >= 0:
            mts, mte = direct_ms, direct_me  # last resort: Full Box moov→meta
        if mts < 0: return None
        # meta can be a regular box (children at mts+8, QuickTime native) or a
        # Full Box (children at mts+12, ISOBMFF). Detect by checking the first
        # child's fourcc at each candidate offset.
        cstart = mts + 12  # default: Full Box
        if mts + 8 + 8 <= mte:
            fc_probe = bytes(data[mts + 8 + 4: mts + 8 + 8])
            if fc_probe in (b'keys', b'ilst', b'hdlr'):
                cstart = mts + 8   # regular box
        ks = ke = is_ = ie = -1
        for fc, ds, be in _mp4_walk_top(data, cstart, mte):
            if fc == b'keys': ks, ke = ds-8, be
            elif fc == b'ilst': is_, ie = ds-8, be
        # meta_in_udta: True only when meta is a child of udta (not a sibling).
        # When False, udta size must NOT be updated when meta grows.
        meta_in_udta = us >= 0 and mts > us and mts < ue
        return dict(moov=(ms, me), udta=(us, ue), meta=(mts, mte),
                    keys=(ks, ke), ilst=(is_, ie), meta_in_udta=meta_in_udta)

    def read_keys(data, ks, ke):
        if ks < 0 or ks+16 > len(data): return []
        count = struct.unpack_from('>I', data, ks+12)[0]
        pos, res = ks+16, []
        for i in range(count):
            if pos+8 > min(ke, len(data)): break
            esz = struct.unpack_from('>I', data, pos)[0]
            if esz < 8 or pos+esz > len(data): break
            name = bytes(data[pos+8:pos+esz]).decode('utf-8', errors='replace')
            res.append((name, bytes(data[pos:pos+esz]), i+1))
            pos += esz
        return res

    def read_ilst(data, is_, ie):
        if is_ < 0: return {}
        pos, items = is_+8, {}
        while pos+8 <= min(ie, len(data)):
            isz = struct.unpack_from('>I', data, pos)[0]
            idx = struct.unpack_from('>I', data, pos+4)[0]
            if isz < 8 or pos+isz > len(data): break
            items[idx] = bytes(data[pos:pos+isz])
            pos += isz
        return items

    # Parse source
    sc = find_meta_chain(src_b)
    if not sc or sc['keys'][0] < 0: return False
    src_keys = read_keys(src_b, *sc['keys'])
    src_ilst = read_ilst(src_b, *sc['ilst'])
    src_lut = {}
    for name, entry_b, idx in src_keys:
        if name in key_names and idx in src_ilst:
            src_lut[name] = (entry_b, src_ilst[idx][8:])  # entry bytes + ilst payload

    if not src_lut: return False

    # Parse destination
    dc = find_meta_chain(dst_b)
    if not dc or dc['keys'][0] < 0: return False
    dst_keys = read_keys(dst_b, *dc['keys'])
    existing = {n for n, _, _ in dst_keys}
    n_existing = len(dst_keys)
    dst_ilst = read_ilst(dst_b, *dc['ilst'])

    to_add = [(n, src_lut[n]) for n in key_names if n in src_lut and n not in existing]

    # Keys present in dst but with wrong payload (e.g. ExifTool wrote as UTF-8 string
    # instead of the original binary integer — it doesn't know the proprietary type).
    dst_key_idx = {n: i for n, _, i in dst_keys}
    to_replace = []
    for name in key_names:
        if name in src_lut and name in existing:
            di = dst_key_idx.get(name)
            if di is not None and di in dst_ilst:
                _, src_pl = src_lut[name]
                if dst_ilst[di][8:] != src_pl:
                    to_replace.append((di, src_pl))

    # Both additions and replacements require an existing ilst box.
    # Creating ilst from scratch would require restructuring meta — not done here.
    if dc['ilst'][0] < 0:
        to_add = []
        to_replace = []
    if not to_add and not to_replace: return True  # already present with correct values

    # Build new bytes for additions
    new_keys_b = b''.join(eb for _, (eb, _) in to_add)
    new_ilst_b = b''
    for i, (_, (_, payload)) in enumerate(to_add):
        new_idx = n_existing + i + 1
        new_ilst_b += struct.pack('>II', 8 + len(payload), new_idx) + payload

    nk, ni = len(new_keys_b), len(new_ilst_b)
    total = nk + ni

    ks, ke = dc['keys']
    is_, ie = dc['ilst']
    mts, ms = dc['meta'][0], dc['moov'][0]
    # Only include udta in the parent-size-update chain if meta lives inside it.
    us = dc['udta'][0] if dc.get('meta_in_udta') else -1

    # Handle replacements first (last-to-first to preserve earlier positions)
    replace_delta = 0
    if to_replace and is_ >= 0:
        src_map = {di: pl for di, pl in to_replace}
        repl_list = []
        rpos = is_ + 8
        while rpos + 8 <= min(ie, len(dst_b)):
            isz = struct.unpack_from('>I', dst_b, rpos)[0]
            if isz < 8 or rpos + isz > len(dst_b): break
            item_idx = struct.unpack_from('>I', dst_b, rpos + 4)[0]
            if item_idx in src_map:
                new_item = struct.pack('>II', 8 + len(src_map[item_idx]), item_idx) + src_map[item_idx]
                repl_list.append((rpos, rpos + isz, bytes(new_item)))
            rpos += isz
        repl_list.sort(reverse=True)
        for rstart, rend, new_bytes in repl_list:
            delta = len(new_bytes) - (rend - rstart)
            dst_b[rstart:rend] = bytearray(new_bytes)
            replace_delta += delta
        if replace_delta != 0:
            struct.pack_into('>I', dst_b, is_, struct.unpack_from('>I', dst_b, is_)[0] + replace_delta)
            for box_s in (mts, us, ms):
                if box_s < 0: continue
                struct.pack_into('>I', dst_b, box_s,
                                 struct.unpack_from('>I', dst_b, box_s)[0] + replace_delta)
            # Adjust ie (and ks/ke if keys come after ilst)
            if ks > ie:  # ilst precedes keys: keys positions shift too
                ks += replace_delta
                ke += replace_delta
            ie += replace_delta

    # Append to keys box first, then ilst (order matters for offset tracking)
    if ie < 0 or ke <= is_:
        # keys before ilst (or no ilst)
        dst_b[ke:ke] = new_keys_b
        struct.pack_into('>I', dst_b, ks, struct.unpack_from('>I', dst_b, ks)[0] + nk)
        struct.pack_into('>I', dst_b, ks+12, n_existing + len(to_add))
        if is_ >= 0:
            dst_b[ie+nk:ie+nk] = new_ilst_b
            struct.pack_into('>I', dst_b, is_+nk,
                             struct.unpack_from('>I', dst_b, is_+nk)[0] + ni)
    else:
        # ilst before keys
        dst_b[ie:ie] = new_ilst_b
        struct.pack_into('>I', dst_b, is_, struct.unpack_from('>I', dst_b, is_)[0] + ni)
        dst_b[ke+ni:ke+ni] = new_keys_b
        struct.pack_into('>I', dst_b, ks+ni,
                         struct.unpack_from('>I', dst_b, ks+ni)[0] + nk)
        struct.pack_into('>I', dst_b, ks+ni+12, n_existing + len(to_add))

    # Update parent box sizes: meta → udta (if present) → moov
    for box_s in (mts, us, ms):
        if box_s < 0: continue
        struct.pack_into('>I', dst_b, box_s,
                         struct.unpack_from('>I', dst_b, box_s)[0] + total)

    # Update stco/co64 chunk offsets when moov precedes mdat (faststart)
    mdat_s = -1
    for fc, ds, be in _mp4_walk_top(dst_b, 0, len(dst_b)):
        if fc == b'mdat': mdat_s = ds-8; break
    total_delta = replace_delta + total
    if ms >= 0 and mdat_s >= 0 and ms < mdat_s and total_delta != 0:
        # Use earliest modification: replacements are within ilst; additions start at ke
        if to_replace and is_ >= 0:
            first_ins = is_ + 8
        elif ie >= 0:
            first_ins = min(ke, ie)
        else:
            first_ins = ke
        _mp4_shift_offsets(dst_b, first_ins, total_delta)

    tmp = dst + '._iak_tmp'
    try:
        with open(tmp, 'wb') as f:
            f.write(bytes(dst_b))
        os.replace(tmp, dst)
        return True
    except OSError:
        try: os.unlink(tmp)
        except: pass
        return False


def _sync_meta(src: str, dst: str, rotation: int = 0) -> bool:
    if not EXIFTOOL_BIN: return False

    # Step 1: Copy critical Apple QuickTime Keys atoms explicitly.
    # IMPORTANT: Do NOT use -Keys:all — it copies Keys:FocalLength (physical mm
    # e.g. 6.765mm) which overrides display-formatted Keys:CameraLensModel
    # (e.g. "24 mm f/1.78") causing iOS to show wrong focal length.
    cmd1 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-Keys:GPSCoordinates",
        "-Keys:LocationBody",
        "-Keys:LocationDate",
        "-Keys:LocationName",
        "-Keys:Make",
        "-Keys:Model",
        "-Keys:Software",
        "-Keys:CameraLensModel",
        "-Keys:CreationDate",
        "-Keys:ContentIdentifier",
        "-Keys:HDRVideoFormat",
        "-Keys:LocationAccuracyHorizontal",
        "-Keys:FullFrameRatePlaybackIntent",
        "-Keys:Metadata1",
        # Mirror critical Keys into legacy UserData atoms
        "-UserData:Make<Keys:Make",
        "-UserData:Model<Keys:Model",
        "-UserData:GPSCoordinates<Keys:GPSCoordinates",
        "-UserData:CreationDate<Keys:CreationDate",
        "-overwrite_original",
        dst
    ]
    r1 = None
    try:
        r1 = subprocess.run(cmd1, capture_output=True, text=True, timeout=30)
    except subprocess.TimeoutExpired:
        pass

    # Step 2: Copy all remaining metadata (excluding Keys to avoid overwriting step 1).
    cmd2 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-all:all",
        "-Rotation",
        "-MatrixStructure",
        "--FileModifyDate",
        "--FileCreateDate",
        "--Keys:all",
        "-overwrite_original",
        dst
    ]
    r2 = None
    try:
        r2 = subprocess.run(cmd2, capture_output=True, text=True, encoding="utf-8", errors="replace", timeout=30)
    except subprocess.TimeoutExpired:
        pass

    # Step 3: Binary-inject proprietary Apple Keys that ExifTool doesn't know.
    # ExifTool rewrites the Keys/ilst box and drops tags not in its database
    # (e.g. front-and-back-camera-composition, action-video.stabilization-strength).
    # We restore them directly from the source file at the binary level.
    _inject_apple_keys(src, dst, _APPLE_KEYS_TO_PRESERVE)

    # Step 4: Move moov→udta→meta to moov→meta (direct child).
    # iOS Photos requires QuickTime Keys (front-and-back-camera-composition,
    # action-video.stabilization-strength, etc.) at moov→meta (QuickTime native
    # position). ExifTool writes to moov→udta→meta, which iOS ignores for these keys.
    _promote_udta_meta_to_moov(dst)

    if rotation:
        cmd_rot = [
            EXIFTOOL_BIN,
            f"-Rotation={rotation}",
            "-overwrite_original",
            dst
        ]
        try:
            subprocess.run(cmd_rot, capture_output=True, timeout=15)
        except:
            pass

    return bool(r1 and r1.returncode == 0) or bool(r2 and r2.returncode == 0)


def _promote_udta_meta_to_moov(path: str) -> bool:
    """
    Ensure the Keys meta box is a Regular Box direct child of moov.

    iOS Photos requires QuickTime Keys (front-and-back-camera-composition etc.)
    at moov→meta (Regular Box, no version+flags).  ExifTool may write either:
      - moov→udta→meta (wrong location) or
      - moov→meta as Full Box (wrong format — version+flags read as terminator)
    We fix both cases here, adjusting stco/co64 offsets when the file shrinks.
    """
    _MAX_LOAD_MB = 500
    try:
        size = os.path.getsize(path)
    except OSError:
        return False
    if size > _MAX_LOAD_MB * 1024 * 1024:
        import warnings
        warnings.warn(
            f"_promote_udta_meta_to_moov: skipping {path!r} ({size // 1024 // 1024} MB > {_MAX_LOAD_MB} MB limit) "
            "— iOS QuickTime Keys metadata may not be promoted. "
            "Dual Camera / Action Mode icons in Photos may be missing.",
            stacklevel=2
        )
        return False
    try:
        with open(path, 'rb') as f:
            data = bytearray(f.read())
    except OSError:
        return False

    def _is_full(mts):
        """True if meta box at mts is a Full Box (has 4-byte version+flags after header)."""
        probe = bytes(data[mts + 12: mts + 16]) if mts + 16 <= len(data) else b''
        return probe not in (b'hdlr', b'keys', b'ilst')

    def _to_regular(mts, mte):
        """Return bytes of meta converted from Full Box to Regular Box (removes 4 bytes)."""
        return struct.pack('>I', mte - mts - 4) + b'meta' + bytes(data[mts + 12: mte])

    # Find moov
    moov_s = moov_e = -1
    for fc, ds, be in _mp4_walk_top(data, 0, len(data)):
        if fc == b'moov':
            moov_s, moov_e = ds - 8, be
            break
    if moov_s < 0:
        return False

    # Scan top-level moov children for meta and udta
    moov_meta_s = moov_meta_e = udta_s = udta_e = -1
    for fc, ds, be in _mp4_walk_top(data, moov_s + 8, moov_e):
        if fc == b'meta' and moov_meta_s < 0:
            moov_meta_s, moov_meta_e = ds - 8, be
        elif fc == b'udta' and udta_s < 0:
            udta_s, udta_e = ds - 8, be

    # --- Case 1: meta already at moov level ---
    if moov_meta_s >= 0:
        if not _is_full(moov_meta_s):
            return False  # already Regular Box at moov level — nothing to do
        # Full Box at moov level: convert in-place (removes 4 bytes)
        new_meta = _to_regular(moov_meta_s, moov_meta_e)
        data = data[:moov_meta_s] + bytearray(new_meta) + data[moov_meta_e:]
        struct.pack_into('>I', data, moov_s, moov_e - moov_s - 4)  # moov shrinks 4
        # mdat shifted 4 bytes earlier; fix stco/co64
        _mp4_shift_offsets(data, moov_e - 4, -4)

    # --- Case 2: meta inside udta, not yet at moov level ---
    elif udta_s >= 0:
        # Find meta inside udta
        udta_meta_s = udta_meta_e = -1
        for fc, ds, be in _mp4_walk_top(data, udta_s + 8, udta_e):
            if fc == b'meta':
                udta_meta_s, udta_meta_e = ds - 8, be
                break
        if udta_meta_s < 0:
            return False

        meta_size = udta_meta_e - udta_meta_s
        if _is_full(udta_meta_s):
            new_meta = _to_regular(udta_meta_s, udta_meta_e)
            net = -4
        else:
            new_meta = bytes(data[udta_meta_s: udta_meta_e])
            net = 0

        # Remove meta from udta
        data = data[:udta_meta_s] + data[udta_meta_e:]
        struct.pack_into('>I', data, udta_s, udta_e - udta_s - meta_size)

        # Insert at end of moov
        insert_pos = moov_e - meta_size
        data = data[:insert_pos] + bytearray(new_meta) + data[insert_pos:]

        # Update moov size and fix stco if file shrank
        struct.pack_into('>I', data, moov_s, (moov_e - moov_s) + net)
        if net != 0:
            _mp4_shift_offsets(data, insert_pos + len(new_meta), net)

    else:
        return False

    tmp = path + '._promo_tmp'
    try:
        with open(tmp, 'wb') as f:
            f.write(bytes(data))
        os.replace(tmp, path)
        return True
    except OSError:
        try:
            os.unlink(tmp)
        except Exception:
            pass
        return False


def _get_dv_info(probe_data) -> tuple:
    """Return (is_dolby_vision, dv_profile) from probe data."""
    for stream in probe_data.get("streams", []):
        for sd in stream.get("side_data_list", []):
            sdt = sd.get("side_data_type", "").lower()
            if "dowi" in sdt or "dovi" in sdt:
                return True, sd.get("dv_profile", 0)
    return False, 0


def _detect_dolby_vision(probe_data) -> bool:
    return _get_dv_info(probe_data)[0]


def _fix_mebx_tracks(dst: str) -> bool:
    """
    Binary-patch mebx track metadata corrupted by FFmpeg's 'unknown hdlr_type' warning.
    FFmpeg copies Apple mebx (Core Media Metadata) tracks but writes wrong
    hdlr.handler_type and stsd entry fourcc (both become 'stts' instead of 'mebx').
    Patches them back to b'mebx' for all tracks whose handler_name contains
    'Core Media Metadata'. Returns True if at least one track was patched.
    """
    try:
        with open(dst, 'rb') as f:
            buf = bytearray(f.read())
    except OSError:
        return False

    moov_s = moov_e = -1
    for fc, ds, be in _mp4_walk_top(buf, 0, len(buf)):
        if fc == b'moov':
            moov_s, moov_e = ds - 8, be
            break
    if moov_s < 0:
        return False

    patched = 0
    for fc, ds, be in _mp4_walk_top(buf, moov_s + 8, moov_e):
        if fc != b'trak':
            continue

        # hdlr box: [size(4)][hdlr(4)][ver/flags(4)][pre_defined(4)][handler_type(4)][reserved(12)][name...]
        hdlr_s, _ = _mp4_find_in_range(buf, ds, be, b'hdlr')
        if hdlr_s < 0 or hdlr_s + 32 > len(buf):
            continue

        # handler_name (null-terminated C string) starts at offset 32 from box start
        chunk = bytes(buf[hdlr_s + 32: min(hdlr_s + 200, len(buf))])
        null_pos = chunk.find(b'\x00')
        handler_name = chunk[:null_pos] if null_pos >= 0 else chunk
        if b'Core Media Metadata' not in handler_name:
            continue

        # Patch handler_type at offset 16 from box start.
        # Source iPhone MOV uses 'meta' (QuickTime metadata handler), not 'mebx'.
        if bytes(buf[hdlr_s + 16: hdlr_s + 20]) != b'meta':
            buf[hdlr_s + 16: hdlr_s + 20] = b'meta'
            patched += 1

        # stsd box: [size(4)][stsd(4)][ver/flags(4)][entry_count(4)][entry_size(4)][entry_fourcc(4)...]
        # entry1 fourcc is at offset 20 from stsd box start
        stsd_s, _ = _mp4_find_in_range(buf, ds, be, b'stsd')
        if stsd_s >= 0 and stsd_s + 24 <= len(buf):
            if bytes(buf[stsd_s + 20: stsd_s + 24]) != b'mebx':
                buf[stsd_s + 20: stsd_s + 24] = b'mebx'
                patched += 1

    if not patched:
        return False

    tmp = dst + '._mebx_tmp'
    try:
        with open(tmp, 'wb') as f:
            f.write(bytes(buf))
        os.replace(tmp, dst)
        return True
    except OSError:
        try: os.unlink(tmp)
        except: pass
        return False


def _read_qt_keys_critical(path: str) -> dict:
    """
    Read binary type + value of critical Apple QuickTime Keys directly from the file.
    ExifTool converts these binary-integer keys to UTF-8 strings and loses type info.
    Returns dict of display label → type-annotated string, e.g.:
        {"Apple Key: Dual Camera": "1 (int)", "Apple Key: Action Mode": "4 (utf8)"}

    Only the moov box is read into memory (seek past mdat/ftyp headers), so this
    is safe even for 4 GB source files.
    """
    try:
        moov_data = None
        with open(path, 'rb') as f:
            while True:
                hdr = f.read(8)
                if len(hdr) < 8:
                    break
                sz = struct.unpack_from('>I', hdr, 0)[0]
                fc = hdr[4:8]
                hdr_len = 8
                if sz == 1:                        # extended 64-bit size
                    ext = f.read(8)
                    if len(ext) < 8:
                        break
                    sz = struct.unpack_from('>Q', ext, 0)[0]
                    hdr_len = 16
                if sz < hdr_len:
                    break
                body_len = sz - hdr_len
                if fc == b'moov':
                    moov_data = hdr + (ext if hdr_len == 16 else b'') + f.read(body_len)
                    break
                f.seek(body_len, 1)               # skip non-moov boxes (O(1) even for huge mdat)
    except OSError:
        return {}

    if not moov_data:
        return {}

    ms, me = _mp4_find_in_range(moov_data, 0, len(moov_data), b'moov')
    if ms < 0: return {}

    # Prefer moov→meta Regular Box (QuickTime native, iOS reads this) over udta→meta.
    us = ue = direct_ms = direct_me = -1
    for fc, ds, be in _mp4_walk_top(moov_data, ms + 8, me):
        if fc == b'udta' and us < 0: us, ue = ds - 8, be
        elif fc == b'meta' and direct_ms < 0: direct_ms, direct_me = ds - 8, be
    mts = mte = -1
    if direct_ms >= 0:
        fc_probe = bytes(moov_data[direct_ms + 12: direct_ms + 16]) if direct_ms + 16 <= direct_me else b''
        if fc_probe in (b'hdlr', b'keys', b'ilst'):
            mts, mte = direct_ms, direct_me
    if mts < 0 and us >= 0:
        mts, mte = _mp4_find_in_range(moov_data, us + 8, ue, b'meta')
    if mts < 0 and direct_ms >= 0:
        mts, mte = direct_ms, direct_me
    if mts < 0: return {}

    # Regular Box (QuickTime): children start at mts+8; Full Box (ISOBMFF): mts+12
    probe = bytes(moov_data[mts + 12: mts + 16]) if mts + 16 <= len(moov_data) else b''
    cstart = mts + 8 if probe in (b'hdlr', b'keys', b'ilst') else mts + 12

    ks = ke = is_ = ie = -1
    for fc, ds, be in _mp4_walk_top(moov_data, cstart, mte):
        if fc == b'keys': ks, ke = ds - 8, be
        elif fc == b'ilst': is_, ie = ds - 8, be
    if ks < 0 or is_ < 0: return {}

    # Build index: 1-based key index → full key name (only for critical keys)
    count = struct.unpack_from('>I', moov_data, ks + 12)[0]
    pos, key_idx = ks + 16, {}
    for i in range(count):
        if pos + 8 > ke: break
        esz = struct.unpack_from('>I', moov_data, pos)[0]
        if esz < 8: break
        name = moov_data[pos + 8: pos + esz].decode('utf-8', errors='replace')
        if name in _QT_KEYS_CRITICAL:
            key_idx[i + 1] = name
        pos += esz
    if not key_idx: return {}

    # Parse ilst items and format type-annotated values
    result = {}
    pos = is_ + 8
    while pos + 8 <= ie:
        isz = struct.unpack_from('>I', moov_data, pos)[0]
        if isz < 8 or pos + isz > len(moov_data): break
        idx = struct.unpack_from('>I', moov_data, pos + 4)[0]
        if idx in key_idx:
            label = _QT_KEYS_CRITICAL[key_idx[idx]]
            db = pos + 8   # start of 'data' sub-box
            if db + 16 <= pos + isz:
                db_sz  = struct.unpack_from('>I', moov_data, db)[0]
                db_typ = struct.unpack_from('>I', moov_data, db + 8)[0]
                db_val = moov_data[db + 16: db + db_sz]
            else:
                db_typ, db_val = -1, b''
            tname = _QT_TYPE_NAMES.get(db_typ, f"0x{db_typ:08x}")
            if db_typ == 0x01:    # utf8
                val_str = db_val.decode('utf-8', errors='replace')
            elif db_typ == 0x15:  # unsigned int (1–8 bytes, big-endian)
                val_str = str(int.from_bytes(db_val, 'big')) if db_val else '0'
            else:
                val_str = db_val.hex() if db_val else ''
            result[f"Apple Key: {label}"] = f"{val_str} ({tname})"
        pos += isz
    return result

# ── Dolby Vision: Annex-B parser + timestamp-preserving mux ──────────────────

def _annexb_to_mp4_samples(hevc_path: str) -> list:
    """
    Parse dovi_tool inject-rpu Annex-B output into per-frame MP4 samples.
    Each sample = concatenation of 4-byte-BE-length-prefixed NAL units.
    Frames are delimited by AUD NAL units (HEVC type 35) which dovi_tool adds
    by default between every frame.
    Returns [] on error.
    """
    try:
        with open(hevc_path, 'rb') as f:
            raw = f.read()
    except OSError:
        return []

    total = len(raw)
    # ── find all start codes (00 00 01 or 00 00 00 01) ──────────────────────
    sc_pos = []   # (first_byte_after_sc, sc_length)
    i = 0
    while i < total - 2:
        if raw[i] == 0 and raw[i + 1] == 0:
            if i + 3 < total and raw[i + 2] == 0 and raw[i + 3] == 1:
                sc_pos.append((i + 4, 4)); i += 4; continue
            if raw[i + 2] == 1:
                sc_pos.append((i + 3, 3)); i += 3; continue
        i += 1

    if not sc_pos:
        return []

    # ── extract NAL unit data for each SC position ───────────────────────────
    nals = []   # (nal_type, nal_bytes)
    for idx, (nal_start, sc_len) in enumerate(sc_pos):
        nal_end = sc_pos[idx + 1][0] - sc_pos[idx + 1][1] if idx + 1 < len(sc_pos) else total
        nal = raw[nal_start:nal_end]
        # nal_end already points to the byte before the next start code
        if nal:
            nals.append(((nal[0] >> 1) & 0x3F, nal))

    # ── group by AUD (type 35) → one MP4 sample per frame ───────────────────
    samples, cur = [], []
    for nal_type, nal in nals:
        if nal_type == 35:                     # AUD: signals start of new AU
            if cur:
                samples.append(b''.join(struct.pack('>I', len(nd)) + nd
                                        for _, nd in cur))
                cur = []
        else:
            cur.append((nal_type, nal))
    if cur:
        samples.append(b''.join(struct.pack('>I', len(nd)) + nd for _, nd in cur))

    return samples


def _mp4_walk_top(buf, start: int, end: int):
    """Iterate top-level boxes in buf[start:end], yield (fourcc, data_start, box_end)."""
    off = start
    while off + 8 <= end:
        sz = struct.unpack_from('>I', buf, off)[0]
        fc = bytes(buf[off + 4:off + 8])
        hdr = 8
        if sz == 1:
            if off + 16 > end: break
            sz = struct.unpack_from('>Q', buf, off + 8)[0]; hdr = 16
        if sz == 0 or off + sz > end: break
        yield fc, off + hdr, off + sz
        off += sz


_DV_CONTAINER_FCS = frozenset({
    b'moov', b'trak', b'mdia', b'minf', b'stbl', b'edts', b'dinf', b'udta',
    b'meta', b'ilst', b'wide',
})


def _mp4_find_in_range(buf, start: int, end: int, target: bytes) -> tuple:
    """DFS search for target fourcc in buf[start:end]. Returns (box_start, box_end) or (-1,-1)."""
    for fc, ds, be in _mp4_walk_top(buf, start, end):
        if fc == target:
            return ds - 8, be          # box_start includes the 8-byte header
        if fc in _DV_CONTAINER_FCS:
            r = _mp4_find_in_range(buf, ds, be, target)
            if r[0] >= 0:
                return r
    return -1, -1


def _mp4_find_in_trak(buf, moov_s: int, moov_e: int,
                       handler: bytes, target: bytes) -> tuple:
    """Find target fourcc inside the trak with matching handler type.
    Handles both ISOBMFF (pre_defined=0, handler at box+16) and QuickTime
    legacy format (component_type at box+12, component_subtype at box+12+4=16
    OR no pre_defined, handler at box+12).
    We check both +12 and +16 for robustness.
    """
    for fc, ds, be in _mp4_walk_top(buf, moov_s, moov_e):
        if fc != b'trak':
            continue
        ho, he = _mp4_find_in_range(buf, ds, be, b'hdlr')
        if ho < 0:
            continue
        # Try offset +16 (ISOBMFF: ver/flags + pre_defined + handler_type)
        # and +12 (QuickTime: ver/flags + handler_type, no pre_defined)
        matched = False
        for ht_off in (ho + 16, ho + 12):
            if ht_off + 4 <= len(buf) and bytes(buf[ht_off: ht_off + 4]) == handler:
                matched = True
                break
        if matched:
            return _mp4_find_in_range(buf, ds, be, target)
    return -1, -1


def _parse_stco_buf(buf, box_s: int) -> list:
    """Parse stco/co64 at buf[box_s], return list of chunk offsets."""
    fc = bytes(buf[box_s + 4: box_s + 8])
    is64 = (fc == b'co64')
    d = box_s + 8                                     # skip box header → version+flags
    count = struct.unpack_from('>I', buf, d + 4)[0]   # skip 4-byte ver/flags → count
    entry_sz, fmt = (8, '>Q') if is64 else (4, '>I')
    return [struct.unpack_from(fmt, buf, d + 8 + i * entry_sz)[0] for i in range(count)]


def _parse_stsz_buf(buf, box_s: int) -> list:
    """Parse stsz at buf[box_s], return list of per-sample sizes."""
    d = box_s + 8                                     # ver+flags
    fixed = struct.unpack_from('>I', buf, d + 4)[0]  # fixed sample-size field
    count = struct.unpack_from('>I', buf, d + 8)[0]
    if fixed:
        return [fixed] * count
    return [struct.unpack_from('>I', buf, d + 12 + i * 4)[0] for i in range(count)]


def _parse_stsc_buf(buf, box_s: int) -> list:
    """Parse stsc, return list of (first_chunk_1based, samples_per_chunk, desc_idx)."""
    d = box_s + 8
    count = struct.unpack_from('>I', buf, d + 4)[0]
    return [(struct.unpack_from('>I', buf, d + 8 + i * 12)[0],
             struct.unpack_from('>I', buf, d + 12 + i * 12)[0],
             struct.unpack_from('>I', buf, d + 16 + i * 12)[0])
            for i in range(count)]


def _stsc_samples_per_chunk(stsc: list, n_chunks: int) -> list:
    """Expand stsc into a per-chunk list of sample counts."""
    result = []
    for ci in range(1, n_chunks + 1):
        spc = stsc[-1][1] if stsc else 1
        for j in range(len(stsc)):
            fc = stsc[j][0]
            nfc = stsc[j + 1][0] if j + 1 < len(stsc) else n_chunks + 1
            if fc <= ci < nfc:
                spc = stsc[j][1]; break
        result.append(spc)
    return result


def _dovi_mux_with_timestamps(base_mov: str, dv_samples: list, dst: str) -> bool:
    """
    Rebuild a MOV file that has:
      • Video bitstream data from dv_samples  (MP4-format, one bytes per frame)
      • Video timestamps (stts/ctts/stss)     preserved from base_mov's video trak
      • Audio data and timing                 preserved from base_mov's audio trak

    Strategy: the stsz and stco boxes have the same NUMBER of entries before and
    after RPU injection (same frame count, same chunk count), so their box SIZE
    does not change. We can therefore patch them in-place inside the bytearray
    copy of base_mov, then replace the mdat section with new video + audio data.

    Returns True on success.
    """
    try:
        with open(base_mov, 'rb') as f:
            buf = bytearray(f.read())
    except OSError:
        return False

    # ── locate top-level boxes (ftyp, moov, mdat) ────────────────────────────
    moov_s = moov_e = mdat_s = mdat_e = -1
    for fc, ds, be in _mp4_walk_top(buf, 0, len(buf)):
        if fc == b'moov': moov_s, moov_e = ds - 8, be
        elif fc == b'mdat': mdat_s, mdat_e = ds - 8, be
    if moov_s < 0 or mdat_s < 0:
        return False

    moov_body_s = moov_s + 8
    moov_body_e = moov_e

    # ── locate video track stbl entries ──────────────────────────────────────
    vs_s, vs_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stsz')
    vc_s, vc_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stco')
    vsc_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stsc')
    # also try co64
    if vc_s < 0:
        vc_s, vc_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'co64')

    if vs_s < 0 or vc_s < 0 or vsc_s < 0:
        return False

    v_stsz = _parse_stsz_buf(buf, vs_s)
    v_stco = _parse_stco_buf(buf, vc_s)
    v_stsc = _parse_stsc_buf(buf, vsc_s)
    n_video = len(v_stsz)
    n_chunks = len(v_stco)

    # Safety: frame count must match
    if len(dv_samples) != n_video:
        # Trim to min; mild mismatch (off-by-one) is tolerable
        min_n = min(len(dv_samples), n_video)
        if abs(len(dv_samples) - n_video) > 2:
            return False
        dv_samples = dv_samples[:min_n]
        n_video = min_n

    new_sizes = [len(s) for s in dv_samples]

    # ── build new video chunks grouped by stsc structure ─────────────────────
    spc_list = _stsc_samples_per_chunk(v_stsc, n_chunks)
    video_chunks = []
    si = 0
    for spc in spc_list:
        chunk = b''.join(dv_samples[si + j] for j in range(spc)
                         if si + j < len(dv_samples))
        video_chunks.append(chunk)
        si += spc

    # ── collect audio chunks from the original mdat ───────────────────────────
    audio_chunks = []
    ac_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stco')
    if ac_s < 0:
        ac_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'co64')
    as_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stsz')
    asc_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stsc')

    if ac_s >= 0 and as_s >= 0 and asc_s >= 0:
        a_stco = _parse_stco_buf(buf, ac_s)
        a_stsz = _parse_stsz_buf(buf, as_s)
        a_stsc = _parse_stsc_buf(buf, asc_s)
        a_spc_list = _stsc_samples_per_chunk(a_stsc, len(a_stco))
        asi = 0
        for ci, spc in enumerate(a_spc_list):
            chunk_sz = sum(a_stsz[asi + j] for j in range(spc)
                           if asi + j < len(a_stsz))
            chunk_off = a_stco[ci]
            if 0 < chunk_off < len(buf) and chunk_off + chunk_sz <= len(buf):
                audio_chunks.append(buf[chunk_off: chunk_off + chunk_sz])
            else:
                audio_chunks.append(b'')   # safety fallback
            asi += spc

    # ── compute new mdat layout ───────────────────────────────────────────────
    # mdat position stays the same (moov size unchanged, since stsz/stco entries
    # count is the same → box sizes unchanged → moov size unchanged).
    new_mdat_off = mdat_s
    new_mdat_body_s = new_mdat_off + 8         # after 8-byte mdat box header

    # video chunk offsets in new mdat
    cur = new_mdat_body_s
    new_v_stco = []
    for chunk in video_chunks:
        new_v_stco.append(cur)
        cur += len(chunk)

    # audio chunk offsets in new mdat (sequentially after video)
    new_a_stco = []
    for chunk in audio_chunks:
        new_a_stco.append(cur)
        cur += len(chunk)

    # ── patch video stsz entries in-place ────────────────────────────────────
    # stsz fixed-size field is at vs_s+12; if 0 → variable-size (entries at vs_s+20+)
    old_fixed = struct.unpack_from('>I', buf, vs_s + 12)[0]
    if old_fixed == 0:
        for i, sz_val in enumerate(new_sizes):
            eoff = vs_s + 20 + i * 4
            if eoff + 4 <= len(buf):
                struct.pack_into('>I', buf, eoff, sz_val)
    elif len(set(new_sizes)) == 1 and new_sizes[0] == old_fixed:
        pass                                   # fixed-size unchanged, nothing to do
    else:
        return False                           # can't patch fixed-size stsz in-place

    # ── patch video stco entries in-place ─────────────────────────────────────
    is64_v = (bytes(buf[vc_s + 4: vc_s + 8]) == b'co64')
    entry_sz_v, fmt_v = (8, '>Q') if is64_v else (4, '>I')
    for i, off_val in enumerate(new_v_stco):
        eoff = vc_s + 16 + i * entry_sz_v     # 8 hdr + 4 ver/flags + 4 count = 16
        if eoff + entry_sz_v <= len(buf):
            struct.pack_into(fmt_v, buf, eoff, off_val)

    # ── patch audio stco entries in-place ─────────────────────────────────────
    if new_a_stco and ac_s >= 0:
        is64_a = (bytes(buf[ac_s + 4: ac_s + 8]) == b'co64')
        entry_sz_a, fmt_a = (8, '>Q') if is64_a else (4, '>I')
        for i, off_val in enumerate(new_a_stco):
            eoff = ac_s + 16 + i * entry_sz_a
            if eoff + entry_sz_a <= len(buf):
                struct.pack_into(fmt_a, buf, eoff, off_val)

    # ── build new mdat and write output ──────────────────────────────────────
    new_mdat_body = b''.join(video_chunks) + b''.join(audio_chunks)
    new_mdat = struct.pack('>I', 8 + len(new_mdat_body)) + b'mdat' + new_mdat_body

    # Replace old mdat (and anything after it) with new mdat
    new_buf = bytes(buf[:mdat_s]) + new_mdat

    try:
        tmp_path = dst + '._dv_tmp'
        with open(tmp_path, 'wb') as f:
            f.write(new_buf)
        os.replace(tmp_path, dst)
        return True
    except OSError:
        try:
            os.unlink(tmp_path)
        except Exception:
            pass
        return False


# ── MP4 container helpers for dvcC injection ─────────────────────────────────

_MP4_CONTAINERS = frozenset({
    b'moov', b'trak', b'mdia', b'minf', b'stbl', b'dinf', b'udta',
    b'ilst', b'moof', b'traf', b'edts',
})
_MP4_SAMPLE_ENTRIES = frozenset({
    b'dvh1', b'dvhe', b'hvc1', b'hev1', b'avc1', b'avc3',
})


def _mp4_extract_box(data: bytes, btype: bytes) -> bytes:
    """Return full box bytes (size+type+payload) for first occurrence of btype."""
    idx = data.find(btype)
    while idx >= 4:
        sz = struct.unpack_from('>I', data, idx - 4)[0]
        if 8 <= sz <= len(data) - (idx - 4):
            return data[idx - 4: idx - 4 + sz]
        idx = data.find(btype, idx + 1)
    return b''


def _mp4_box_off(data: bytes, btype: bytes) -> int:
    """Return byte offset of size field for first occurrence of btype, or -1."""
    idx = data.find(btype)
    while idx >= 4:
        sz = struct.unpack_from('>I', data, idx - 4)[0]
        if sz >= 8:
            return idx - 4
        idx = data.find(btype, idx + 1)
    return -1


def _mp4_walk_parents(data: bytes, ins: int) -> list:
    """Return start offsets of all boxes that contain byte position ins."""
    result = []

    def walk(pos, end):
        while pos + 8 <= min(end, len(data)):
            sz = struct.unpack_from('>I', data, pos)[0]
            bt = data[pos + 4: pos + 8]
            if sz == 1 and pos + 16 <= len(data):
                sz = struct.unpack_from('>Q', data, pos + 8)[0]
            elif sz == 0:
                sz = len(data) - pos
            if sz < 8:
                break
            box_end = pos + sz
            if pos < ins <= box_end:
                result.append(pos)
                if bt in _MP4_CONTAINERS:
                    walk(pos + 8, box_end)
                elif bt == b'stsd':
                    walk(pos + 16, box_end)          # full-box header + entry_count
                elif bt in _MP4_SAMPLE_ENTRIES:
                    walk(pos + 16, box_end)          # box header + reserved(6) + data_ref(2)
                break
            pos = box_end

    walk(0, len(data))
    return result


def _mp4_shift_offsets(buf: bytearray, ins: int, n: int):
    """Increment stco/co64 chunk offsets that are >= ins by n."""
    raw = bytes(buf)
    for tag, esz, fmt in [(b'stco', 4, '>I'), (b'co64', 8, '>Q')]:
        p = raw.find(tag)
        while p >= 4:
            bsz = struct.unpack_from('>I', buf, p - 4)[0]
            if bsz >= 16:
                count = struct.unpack_from('>I', buf, p + 8)[0]
                for i in range(count):
                    op = p + 12 + i * esz
                    if op + esz > len(buf):
                        break
                    v = struct.unpack_from(fmt, buf, op)[0]
                    if v >= ins:
                        struct.pack_into(fmt, buf, op, v + n)
            p = raw.find(tag, p + 1)


def _inject_dvcc(src: str, dst: str) -> bool:
    """
    Copy dvcC (Dolby Vision Configuration) box from src MP4/MOV into dst.
    Call after muxing to restore DV container signaling lost during re-encode.
    Returns True if dvcC is now present in dst.
    """
    try:
        with open(src, 'rb') as f:
            src_bytes = f.read()
    except Exception:
        return False

    # MOV containers use 'dvvC', MP4 use 'dvcC' — same payload, different 4CC
    dvcc = _mp4_extract_box(src_bytes, b'dvcC')
    if not dvcc:
        dvvC = _mp4_extract_box(src_bytes, b'dvvC')
        if not dvvC:
            return False
        # Re-label dvvC → dvcC for MP4 output (payload is identical)
        dvcc = dvvC[:4] + b'dvcC' + dvvC[8:]

    try:
        with open(dst, 'rb') as f:
            buf = bytearray(f.read())
    except Exception:
        return False

    buf_bytes = bytes(buf)

    # Determine the correct DV config atom name for this container:
    # QuickTime/MOV containers (major_brand b'qt  ') use 'dvvC',
    # ISOBMFF/MP4 containers use 'dvcC'. Payload is identical.
    major_brand = buf_bytes[8:12] if len(buf_bytes) >= 12 else b''
    target_name = b'dvvC' if major_brand == b'qt  ' else b'dvcC'

    # Relabel the extracted atom to the correct name for this output format
    dvcc = dvcc[:4] + target_name + dvcc[8:]

    if target_name in buf_bytes:
        return True                        # already present

    hvcc_off = _mp4_box_off(bytes(buf), b'hvcC')
    if hvcc_off < 0:
        return False
    hvcc_sz = struct.unpack_from('>I', buf, hvcc_off)[0]
    ins = hvcc_off + hvcc_sz               # inject dvcC right after hvcC
    n   = len(dvcc)

    parents = _mp4_walk_parents(bytes(buf), ins)
    pre_insert = bytes(buf)    # Snapshot BEFORE insertion — prevents double-counting:
                               # _mp4_walk_parents returns boxes outermost-first, so
                               # without a snapshot, outer boxes would read already-
                               # incremented inner sizes and add n twice.

    buf[ins:ins] = dvcc

    for off in parents:
        sz = struct.unpack_from('>I', pre_insert, off)[0]   # always from pre-insert
        struct.pack_into('>I', buf, off, sz + n)

    # Update chunk offsets only when moov precedes mdat (faststart layout)
    moov_off = _mp4_box_off(bytes(buf), b'moov')
    mdat_off = _mp4_box_off(bytes(buf), b'mdat')
    if moov_off >= 0 and mdat_off >= 0 and moov_off < mdat_off:
        _mp4_shift_offsets(buf, ins, n)

    tmp = dst + '._tmp'
    try:
        with open(tmp, 'wb') as f:
            f.write(buf)
        os.replace(tmp, dst)
        return True
    except Exception:
        try:
            os.unlink(tmp)
        except Exception:
            pass
        return False

def _unique_output_path(stem: str, ext: str, base_dir: Path = None) -> Path:
    base_dir = base_dir or OUTPUT_DIR
    with _output_lock:
        dst = base_dir / f"{stem}{ext}"
        counter = 2
        while counter < 10000:
            try:
                dst.touch(exist_ok=False)
                return dst
            except FileExistsError:
                dst = base_dir / f"{stem}_{counter}{ext}"
                counter += 1
        raise RuntimeError(f"Cannot find unique output path for {stem!r}{ext} after 10000 attempts")

# Threads per concurrent CPU H265/H264 job.
# When _cpu_h_slots jobs run in parallel, each gets an equal share of cores.
# Example: 12 cores / 2 slots → 6 threads/job → 2 jobs × 6 = 12 threads (full CPU),
# but RAM per job drops from ~3.5 GB to ~1.5 GB → safe for concurrent 4K batches.
_cpu_h_per_job = max(4, _cpu_cores // max(1, _cpu_h_slots))
_X265_LOW_RAM_MB    = 4000   # threshold for 4K jobs
_X265_LOW_RAM_1080P = 1500   # threshold for 1080p jobs

def _build_x265_params(width: int = 0, height: int = 0) -> list:
    """
    Build -x265-params string for libx265.

    Concurrency strategy for 4K:
    - _cpu_h_slots >= 2: partition thread pool so each concurrent job uses
      _cpu_h_per_job threads instead of all cores. Total CPU remains 100%,
      but RAM per job halves → no OOM on concurrent 4K batches.
    - _cpu_h_slots == 1 (single job): use all cores freely; apply RAM-based
      limits only if the system is genuinely low on memory.

    This function is called AFTER acquiring the CPU semaphore (post-queue-wait),
    so free_ram_mb() reflects actual memory pressure at encode start time.
    """
    params  = "psy-rd=1.0:psy-rdoq=0.5:aq-mode=3:aq-strength=0.8:pme=1"
    is_4k   = (width >= 3840 or height >= 2160)
    ram_now = free_ram_mb()

    if is_4k and _cpu_h_slots >= 2:
        # Partition: each job gets _cpu_h_per_job threads.
        # frame-threads ≈ threads/4, capped at 4 (diminishing returns above).
        ft = max(1, min(4, _cpu_h_per_job // 4))
        params += (f":pools={_cpu_h_per_job}:frame-threads={ft}"
                   ":lookahead-threads=1:rc-lookahead=20:bframes=3")
    elif ram_now < _X265_LOW_RAM_MB:
        # Single-job or low-RAM fallback: apply conservative limits.
        if is_4k:
            params += ":pools=1:frame-threads=1:lookahead-threads=1:rc-lookahead=20:bframes=3"
        elif ram_now < _X265_LOW_RAM_1080P:
            params += ":pools=2:lookahead-threads=1:rc-lookahead=20"
    return ["-x265-params", params]

def _auto_crf(enc: str, width: int, height: int) -> int:
    """Return resolution-aware CRF for Auto mode.
    Uses Highest quality base (CRF_DEFAULTS[enc][0]) + resolution offset,
    so perceptual quality stays consistent across resolutions in a batch.
    av1_amf uses QVBR (inverted scale) → offset is subtracted.
    """
    base = CRF_DEFAULTS.get(enc, (24, 28, 32, 51))[0]
    px   = (width or 0) * (height or 0)
    for limit, off in CRF_AUTO_OFFSETS:
        if px < limit:
            return base - off if enc == "av1_amf" else base + off
    return base  # fallback — unreachable in practice (float('inf') always matches)



def _job_msg(job_id, msg, pct):
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["message"] = msg
            jobs[job_id]["progress"] = pct


def _dovi_cancelled(job_id) -> bool:
    with _jobs_lock:
        return jobs.get(job_id, {}).get("cancelled", False)


def _compress_dovi_pipeline(job_id, src, dst, enc, real_enc, enc_extra, crf,
                              mute, best_audio_idx, target_kbps, total_s, ppath,
                              meta_stream_indices=None, x265_params=None):
    """
    Dolby Vision preserving pipeline — 9 steps.

    Key insight: B-frame PTS/DTS live in the MOV container's CTTS table, not in
    the raw HEVC bitstream. By encoding to a full temp.mov first (so CTTS is
    written correctly), then binary-patching its video samples with RPU-injected
    data while keeping the CTTS/stts/stss tables intact, we preserve smooth
    B-frame playback on iOS AND keep Dolby Vision — without the -bf 0 penalty.
    """
    tmp = Path(TEMP_DIR) / f"dovi_{job_id}"
    tmp.mkdir(parents=True, exist_ok=True)
    try:
        raw_hevc  = tmp / "raw.hevc"     # copy of source video (for RPU extraction)
        rpu_bin   = tmp / "rpu.bin"      # extracted DV RPU metadata
        temp_mov  = tmp / "encoded.mov" # full encoded output (video+audio, correct timestamps)
        enc_hevc  = tmp / "enc.hevc"     # Annex-B HEVC extracted from temp_mov
        dv_hevc   = tmp / "dv.hevc"      # enc.hevc with RPU injected

        # Step 1: Extract raw HEVC from source (Annex-B, for RPU extraction)
        _job_msg(job_id, "Extracting Dolby Vision stream...", 2)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(src),
                            "-c:v", "copy", "-an", "-f", "hevc", str(raw_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 2: Extract DV RPU metadata
        _job_msg(job_id, "Extracting DV RPU metadata...", 5)
        r = subprocess.run([DOVI_TOOL_BIN, "extract-rpu",
                            "-i", str(raw_hevc), "-o", str(rpu_bin)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"RPU extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 3: Encode to temp.mov — full output with B-frames, audio, metadata.
        # B-frames are ALLOWED here: the container (temp.mov) correctly stores their
        # PTS/DTS in the CTTS box. We'll preserve this CTTS in the final output.
        _job_msg(job_id, f"Encoding DV base layer ({real_enc})...", 8)
        x265_mem = (x265_params or []) if real_enc == "libx265" else []
        audio_map = [] if mute else ["-map", f"0:a:{best_audio_idx}"]
        audio_enc = [] if mute else ["-c:a", "aac", "-b:a", f"{target_kbps}k",
                                     "-ar", "48000", "-ac", "2"]
        encode_cmd = ([FFMPEG_BIN, "-noautorotate"] + _hwaccel_args(real_enc) +
                      ["-i", str(src), "-map", "0:v:0"] + audio_map +
                      ["-c:v", real_enc] +
                      _quality_args(enc, crf) + enc_extra + x265_mem +
                      ["-tag:v", "hvc1"] + audio_enc +
                      ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"] +
                      ["-movflags", "use_metadata_tags+faststart"] +
                      ["-y", "-progress", ppath, "-stats_period", "0.5", "-nostats",
                       str(temp_mov)])
        rc, is_oom = _run_progress(encode_cmd, job_id, total_s, ppath)
        if rc != 0:
            return rc, is_oom

        if _dovi_cancelled(job_id): return -1, False

        # Step 4: Extract Annex-B HEVC from temp_mov (loses container timestamps,
        # but that's OK — we will restore them via binary patching in Step 7).
        _job_msg(job_id, "Extracting encoded HEVC stream...", 97)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(temp_mov),
                            "-c:v", "copy", "-bsf:v", "hevc_mp4toannexb",
                            "-an", "-f", "hevc", str(enc_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract from temp failed: {r.stderr.decode(errors='replace')[-1000:]}")

        if _dovi_cancelled(job_id): return -1, False

        # Step 5: Inject RPU NAL units into the encoded HEVC bitstream
        _job_msg(job_id, "Injecting Dolby Vision metadata...", 98)
        r = subprocess.run([DOVI_TOOL_BIN, "inject-rpu",
                            "-i", str(enc_hevc), "--rpu-in", str(rpu_bin),
                            "-o", str(dv_hevc)],
                           capture_output=True, timeout=300)
        if r.returncode != 0:
            raise RuntimeError(f"RPU inject failed: {r.stderr.decode(errors='replace')[-1000:]}")
        if not dv_hevc.exists() or dv_hevc.stat().st_size < 1024:
            raise RuntimeError(f"RPU inject produced empty output")

        # Step 6: Parse dv.hevc (Annex-B) into per-frame MP4-format samples
        _job_msg(job_id, "Parsing DV bitstream...", 98)
        dv_samples = _annexb_to_mp4_samples(str(dv_hevc))
        if not dv_samples:
            raise RuntimeError("Failed to parse DV HEVC samples from dv.hevc")

        # Step 7: Binary-patch temp_mov: swap video samples → dv_samples while
        # keeping CTTS/stts/stss intact. Write result to dst.
        _job_msg(job_id, "Muxing Dolby Vision output (timestamp-preserving)...", 99)
        if not _dovi_mux_with_timestamps(str(temp_mov), dv_samples, str(dst)):
            raise RuntimeError("_dovi_mux_with_timestamps failed — falling back not available")

        # Step 8: Transplant mebx (Core Media Metadata) tracks from source into dst.
        # These tracks carry dual-camera composition and Action Mode timed metadata
        # that iOS Photos uses to display feature icons and "Camera mặt trước" info.
        # FFmpeg corrupts mebx hdlr/stsd; _fix_mebx_tracks patches them back.
        if meta_stream_indices:
            _job_msg(job_id, "Copying metadata tracks...", 99)
            mebx_out = tmp / "with_mebx.mov"
            mebx_maps = []
            for idx in meta_stream_indices:
                mebx_maps += ["-map", f"1:{idx}"]
            mebx_cmd = ([FFMPEG_BIN, "-y",
                         "-i", str(dst), "-i", str(src),
                         "-map", "0", "-c", "copy"] +
                        mebx_maps +
                        ["-c:d", "copy",
                         "-tag:v", "hvc1",
                         "-movflags", "use_metadata_tags+faststart",
                         str(mebx_out)])
            try:
                r = subprocess.run(mebx_cmd, capture_output=True, timeout=120)
            except subprocess.TimeoutExpired:
                r = None  # graceful: keep dst as-is (core DV already written)
            if r is not None and r.returncode == 0 and mebx_out.exists() and mebx_out.stat().st_size > 1024:
                _fix_mebx_tracks(str(mebx_out))
                os.replace(str(mebx_out), str(dst))

        # Step 9: Copy dvvC/dvcC atom from source → output
        _job_msg(job_id, "Restoring Dolby Vision container signaling...", 99)
        _inject_dvcc(str(src), str(dst))

        return 0, False

    except RuntimeError as e:
        with _jobs_lock:
            jobs[job_id].update({"status": "error", "message": str(e)})
        return -1, False
    finally:
        shutil.rmtree(tmp, ignore_errors=True)


@job_guard
def _compress_job(job_id, src: Path, enc: str, crf: int,
                  out_ext: str, mute: bool = False, preserve_dv: bool = False,
                  crf_auto: bool = False):
    with _jobs_lock:
        if job_id not in jobs:
            return
        job = jobs[job_id]

    if not src.exists():
        with _jobs_lock:
            job["status"]  = "error"
            job["message"] = "Source file not found. Please reselect."
        return

    family = _get_codec_family(enc)
    t0     = None

    src_name = job.get("src_name", "video")
    src_stem = re.sub(r"[^\w\-. ]", "_", Path(src_name).stem)

    av1_warning = ""
    if family == "av1" and out_ext == ".mov":
        out_ext = ".mp4"
        av1_warning = " (auto-converted to MP4)"

    supported = CONTAINER_CODEC_SUPPORT.get(out_ext, ["h264"])
    if family not in supported:
        if out_ext == ".webm":
            enc = "libaom-av1"
            _msg = "⚠️ .WebM only supports AV1 → auto-switching to libaom-AV1"
        else:
            enc = "libx264"
            _msg = f"⚠️ {out_ext.upper()} does not support {family.upper()} → switching to H.264"
        family = _get_codec_family(enc)
        with _jobs_lock:
            job["message"] = _msg

    dst = _unique_output_path(src_stem, out_ext)
    if not _check_path_len_job(dst, job_id):
        try: dst.unlink()
        except: pass
        try: src.unlink()
        except: pass
        return  # ppath not yet created at this point

    real_enc, enc_extra = _get_real_enc_extra(enc)

    pfd, ppath = tempfile.mkstemp(suffix=".txt", prefix="vs_")
    os.close(pfd)

    probe_data     = _probe(str(src))
    total_s        = float(probe_data.get("format", {}).get("duration") or 0)

    # Extract video dimensions and rotation for resolution-aware params and metadata sync
    video_w = video_h = 0
    src_rotation = 0
    for _s in probe_data.get("streams", []):
        if _s.get("codec_type") == "video":
            video_w = int(_s.get("width",  0) or 0)
            video_h = int(_s.get("height", 0) or 0)
            src_rotation = _get_rotation(_s)
            break
    if crf_auto:
        crf = _auto_crf(enc, video_w, video_h)
        with _jobs_lock:
            if job_id in jobs:
                jobs[job_id]["effective_crf"] = crf
                jobs[job_id]["crf_auto"] = True

    # x265_params will be rebuilt inside the encode loop (after queue wait)
    # so it reads free RAM at the moment encoding actually starts — not when
    # multiple siblings may not have loaded yet.
    _is_x265   = (real_enc == "libx265")
    x265_params = []  # populated inside retry loop for libx265; empty for all others

    # Dolby Vision detection
    is_dv, dv_profile = _get_dv_info(probe_data)
    use_dovi = is_dv and _DOVI_AVAILABLE and preserve_dv

    # Store source metadata snapshot
    src_exif = _get_exif_snapshot(str(src))
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["src_exif"] = src_exif

    best_audio_idx = 0
    best_audio_br  = 0
    _apac_idx      = None  # last-resort fallback if apac is the only stream
    audio_streams_raw = [
        s for s in probe_data.get("streams", [])
        if s.get("codec_type") == "audio" and s.get("codec_name") and s.get("codec_name") != "none"
    ]
    audio_streams = list(enumerate(audio_streams_raw))
    for stream_idx, s in audio_streams:
        # Prefer decodable AAC (mp4a) over Apple Lossless/spatial (apac).
        # apac is a 3-4ch spatial stream: re-encoding to stereo AAC without proper
        # downmix causes iOS playback stutter (timing desync with mebx tracks).
        # Track apac separately — only use if no other decodable stream exists.
        codec_tag = s.get("codec_tag_string", "").lower()
        if codec_tag == "apac":
            if _apac_idx is None:
                _apac_idx = stream_idx
            continue
        br = 0
        raw_br = s.get("bit_rate")
        if raw_br:
            try: br = int(raw_br)
            except: pass
        if br < 10000:
            tags = s.get("tags", {})
            bps  = tags.get("BPS") or tags.get("BPS-eng")
            if bps:
                try: br = int(bps)
                except: pass
        if br < 10000:
            ch = s.get("channels", 0)
            sr = s.get("sample_rate", 0)
            if ch and sr:
                br = ch * int(sr) * 16
        if br > best_audio_br:
            best_audio_br  = br
            best_audio_idx = stream_idx

    # If no decodable stream found, fall back to apac.
    # best_audio_br stays 0 → target_kbps correctly falls back to 192k default.
    if best_audio_br == 0 and _apac_idx is not None:
        best_audio_idx = _apac_idx

    # Detect Apple metadata tracks (mebx, tmcd) — carry iPhone feature signaling
    # such as Action Mode, Dual Camera, GPS track. Copy them as-is to preserve
    # iOS Photos feature badges without re-encoding.
    meta_stream_indices = []
    for s in probe_data.get("streams", []):
        if s.get("codec_type") in ("data", "tmcd"):
            handler = s.get("tags", {}).get("handler_name", "")
            tag = s.get("codec_tag_string", "")
            # mebx = CoreMedia timed metadata; tmcd = timecode
            if tag in ("mebx", "tmcd") or any(h in handler for h in _META_HANDLER_TAGS):
                meta_stream_indices.append(s["index"])

    cmd = [FFMPEG_BIN, "-noautorotate"] + _hwaccel_args(real_enc) + ["-i", str(src), "-map", "0:v:0"]

    if not mute and audio_streams:
        audio_map = (f"0:a:{best_audio_idx}"
                     if len(audio_streams) > 1 else "0:a:0")
        cmd += ["-map", audio_map]

    for idx in meta_stream_indices:
        cmd += ["-map", f"0:{idx}"]

    cmd += ["-c:v", real_enc]
    cmd += _quality_args(enc, crf)
    cmd += enc_extra
    cmd += x265_params

    if family == "h265" and out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-tag:v", "hvc1"]

    if meta_stream_indices:
        cmd += ["-c:d", "copy"]

    target_kbps = 192
    if not mute and audio_streams:
        target_kbps = max(128, min(320, int(best_audio_br / 1000))) if best_audio_br > 0 else 192
        audio_br_str = f"{target_kbps}k"
        cmd += [
            "-c:a", "aac",
            "-b:a", audio_br_str,
            "-ar",  "48000",
            "-ac",  "2",
        ]

    cmd += ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"]
    if not mute and audio_streams:
        cmd += ["-map_metadata:s:a", "0:s:a"]

    if out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-movflags", "use_metadata_tags+faststart"]
    elif out_ext != ".mkv":
        cmd += ["-movflags", "+faststart"]

    cmd += ["-y", "-progress", ppath,
            "-stats_period", "0.5", "-nostats", str(dst)]

    oom_retries = 0
    while True:
        with _jobs_lock:
            job["status"]  = "waiting"
            job["message"] = "Waiting in queue..."
        with _pick_sem(real_enc):
            with _jobs_lock:
                if job.get("cancelled"): return
            t0 = time.time()
            # Rebuild x265 thread limits now (after queue wait) so free_ram_mb()
            # reflects the actual pressure from other running jobs.
            if _is_x265:
                x265_params = _build_x265_params(video_w, video_h)
            with _jobs_lock:
                job["status"]  = "processing"
                crf_note = f" — CRF {crf} (Auto)" if crf_auto else ""
                job["message"] = f"Encoding ({real_enc}){crf_note}{av1_warning}{' — no audio' if mute else ''}..."
                job["encode_start_time"] = t0
            try:
                if use_dovi:
                    rc, is_oom = _compress_dovi_pipeline(
                        job_id, src, dst, enc, real_enc, enc_extra, crf,
                        mute, best_audio_idx, target_kbps, total_s, ppath,
                        meta_stream_indices, x265_params=x265_params)
                else:
                    final_cmd = cmd.copy()
                    if _is_x265 and x265_params:
                        final_cmd = final_cmd[:-1] + x265_params + [final_cmd[-1]]
                    rc, is_oom = _run_progress(final_cmd, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            with _jobs_lock:
                if job.get("cancelled"): return
            oom_retries += 1
            if oom_retries >= 3:
                if use_dovi:
                    # DV pipeline OOM → fall back to regular encode (no DV)
                    use_dovi = False
                    oom_retries = 0
                    with _jobs_lock:
                        job["status"]  = "waiting"
                        job["message"] = "⚠️ DV preservation failed (low RAM) — encoding as HLG..."
                    try: dst.unlink()
                    except: pass
                    time.sleep(2)
                else:
                    with _jobs_lock:
                        job.update({"status": "error", "message": "Insufficient RAM — failed after 3 retries. Too many 4K jobs running at once."})
                    try: dst.unlink()
                    except: pass
                    return
            else:
                with _jobs_lock:
                    if job.get("cancelled"): return
                    # Distinguish GPU VRAM exhaustion from CPU RAM exhaustion
                    ram_msg = "RAM" if real_enc in ("libx265", "libx264", "libsvtav1", "libaom-av1") else "GPU"
                    job["status"]  = "waiting"
                    job["message"] = f"{ram_msg} full, retrying ({oom_retries}/3)..."
                time.sleep(3)
                with _jobs_lock:
                    if job.get("cancelled"): return

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        # _fix_mebx_tracks already runs inside _compress_dovi_pipeline (step 8)
        # for DV videos, so skip the redundant second pass.
        if meta_stream_indices and not use_dovi:
            _fix_mebx_tracks(str(dst))
        _sync_meta(str(src), str(dst), src_rotation)

        dst_exif = _get_exif_snapshot(str(dst))
        with _jobs_lock:
            if job_id in jobs:
                jobs[job_id]["dst_exif"] = dst_exif

        ela      = time.time() - t0 if t0 else 0
        enc_time = _fmt_time(ela)

        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = out_duration = "N/A"

        try:
            d = float(out_data.get("format", {}).get("duration", 0))
            if d > 0:
                out_duration = f"{int(d//60)}:{int(d%60):02d}"
        except: pass

        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                w, h = s.get('width', '?'), s.get('height', '?')
                rotation = _get_rotation(s)
                if rotation in (90, 270) and w != '?' and h != '?':
                    out_res = f"{h}x{w}"
                else:
                    out_res = f"{w}x{h}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break

        if mute:
            out_audio = "Muted"
        else:
            for s in out_data.get("streams", []):
                if s.get("codec_type") == "audio":
                    ac     = s.get("codec_name", "N/A").upper()
                    br     = s.get("bit_rate")
                    br_int = int(br) if br else 0
                    if br_int < 10000:
                        tags   = s.get("tags", {})
                        br_tag = tags.get("BPS") or tags.get("BPS-eng")
                        if br_tag:
                            try: br_int = int(br_tag)
                            except: pass
                    if br_int < 10000:
                        br_int = target_kbps * 1000
                    brs       = f" {br_int//1000}kbps" if br_int > 10000 else ""
                    out_audio = f"{ac}{brs}"
                    break

        result = {
            "status":        "done",
            "progress":      100,
            "message":       "Completed!",
            "output":        str(dst),
            "sz_before":     _fmt_size(str(src)),
            "sz_after":      _fmt_size(str(dst)),
            "encode_time":   enc_time,
            "out_res":       out_res,
            "out_fps":       out_fps,
            "out_audio":     out_audio,
            "out_codec":     real_enc,
            "out_duration":  out_duration,
            "src_is_dv":     is_dv,
            "src_dv_profile":dv_profile,
            "dst_is_dv":     _detect_dolby_vision(out_data),
        }
        if family == "av1" and av1_warning:
            result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 does not support MOV format — auto-converted to MP4"
        with _jobs_lock:
            job.update(result)
            log_name = job.get("log_file")
        if log_name:
            try: (Path(OUTPUT_DIR) / log_name).unlink(missing_ok=True)
            except: pass
    else:
        with _jobs_lock:
            if not job.get("cancelled") and job.get("status") != "error":
                job.update({"status": "error", "message": "Compression failed!"})
        try: dst.unlink()
        except: pass

    try: src.unlink()
    except: pass
    try: os.unlink(ppath)   # no-op if _run_progress already cleaned it
    except: pass

@job_guard
def _social_video_job(job_id, src: Path, preset: dict, preset_key: str):
    job = jobs[job_id]

    stem = re.sub(r"[^\w\-. ]", "_", src.stem)
    dst  = _unique_output_path(f"{stem}_{preset_key}", ".mp4")

    if not _check_path_len_job(dst, job_id):
        try: dst.unlink()
        except: pass
        try: src.unlink()
        except: pass
        return

    total_s = _duration(str(src))

    filters = []
    scale, fps = preset.get("scale"), preset.get("fps")
    if scale == "9:16":
        filters.append(
            "scale=1080:1920:force_original_aspect_ratio=decrease,"
            "pad=1080:1920:(ow-iw)/2:(oh-ih)/2:black,format=yuv420p"
        )
    elif scale == "1080":
        filters.append("scale=1080:-2,format=yuv420p")
    if fps:
        filters.append(f"fps={fps}")

    enc = preset["encoder"]
    gpu, _, _ = get_gpu_info()
    gpu_map = {
        "libx264": ["h264_nvenc","h264_amf"],
        "libx265": ["hevc_nvenc","hevc_amf"],
    }
    for gpu_enc in gpu_map.get(enc, []):
        if gpu_enc in gpu:
            enc = gpu_enc
            break

    crf = preset["crf"]

    pfd, ppath = tempfile.mkstemp(suffix=".txt"); os.close(pfd)

    cmd = [FFMPEG_BIN] + (_hwaccel_args(enc) if not filters else []) + ["-i", str(src)]
    if filters: cmd += ["-vf", ",".join(filters)]
    cmd += ["-c:v", enc] + _quality_args(enc, crf)
    if "265" in enc or "hevc" in enc: cmd += ["-tag:v","hvc1"]
    ab   = preset["audio"]
    cmd += [
        "-c:a", "aac",
        "-b:a", ab,
        "-ar",  "48000",
        "-ac",  "2",
        "-map_metadata","0",
        "-movflags","use_metadata_tags+faststart",
        "-y","-progress",ppath,"-stats_period","0.5","-nostats", str(dst)
    ]

    oom_retries = 0
    while True:
        with _jobs_lock:
            job["status"] = "waiting"
            job["message"] = "Waiting in queue..."

        t0 = None
        with _pick_sem(enc):
            with _jobs_lock:
                if job.get("cancelled"): return
                job["status"] = "processing"
                job["message"] = f"Encoding ({enc})..."
            t0 = time.time()
            try:
                rc, is_oom = _run_progress(cmd, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            with _jobs_lock:
                if job.get("cancelled"): return
            oom_retries += 1
            ram_msg = "RAM" if enc in ("libx265", "libx264", "libsvtav1", "libaom-av1") else "VRAM"
            if oom_retries >= 3:
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Insufficient {ram_msg} — failed after 3 retries."})
                try: dst.unlink()
                except: pass
                return
            with _jobs_lock:
                gpu_msg = "RAM" if enc in ("libx265", "libx264", "libsvtav1", "libaom-av1") else "GPU"
                job["status"] = "waiting"
                job["message"] = f"{gpu_msg} full, retrying ({oom_retries}/3)..."
            time.sleep(3)
            with _jobs_lock:
                if job.get("cancelled"): return

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024
    ela      = time.time() - t0 if t0 else 0
    enc_time = _fmt_time(ela)

    if ok:
        _sync_meta(str(src), str(dst))

        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = "N/A"
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "audio":
                ac = s.get("codec_name", "N/A").upper()
                out_audio = f"{ac}"
                break

        with _jobs_lock:
            job.update({
                "status":"done","progress":100,"message":"Completed!",
                "output":str(dst),
                "sz_before":_fmt_size(str(src)),
                "sz_after": _fmt_size(str(dst)),
                "encode_time": enc_time,
                "out_res": out_res,
                "out_fps": out_fps,
                "out_audio": out_audio,
                "out_enc": preset.get("label", "Preset")
            })
    else:
        with _jobs_lock:
            job.update({"status":"error","message":"Export failed!"})
        try: dst.unlink()
        except: pass

    except: pass
