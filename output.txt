def _sanitize_upload_filename(raw: str) -> str:
    name = re.sub(r'[\\/*?:"<>|]', "_", raw)
    name = re.sub(r'[\x00-\x1f]', '', name).strip()
    return name

def _resolve_upload_file(server_file: str):
    p = (UPLOAD_DIR / server_file).resolve()
    try:
        p.relative_to(UPLOAD_DIR.resolve())
    except ValueError:
        return None
    return p

def _check_disk_space():
    total, used, free = shutil.disk_usage(UPLOAD_DIR)
    free_gb = free / (1024 ** 3)
    if free_gb < MIN_FREE_DISK_GB:
        return False, f"Server storage is almost full ({free_gb:.1f}GB left). Please try again later!"
    return True, ""

def _fmt_size(path):
    return _fmt_size_bytes(os.path.getsize(path))

def _fmt_size_bytes(b):
    if b < 1024**2:   return f"{b/1024:.1f} KB"
    if b < 1024**3:   return f"{b/1024**2:.1f} MB"
    return f"{b/1024**3:.2f} GB"

def _fmt_size_mb(path):
    return os.path.getsize(path) / 1024**2

def _fmt_time(s):
    if s < 60: return f"{s:.0f}s"
    return f"{int(s//60)}:{int(s%60):02d}"

def _parse_dl_size(text: str) -> int:
    m = re.search(r'of\s+~?\s*([\d.]+)\s*(GiB|MiB|KiB|B)\b', text)
    if not m: return 0
    mult = {"GiB": 1 << 30, "MiB": 1 << 20, "KiB": 1 << 10, "B": 1}
    return int(float(m.group(1)) * mult.get(m.group(2), 0))

def _probe(path):
    if not FFPROBE_BIN: return {}
    try:
        r = subprocess.run(
            [FFPROBE_BIN,"-v","quiet","-print_format","json",
             "-show_streams","-show_format", path],
            capture_output=True, text=True, timeout=30)
    except subprocess.TimeoutExpired:
        return {}
    try: return json.loads(r.stdout)
    except: return {}

def _extract_ui_meta(tmp_path):
    data = _probe(str(tmp_path))
    info = {"duration": "N/A", "codec": "N/A", "res": "N/A", "fps": "N/A", "audio": "N/A"}
    if not data: return info

    try:
        d = float(data["format"]["duration"])
        info["duration"] = f"{int(d//60)}:{int(d%60):02d}"
    except: pass

    for s in data.get("streams", []):
        if s.get("codec_type") == "video":
            raw = s.get("codec_name", "").lower()
            info["codec"] = CODEC_LABELS.get(raw, raw.upper())
            w, h = s.get('width', '?'), s.get('height', '?')
            rotation = 0
            if "rotate" in s.get("tags", {}):
                try: rotation = abs(int(s["tags"]["rotate"]))
                except: pass
            else:
                for sd in s.get("side_data_list", []):
                    if sd.get("side_data_type") == "Display Matrix":
                        try: rotation = abs(int(sd.get("rotation", 0)))
                        except: pass
            if rotation in (90, 270) and w != '?' and h != '?': info["res"] = f"{h}x{w}"
            else: info["res"] = f"{w}x{h}"
            r = s.get("r_frame_rate", "0/1")
            if "/" in r:
                n, d2 = r.split("/")
                if int(d2): info["fps"] = f"{round(int(n)/int(d2),1)}"
            break

    for s in data.get("streams", []):
        if s.get("codec_type") == "audio":
            raw = s.get("codec_name", "").lower()
            ac = AUDIO_LABELS.get(raw, raw.upper())
            br = s.get("bit_rate")
            br_int = int(br) if br else 0
            if br_int < 10000:
                br_tag = s.get("tags", {}).get("BPS") or s.get("tags", {}).get("BPS-eng")
                if br_tag:
                    try: br_int = int(br_tag)
                    except: pass
            brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
            info["audio"] = f"{ac}{brs}"
            break

    info["is_dolby_vision"], info["dv_profile"] = _get_dv_info(data)

    # Detect Apple iPhone-specific shooting modes from QuickTime format tags
    # These tags come from ffprobe format.tags — no extra ExifTool call needed.
    fmt_tags = data.get("format", {}).get("tags", {})

    # Dual Camera: front-and-back-camera-composition = "1" means iPhone filmed with
    # both front and rear cameras simultaneously (Dual Cam / FaceTime Camera feature)
    info["is_dual_camera"] = (
        fmt_tags.get("com.apple.quicktime.front-and-back-camera-composition") == "1"
    )

    # Action Mode: action-video.stabilization-strength present means EIS was active
    # (value 4 = max stabilization, any non-zero value = Action Mode on)
    stab = fmt_tags.get("com.apple.quicktime.action-video.stabilization-strength", "")
    try:
        info["is_action_mode"] = int(stab) > 0
    except (ValueError, TypeError):
        info["is_action_mode"] = False

    return info

def _duration(path):
    if not FFPROBE_BIN: return 0.0
    try:
        r = subprocess.run(
            [FFPROBE_BIN,"-v","quiet","-show_entries","format=duration",
             "-of","default=noprint_wrappers=1:nokey=1", path],
            capture_output=True, text=True, timeout=30)
    except subprocess.TimeoutExpired:
        return 0.0
    try: return float(r.stdout.strip())
    except: return 0.0

def _read_progress(fpath):
    try:
        txt  = Path(fpath).read_text(encoding="utf-8", errors="ignore")
        done = "progress=end" in txt
        t    = 0.0
        for ln in reversed(txt.splitlines()):
            if ln.startswith("out_time_ms="):
                try:
                    ms = int(ln.split("=")[1])
                    if ms > 0: t = ms/1_000_000
                except: pass
                break
        return t, done
    except: return 0.0, False

def _sync_meta(src: str, dst: str) -> bool:
    if not EXIFTOOL_BIN: return False

    # Step 1: Copy critical Apple QuickTime Keys atoms explicitly.
    # IMPORTANT: Do NOT use -Keys:all here — it copies Keys:FocalLength (physical mm
    # e.g. 6.765mm) which overrides the display-formatted Keys:CameraLensModel
    # (e.g. "24 mm f/1.78") causing iOS to show wrong focal length and lose the
    # "Apple iPhone xx" camera header (shows "Không có thông tin máy ảnh" instead).
    cmd1 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-Keys:GPSCoordinates",
        "-Keys:LocationBody",
        "-Keys:LocationDate",
        "-Keys:LocationName",
        "-Keys:Make",
        "-Keys:Model",
        "-Keys:Software",
        "-Keys:CameraLensModel",
        "-Keys:CreationDate",
        "-Keys:ContentIdentifier",
        "-Keys:HDRVideoFormat",
        "-Keys:LocationAccuracyHorizontal",
        "-Keys:FullFrameRatePlaybackIntent",
        "-Keys:Action-videoStabilization-strength",
        "-Keys:Metadata1",
        "-Keys:Front-and-back-camera-composition",
        # Mirror critical Keys into legacy UserData atoms (compatibility with older
        # players and iOS versions that read Make/Model/GPS from udta, not Keys)
        "-UserData:Make<Keys:Make",
        "-UserData:Model<Keys:Model",
        "-UserData:GPSCoordinates<Keys:GPSCoordinates",
        "-UserData:CreationDate<Keys:CreationDate",
        "-overwrite_original",
        dst
    ]
    r1 = subprocess.run(cmd1, capture_output=True, text=True)

    # Step 2: Copy all remaining metadata (excluding Keys to avoid overwriting step 1).
    cmd2 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-all:all",
        "--FileModifyDate",
        "--FileCreateDate",
        "--Keys:all",       # Exclude Keys — already handled in step 1
        "-overwrite_original",
        dst
    ]
    r2 = subprocess.run(cmd2, capture_output=True, text=True, encoding="utf-8", errors="replace")

    return r1.returncode == 0 or r2.returncode == 0


def _detect_action_mode(path: str) -> bool:
    """Detect iPhone Action Mode via ExifTool Keys:CaptureMode atom.
    Returns True if the video was recorded with Action Mode (EIS stabilization)."""
    if not EXIFTOOL_BIN:
        return False
    try:
        r = subprocess.run(
            [EXIFTOOL_BIN, "-api", "LargeFileSupport=1", "-j",
             "-Keys:CaptureMode",
             str(path)],
            capture_output=True, text=True, timeout=10
        )
        data = json.loads(r.stdout or "[]")
        if not data:
            return False
        capture_mode = str(data[0].get("CaptureMode", "")).lower()
        return "action" in capture_mode
    except Exception:
        return False


def _get_dv_info(probe_data) -> tuple:
    """Return (is_dolby_vision, dv_profile) from probe data."""
    for stream in probe_data.get("streams", []):
        for sd in stream.get("side_data_list", []):
            sdt = sd.get("side_data_type", "").lower()
            if "dowi" in sdt or "dovi" in sdt:
                return True, sd.get("dv_profile", 0)
    return False, 0


def _detect_dolby_vision(probe_data) -> bool:
    return _get_dv_info(probe_data)[0]


def _get_exif_snapshot(path) -> dict:
    """Run ExifTool on path, return selected Apple Keys atoms + HDR metadata.
    Uses an explicit field list — NOT -Keys:all — to avoid copying Keys:FocalLength
    (physical mm, e.g. 6.765mm) which conflicts with the display-formatted
    Keys:CameraLensModel atom (e.g. '24 mm f/1.78') causing iOS to show wrong
    focal length and lose the device camera header."""
    if not EXIFTOOL_BIN:
        return {}
    try:
        r = subprocess.run(
            [EXIFTOOL_BIN, "-api", "LargeFileSupport=1",
             "-a", "-u", "-G1", "-j",
             # Apple QuickTime Keys — explicit list (no Keys:FocalLength noise)
             "-Keys:Make", "-Keys:Model", "-Keys:Software",
             "-Keys:CameraLensModel",
             "-Keys:GPSCoordinates", "-Keys:LocationName",
             "-Keys:LocationBody", "-Keys:LocationDate",
             "-Keys:CreationDate", "-Keys:ContentIdentifier",
             "-Keys:HDRVideoFormat",
             # Video color metadata (for HDR/DV comparison)
             "-ColorPrimaries", "-TransferCharacteristics", "-MatrixCoefficients",
             str(path)],
            capture_output=True, text=True, timeout=30
        )
        data = json.loads(r.stdout or "[]")
        return data[0] if data else {}
    except Exception:
        return {}


# ── Dolby Vision: Annex-B parser + timestamp-preserving mux ──────────────────

def _annexb_to_mp4_samples(hevc_path: str) -> list:
    """
    Parse dovi_tool inject-rpu Annex-B output into per-frame MP4 samples.
    Each sample = concatenation of 4-byte-BE-length-prefixed NAL units.
    Frames are delimited by AUD NAL units (HEVC type 35) which dovi_tool adds
    by default between every frame.
    Returns [] on error.
    """
    try:
        with open(hevc_path, 'rb') as f:
            raw = f.read()
    except OSError:
        return []

    total = len(raw)
    # ── find all start codes (00 00 01 or 00 00 00 01) ──────────────────────
    sc_pos = []   # (first_byte_after_sc, sc_length)
    i = 0
    while i < total - 2:
        if raw[i] == 0 and raw[i + 1] == 0:
            if i + 3 < total and raw[i + 2] == 0 and raw[i + 3] == 1:
                sc_pos.append((i + 4, 4)); i += 4; continue
            if raw[i + 2] == 1:
                sc_pos.append((i + 3, 3)); i += 3; continue
        i += 1

    if not sc_pos:
        return []

    # ── extract NAL unit data for each SC position ───────────────────────────
    nals = []   # (nal_type, nal_bytes)
    for idx, (nal_start, sc_len) in enumerate(sc_pos):
        nal_end = sc_pos[idx + 1][0] - sc_pos[idx + 1][1] if idx + 1 < len(sc_pos) else total
        nal = raw[nal_start:nal_end]
        # strip trailing emulation-prevention zeros
        while len(nal) > 1 and nal[-1] == 0:
            nal = nal[:-1]
        if nal:
            nals.append(((nal[0] >> 1) & 0x3F, nal))

    # ── group by AUD (type 35) → one MP4 sample per frame ───────────────────
    samples, cur = [], []
    for nal_type, nal in nals:
        if nal_type == 35:                     # AUD: signals start of new AU
            if cur:
                samples.append(b''.join(struct.pack('>I', len(nd)) + nd
                                        for _, nd in cur))
                cur = []
        else:
            cur.append((nal_type, nal))
    if cur:
        samples.append(b''.join(struct.pack('>I', len(nd)) + nd for _, nd in cur))

    return samples


def _mp4_walk_top(buf, start: int, end: int):
    """Iterate top-level boxes in buf[start:end], yield (fourcc, data_start, box_end)."""
    off = start
    while off + 8 <= end:
        sz = struct.unpack_from('>I', buf, off)[0]
        fc = bytes(buf[off + 4:off + 8])
        hdr = 8
        if sz == 1:
            if off + 16 > end: break
            sz = struct.unpack_from('>Q', buf, off + 8)[0]; hdr = 16
        if sz == 0 or off + sz > end: break
        yield fc, off + hdr, off + sz
        off += sz


_DV_CONTAINER_FCS = frozenset({
    b'moov', b'trak', b'mdia', b'minf', b'stbl', b'edts', b'dinf', b'udta',
    b'meta', b'ilst', b'wide',
})


def _mp4_find_in_range(buf, start: int, end: int, target: bytes) -> tuple:
    """DFS search for target fourcc in buf[start:end]. Returns (box_start, box_end) or (-1,-1)."""
    for fc, ds, be in _mp4_walk_top(buf, start, end):
        if fc == target:
            return ds - 8, be          # box_start includes the 8-byte header
        if fc in _DV_CONTAINER_FCS:
            r = _mp4_find_in_range(buf, ds, be, target)
            if r[0] >= 0:
                return r
    return -1, -1


def _mp4_find_in_trak(buf, moov_s: int, moov_e: int,
                       handler: bytes, target: bytes) -> tuple:
    """Find target fourcc inside the trak with matching handler type.
    Handles both ISOBMFF (pre_defined=0, handler at box+16) and QuickTime
    legacy format (component_type at box+12, component_subtype at box+12+4=16
    OR no pre_defined, handler at box+12).
    We check both +12 and +16 for robustness.
    """
    for fc, ds, be in _mp4_walk_top(buf, moov_s, moov_e):
        if fc != b'trak':
            continue
        ho, he = _mp4_find_in_range(buf, ds, be, b'hdlr')
        if ho < 0:
            continue
        # Try offset +16 (ISOBMFF: ver/flags + pre_defined + handler_type)
        # and +12 (QuickTime: ver/flags + handler_type, no pre_defined)
        matched = False
        for ht_off in (ho + 16, ho + 12):
            if ht_off + 4 <= len(buf) and bytes(buf[ht_off: ht_off + 4]) == handler:
                matched = True
                break
        if matched:
            return _mp4_find_in_range(buf, ds, be, target)
    return -1, -1


def _parse_stco_buf(buf, box_s: int) -> list:
    """Parse stco/co64 at buf[box_s], return list of chunk offsets."""
    fc = bytes(buf[box_s + 4: box_s + 8])
    is64 = (fc == b'co64')
    d = box_s + 8                                     # skip box header → version+flags
    count = struct.unpack_from('>I', buf, d + 4)[0]   # skip 4-byte ver/flags → count
    entry_sz, fmt = (8, '>Q') if is64 else (4, '>I')
    return [struct.unpack_from(fmt, buf, d + 8 + i * entry_sz)[0] for i in range(count)]


def _parse_stsz_buf(buf, box_s: int) -> list:
    """Parse stsz at buf[box_s], return list of per-sample sizes."""
    d = box_s + 8                                     # ver+flags
    fixed = struct.unpack_from('>I', buf, d + 4)[0]  # fixed sample-size field
    count = struct.unpack_from('>I', buf, d + 8)[0]
    if fixed:
        return [fixed] * count
    return [struct.unpack_from('>I', buf, d + 12 + i * 4)[0] for i in range(count)]


def _parse_stsc_buf(buf, box_s: int) -> list:
    """Parse stsc, return list of (first_chunk_1based, samples_per_chunk, desc_idx)."""
    d = box_s + 8
    count = struct.unpack_from('>I', buf, d + 4)[0]
    return [(struct.unpack_from('>I', buf, d + 8 + i * 12)[0],
             struct.unpack_from('>I', buf, d + 12 + i * 12)[0],
             struct.unpack_from('>I', buf, d + 16 + i * 12)[0])
            for i in range(count)]


def _stsc_samples_per_chunk(stsc: list, n_chunks: int) -> list:
    """Expand stsc into a per-chunk list of sample counts."""
    result = []
    for ci in range(1, n_chunks + 1):
        spc = stsc[-1][1] if stsc else 1
        for j in range(len(stsc)):
            fc = stsc[j][0]
            nfc = stsc[j + 1][0] if j + 1 < len(stsc) else n_chunks + 1
            if fc <= ci < nfc:
                spc = stsc[j][1]; break
        result.append(spc)
    return result


def _dovi_mux_with_timestamps(base_mov: str, dv_samples: list, dst: str) -> bool:
    """
    Rebuild a MOV file that has:
      • Video bitstream data from dv_samples  (MP4-format, one bytes per frame)
      • Video timestamps (stts/ctts/stss)     preserved from base_mov's video trak
      • Audio data and timing                 preserved from base_mov's audio trak

    Strategy: the stsz and stco boxes have the same NUMBER of entries before and
    after RPU injection (same frame count, same chunk count), so their box SIZE
    does not change. We can therefore patch them in-place inside the bytearray
    copy of base_mov, then replace the mdat section with new video + audio data.

    Returns True on success.
    """
    try:
        with open(base_mov, 'rb') as f:
            buf = bytearray(f.read())
    except OSError:
        return False

    # ── locate top-level boxes (ftyp, moov, mdat) ────────────────────────────
    moov_s = moov_e = mdat_s = mdat_e = -1
    for fc, ds, be in _mp4_walk_top(buf, 0, len(buf)):
        if fc == b'moov': moov_s, moov_e = ds - 8, be
        elif fc == b'mdat': mdat_s, mdat_e = ds - 8, be
    if moov_s < 0 or mdat_s < 0:
        return False

    moov_body_s = moov_s + 8
    moov_body_e = moov_e

    # ── locate video track stbl entries ──────────────────────────────────────
    vs_s, vs_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stsz')
    vc_s, vc_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stco')
    vsc_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'stsc')
    # also try co64
    if vc_s < 0:
        vc_s, vc_e = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'vide', b'co64')

    if vs_s < 0 or vc_s < 0 or vsc_s < 0:
        return False

    v_stsz = _parse_stsz_buf(buf, vs_s)
    v_stco = _parse_stco_buf(buf, vc_s)
    v_stsc = _parse_stsc_buf(buf, vsc_s)
    n_video = len(v_stsz)
    n_chunks = len(v_stco)

    # Safety: frame count must match
    if len(dv_samples) != n_video:
        # Trim to min; mild mismatch (off-by-one) is tolerable
        min_n = min(len(dv_samples), n_video)
        if abs(len(dv_samples) - n_video) > 2:
            return False
        dv_samples = dv_samples[:min_n]
        n_video = min_n

    new_sizes = [len(s) for s in dv_samples]

    # ── build new video chunks grouped by stsc structure ─────────────────────
    spc_list = _stsc_samples_per_chunk(v_stsc, n_chunks)
    video_chunks = []
    si = 0
    for spc in spc_list:
        chunk = b''.join(dv_samples[si + j] for j in range(spc)
                         if si + j < len(dv_samples))
        video_chunks.append(chunk)
        si += spc

    # ── collect audio chunks from the original mdat ───────────────────────────
    audio_chunks = []
    ac_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stco')
    if ac_s < 0:
        ac_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'co64')
    as_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stsz')
    asc_s, _ = _mp4_find_in_trak(buf, moov_body_s, moov_body_e, b'soun', b'stsc')

    if ac_s >= 0 and as_s >= 0 and asc_s >= 0:
        a_stco = _parse_stco_buf(buf, ac_s)
        a_stsz = _parse_stsz_buf(buf, as_s)
        a_stsc = _parse_stsc_buf(buf, asc_s)
        a_spc_list = _stsc_samples_per_chunk(a_stsc, len(a_stco))
        asi = 0
        for ci, spc in enumerate(a_spc_list):
            chunk_sz = sum(a_stsz[asi + j] for j in range(spc)
                           if asi + j < len(a_stsz))
            chunk_off = a_stco[ci]
            if 0 < chunk_off < len(buf) and chunk_off + chunk_sz <= len(buf):
                audio_chunks.append(buf[chunk_off: chunk_off + chunk_sz])
            else:
                audio_chunks.append(b'')   # safety fallback
            asi += spc

    # ── compute new mdat layout ───────────────────────────────────────────────
    # mdat position stays the same (moov size unchanged, since stsz/stco entries
    # count is the same → box sizes unchanged → moov size unchanged).
    new_mdat_off = mdat_s
    new_mdat_body_s = new_mdat_off + 8         # after 8-byte mdat box header

    # video chunk offsets in new mdat
    cur = new_mdat_body_s
    new_v_stco = []
    for chunk in video_chunks:
        new_v_stco.append(cur)
        cur += len(chunk)

    # audio chunk offsets in new mdat (sequentially after video)
    new_a_stco = []
    for chunk in audio_chunks:
        new_a_stco.append(cur)
        cur += len(chunk)

    # ── patch video stsz entries in-place ────────────────────────────────────
    # stsz fixed-size field is at vs_s+12; if 0 → variable-size (entries at vs_s+20+)
    old_fixed = struct.unpack_from('>I', buf, vs_s + 12)[0]
    if old_fixed == 0:
        for i, sz_val in enumerate(new_sizes):
            eoff = vs_s + 20 + i * 4
            if eoff + 4 <= len(buf):
                struct.pack_into('>I', buf, eoff, sz_val)
    elif len(set(new_sizes)) == 1 and new_sizes[0] == old_fixed:
        pass                                   # fixed-size unchanged, nothing to do
    else:
        return False                           # can't patch fixed-size stsz in-place

    # ── patch video stco entries in-place ─────────────────────────────────────
    is64_v = (bytes(buf[vc_s + 4: vc_s + 8]) == b'co64')
    entry_sz_v, fmt_v = (8, '>Q') if is64_v else (4, '>I')
    for i, off_val in enumerate(new_v_stco):
        eoff = vc_s + 16 + i * entry_sz_v     # 8 hdr + 4 ver/flags + 4 count = 16
        if eoff + entry_sz_v <= len(buf):
            struct.pack_into(fmt_v, buf, eoff, off_val)

    # ── patch audio stco entries in-place ─────────────────────────────────────
    if new_a_stco and ac_s >= 0:
        is64_a = (bytes(buf[ac_s + 4: ac_s + 8]) == b'co64')
        entry_sz_a, fmt_a = (8, '>Q') if is64_a else (4, '>I')
        for i, off_val in enumerate(new_a_stco):
            eoff = ac_s + 16 + i * entry_sz_a
            if eoff + entry_sz_a <= len(buf):
                struct.pack_into(fmt_a, buf, eoff, off_val)

    # ── build new mdat and write output ──────────────────────────────────────
    new_mdat_body = b''.join(video_chunks) + b''.join(audio_chunks)
    new_mdat = struct.pack('>I', 8 + len(new_mdat_body)) + b'mdat' + new_mdat_body

    # Replace old mdat (and anything after it) with new mdat
    new_buf = bytes(buf[:mdat_s]) + new_mdat

    try:
        tmp_path = dst + '._dv_tmp'
        with open(tmp_path, 'wb') as f:
            f.write(new_buf)
        import os; os.replace(tmp_path, dst)
        return True
    except OSError:
        try:
            import os; os.unlink(tmp_path)
        except Exception:
            pass
        return False


# ── MP4 container helpers for dvcC injection ─────────────────────────────────

_MP4_CONTAINERS = frozenset({
    b'moov', b'trak', b'mdia', b'minf', b'stbl', b'dinf', b'udta',
    b'ilst', b'moof', b'traf', b'edts',
})
_MP4_SAMPLE_ENTRIES = frozenset({
    b'dvh1', b'dvhe', b'hvc1', b'hev1', b'avc1', b'avc3',
})


def _mp4_extract_box(data: bytes, btype: bytes) -> bytes:
    """Return full box bytes (size+type+payload) for first occurrence of btype."""
    idx = data.find(btype)
    while idx >= 4:
        sz = struct.unpack_from('>I', data, idx - 4)[0]
        if 8 <= sz <= len(data) - (idx - 4):
            return data[idx - 4: idx - 4 + sz]
        idx = data.find(btype, idx + 1)
    return b''


def _mp4_box_off(data: bytes, btype: bytes) -> int:
    """Return byte offset of size field for first occurrence of btype, or -1."""
    idx = data.find(btype)
    while idx >= 4:
        sz = struct.unpack_from('>I', data, idx - 4)[0]
        if sz >= 8:
            return idx - 4
        idx = data.find(btype, idx + 1)
    return -1


def _mp4_walk_parents(data: bytes, ins: int) -> list:
    """Return start offsets of all boxes that contain byte position ins."""
    result = []

    def walk(pos, end):
        while pos + 8 <= min(end, len(data)):
            sz = struct.unpack_from('>I', data, pos)[0]
            bt = data[pos + 4: pos + 8]
            if sz == 1 and pos + 16 <= len(data):
                sz = struct.unpack_from('>Q', data, pos + 8)[0]
            elif sz == 0:
                sz = len(data) - pos
            if sz < 8:
                break
            box_end = pos + sz
            if pos < ins <= box_end:
                result.append(pos)
                if bt in _MP4_CONTAINERS:
                    walk(pos + 8, box_end)
                elif bt == b'stsd':
                    walk(pos + 16, box_end)          # full-box header + entry_count
                elif bt in _MP4_SAMPLE_ENTRIES:
                    walk(pos + 16, box_end)          # box header + reserved(6) + data_ref(2)
                break
            pos = box_end

    walk(0, len(data))
    return result


def _mp4_shift_offsets(buf: bytearray, ins: int, n: int):
    """Increment stco/co64 chunk offsets that are >= ins by n."""
    raw = bytes(buf)
    for tag, esz, fmt in [(b'stco', 4, '>I'), (b'co64', 8, '>Q')]:
        p = raw.find(tag)
        while p >= 4:
            bsz = struct.unpack_from('>I', buf, p - 4)[0]
            if bsz >= 16:
                count = struct.unpack_from('>I', buf, p + 8)[0]
                for i in range(count):
                    op = p + 12 + i * esz
                    if op + esz > len(buf):
                        break
                    v = struct.unpack_from(fmt, buf, op)[0]
                    if v >= ins:
                        struct.pack_into(fmt, buf, op, v + n)
            p = raw.find(tag, p + 1)


def _inject_dvcc(src: str, dst: str) -> bool:
    """
    Copy dvcC (Dolby Vision Configuration) box from src MP4/MOV into dst.
    Call after muxing to restore DV container signaling lost during re-encode.
    Returns True if dvcC is now present in dst.
    """
    try:
        with open(src, 'rb') as f:
            src_bytes = f.read()
    except Exception:
        return False

    # MOV containers use 'dvvC', MP4 use 'dvcC' — same payload, different 4CC
    dvcc = _mp4_extract_box(src_bytes, b'dvcC')
    if not dvcc:
        dvvC = _mp4_extract_box(src_bytes, b'dvvC')
        if not dvvC:
            return False
        # Re-label dvvC → dvcC for MP4 output (payload is identical)
        dvcc = dvvC[:4] + b'dvcC' + dvvC[8:]

    try:
        with open(dst, 'rb') as f:
            buf = bytearray(f.read())
    except Exception:
        return False

    buf_bytes = bytes(buf)

    # Determine the correct DV config atom name for this container:
    # QuickTime/MOV containers (major_brand b'qt  ') use 'dvvC',
    # ISOBMFF/MP4 containers use 'dvcC'. Payload is identical.
    major_brand = buf_bytes[8:12] if len(buf_bytes) >= 12 else b''
    target_name = b'dvvC' if major_brand == b'qt  ' else b'dvcC'

    # Relabel the extracted atom to the correct name for this output format
    dvcc = dvcc[:4] + target_name + dvcc[8:]

    if target_name in buf_bytes:
        return True                        # already present

    hvcc_off = _mp4_box_off(bytes(buf), b'hvcC')
    if hvcc_off < 0:
        return False
    hvcc_sz = struct.unpack_from('>I', buf, hvcc_off)[0]
    ins = hvcc_off + hvcc_sz               # inject dvcC right after hvcC
    n   = len(dvcc)

    parents = _mp4_walk_parents(bytes(buf), ins)
    pre_insert = bytes(buf)    # Snapshot BEFORE insertion — prevents double-counting:
                               # _mp4_walk_parents returns boxes outermost-first, so
                               # without a snapshot, outer boxes would read already-
                               # incremented inner sizes and add n twice.

    buf[ins:ins] = dvcc

    for off in parents:
        sz = struct.unpack_from('>I', pre_insert, off)[0]   # always from pre-insert
        struct.pack_into('>I', buf, off, sz + n)

    # Update chunk offsets only when moov precedes mdat (faststart layout)
    moov_off = _mp4_box_off(bytes(buf), b'moov')
    mdat_off = _mp4_box_off(bytes(buf), b'mdat')
    if moov_off >= 0 and mdat_off >= 0 and moov_off < mdat_off:
        _mp4_shift_offsets(buf, ins, n)

    tmp = dst + '._tmp'
    try:
        with open(tmp, 'wb') as f:
            f.write(buf)
        os.replace(tmp, dst)
        return True
    except Exception:
        try:
            os.unlink(tmp)
        except Exception:
            pass
        return False

def _unique_output_path(stem: str, ext: str, base_dir: Path = None) -> Path:
    base_dir = base_dir or OUTPUT_DIR
    with _output_lock:
        dst = base_dir / f"{stem}{ext}"
        counter = 2
        while True:
            try:
                dst.touch(exist_ok=False)
                return dst
            except FileExistsError:
                dst = base_dir / f"{stem}_{counter}{ext}"
                counter += 1

export async function probeBatchFile(index) {
  const fObj = S.compress.files[index];
  fObj.status = 'probing';
  renderBatchList();

  const loadEl = $('vinfo-loading-compress');
  loadEl.classList.remove('hidden');
  loadEl.textContent = `Uploading file ${index+1}/${S.compress.files.length} (0%)...`;

  return new Promise((resolve) => {
    const fd = new FormData();
    fd.append('file', fObj.file);
    const xhr = new XMLHttpRequest();
    xhr.open('POST', '/api/probe', true);
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) loadEl.textContent = `Uploading file ${index+1} (${pct}%)...`;
        else loadEl.textContent = `Reading metadata for file ${index+1}...`;
      }
    };
    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          if (data.server_file) {
            fObj.server_file = data.server_file;
            fObj.status = 'ready';
            fObj.meta = data;
          } else {
            fObj.status = 'error';
            fObj.errMsg = data.error || 'No server_file';
          }
        } catch(e) {
          fObj.status = 'error';
          fObj.errMsg = 'JSON Parse Error';
        }
      } else {
        fObj.status = 'error';
        try {
          const errData = JSON.parse(xhr.responseText);
          fObj.errMsg = errData.error || `HTTP ${xhr.status}`;
        } catch(e) {
          fObj.errMsg = `HTTP ${xhr.status}: ${xhr.statusText}`;
        }
      }
      loadEl.classList.add('hidden');
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.onerror = () => {
      fObj.status = 'error';
      fObj.errMsg = 'Network disconnected';
      loadEl.classList.add('hidden');
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.send(fd);
  });
}

      </div>

  }).join('');

  const clientBatchId = Date.now().toString(36) + Math.random().toString(36).substr(2, 5);
  for (let i = 0; i < readyFiles.length; i++) {
    const fObj = readyFiles[i];
    const id   = 'prog-item-' + i;

    const fd = new FormData();
    fd.append('client_batch_id', clientBatchId);
    fd.append('server_file', fObj.server_file);

      sv('msg-' + id, 'Connection error');
    }
  }

  if (S.compress.timer) clearInterval(S.compress.timer);
  S.compress.timer = setInterval(pollBatchProgress, 1000);

let _batchPollBusy = false;

export async function pollBatchProgress() {
  if (_batchPollBusy) return;
  _batchPollBusy = true;
  try { await _doPollBatchProgress(); } finally { _batchPollBusy = false; }
}

async function _doPollBatchProgress() {
  const allJobIds    = Object.keys(S.compress.jobs);
  const activeJobIds = allJobIds.filter(jid => !S.compress.jobs[jid].done);
  const doneCount    = allJobIds.length - activeJobIds.length;

  const progCountEl = $('batch-progress-count');
  if (progCountEl && allJobIds.length > 0) {
    progCountEl.textContent = `(${doneCount}/${allJobIds.length} videos)`;
  }

  if (activeJobIds.length === 0) {
    clearInterval(S.compress.timer);
    S.compress.timer = null;
    setBusy('compress', false);
    const timerEl = $('batch-elapsed-timer');
    if (timerEl && S.compress.batchStartTime) {
      const totalSec = Math.round((Date.now() - S.compress.batchStartTime) / 1000);
      const m = Math.floor(totalSec / 60), s = totalSec % 60;
      timerEl.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
    }
    renderBatchResult();
    return;
  }

  const _timerEl = $('batch-elapsed-timer');
  if (_timerEl && S.compress.batchStartTime) {
    const _elapsed = Math.round((Date.now() - S.compress.batchStartTime) / 1000);
    const _m = Math.floor(_elapsed / 60), _s = _elapsed % 60;
    _timerEl.textContent = `${String(_m).padStart(2,'0')}:${String(_s).padStart(2,'0')}`;
  }

  const _abortCtrl = new AbortController();
  const _abortTid  = setTimeout(() => _abortCtrl.abort(), 3000);
  try {
    const res = await fetch('/api/status_batch', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({ job_ids: activeJobIds }),
      signal: _abortCtrl.signal
    });
    clearTimeout(_abortTid);
    if (!res.ok) return;
    const batchData = await res.json();

    activeJobIds.forEach(jid => {
      const data    = batchData[jid];
      if (!data) return;
      const jobMeta = S.compress.jobs[jid];
      const uid     = jobMeta.ui_id;
      const pct     = data.progress || 0;

      const elPct = $('pct-' + uid);
      const elMsg = $('msg-' + uid);
      const elBar = $('bar-' + uid);
      const elEta = $('eta-' + uid);

      if (elPct) elPct.innerText = pct + '%';
      if (elMsg) elMsg.innerText = data.message || 'Processing...';
      if (elBar) elBar.style.width = pct + '%';

      if (pct > 0 && pct < 100) {
        if (!jobMeta.t0) jobMeta.t0 = Date.now() / 1000;
        const ela = (Date.now() / 1000) - jobMeta.t0;
        const eta = Math.round((ela / pct) * (100 - pct));
        if (elEta) elEta.innerText = eta > 0 ? '~' + eta + 's left' : '';
      } else {
        if (elEta) elEta.innerText = '';
      }

      if (data.status === 'done' || data.status === 'error') {
        jobMeta.done   = true;
        jobMeta.result = data;
        const btnCancel = $('btn-cancel-' + uid);
        if (btnCancel) btnCancel.style.display = 'none';
        const progItem = $(uid);
        if (progItem) {
          progItem.style.background   = 'transparent';
          progItem.style.padding      = '12px 0';
          progItem.style.borderBottom = '1px solid rgba(255,255,255,0.05)';
          const idx    = jobMeta.fileIdx !== undefined ? jobMeta.fileIdx
                       : S.compress.files ? S.compress.files.findIndex(f => f.file.name === jobMeta.file) : -1;
          const idxStr = idx >= 0 ? `<span style="margin-right:4px;">${idx + 1}.</span>` : '';
          progItem.innerHTML = _buildJobResultHTML(jid, jobMeta.file, data, idxStr);
        }
      }
    });
  } catch(e) {
    clearTimeout(_abortTid);
    if (e.name !== 'AbortError') console.error('Batch polling error:', e);
  }
}

let _metaCompareAbort = null;

async function showMetaCompare(jobId) {
  const modal = $('meta-compare-modal');
  const body  = $('meta-compare-body');
  if (!modal || !body) return;

  // Cancel any in-flight request (double-click / rapid open guard)
  if (_metaCompareAbort) { _metaCompareAbort.abort(); }
  _metaCompareAbort = new AbortController();
  const _signal = _metaCompareAbort.signal;

  body.innerHTML = '<div style="text-align:center;padding:30px;color:var(--text-3);">Loading...</div>';
  modal.style.display = 'flex';

  try {
    const res = await fetch('/api/status/' + jobId, { signal: _signal });
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const data = await res.json();

    const src = data.src_exif || {};
    const dst = data.dst_exif || {};

    if (!Object.keys(src).length && !Object.keys(dst).length) {
      body.innerHTML = '<div style="text-align:center;padding:30px;color:var(--text-3);">No metadata available.</div>';
      return;

      if (sv !== undefined && dv === undefined) cls = 'missing';

      (cls ? changed : unchanged).push({ key, sv, dv, cls });
    }

    const clrMap  = { missing: 'var(--red-l)', changed: 'var(--yellow)', added: 'var(--green-l)', '': 'var(--text-2)' };
    const iconMap = { missing: '🔴', changed: '🟡', added: '🟢', '': '✓' };

    const rowHtml = (row) => {
      const clr    = clrMap[row.cls];
      const dstClr = row.cls === 'missing' ? 'var(--red-l)' : row.cls === 'changed' ? 'var(--yellow)' : 'var(--text-2)';
      const srcTxt = row.sv !== undefined ? row.sv : '<em style="color:var(--text-3)">—</em>';
      const dstTxt = row.dv !== undefined ? row.dv : '<em style="color:var(--text-3)">—</em>';

    if (e.name === 'AbortError') return;   // user opened another job, silently discard

# libx265 encoding 4K 10-bit needs ~3 GB RAM with default settings.
# On low-RAM systems, apply thread/lookahead limits to avoid OOM.
_X265_LOW_RAM_MB = 3000

def _x265_mem_params():
    """Return memory-limiting x265-params when free RAM < 3 GB, else []."""
    if free_ram_mb() < _X265_LOW_RAM_MB:
        return ["-x265-params", "pools=1:frame-threads=1:lookahead-threads=1:rc-lookahead=5:bframes=0"]
    return []


def _job_msg(job_id, msg, pct):
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["message"] = msg
            jobs[job_id]["progress"] = pct


def _compress_dovi_pipeline(job_id, src, dst, enc, real_enc, enc_extra, crf,
                              mute, best_audio_idx, target_kbps, total_s, ppath, fps_str):
    """
    Dolby Vision preserving pipeline — 7 steps.

    Key insight: B-frame PTS/DTS live in the MOV container's CTTS table, not in
    the raw HEVC bitstream. By encoding to a full temp.mov first (so CTTS is
    written correctly), then binary-patching its video samples with RPU-injected
    data while keeping the CTTS/stts/stss tables intact, we preserve smooth
    B-frame playback on iOS AND keep Dolby Vision — without the -bf 0 penalty.
    """
    tmp = Path(TEMP_DIR) / f"dovi_{job_id}"
    tmp.mkdir(parents=True, exist_ok=True)
    try:
        raw_hevc  = tmp / "raw.hevc"     # copy of source video (for RPU extraction)
        rpu_bin   = tmp / "rpu.bin"      # extracted DV RPU metadata
        temp_mov  = tmp / "encoded.mov" # full encoded output (video+audio, correct timestamps)
        enc_hevc  = tmp / "enc.hevc"     # Annex-B HEVC extracted from temp_mov
        dv_hevc   = tmp / "dv.hevc"      # enc.hevc with RPU injected

        # Step 1: Extract raw HEVC from source (Annex-B, for RPU extraction)
        _job_msg(job_id, "Extracting Dolby Vision stream...", 2)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(src),
                            "-c:v", "copy", "-an", "-f", "hevc", str(raw_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        # Step 2: Extract DV RPU metadata
        _job_msg(job_id, "Extracting DV RPU metadata...", 5)
        r = subprocess.run([DOVI_TOOL_BIN, "extract-rpu",
                            "-i", str(raw_hevc), "-o", str(rpu_bin)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"RPU extract failed: {r.stderr.decode(errors='replace')[-1000:]}")

        # Step 3: Encode to temp.mov — full output with B-frames, audio, metadata.
        # B-frames are ALLOWED here: the container (temp.mov) correctly stores their
        # PTS/DTS in the CTTS box. We'll preserve this CTTS in the final output.
        _job_msg(job_id, f"Encoding DV base layer ({real_enc})...", 8)
        x265_mem = _x265_mem_params() if real_enc == "libx265" else []
        audio_map = [] if mute else ["-map", f"0:a:{best_audio_idx}"]
        audio_enc = [] if mute else ["-c:a", "aac", "-b:a", f"{target_kbps}k",
                                     "-ar", "48000", "-ac", "2"]
        encode_cmd = ([FFMPEG_BIN] + _hwaccel_args(real_enc) +
                      ["-i", str(src), "-map", "0:v:0"] + audio_map +
                      ["-c:v", real_enc] +
                      _quality_args(enc, crf) + enc_extra + x265_mem +
                      ["-tag:v", "hvc1"] + audio_enc +
                      ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"] +
                      ["-movflags", "use_metadata_tags+faststart"] +
                      ["-y", "-progress", ppath, "-stats_period", "0.5", "-nostats",
                       str(temp_mov)])
        rc, is_oom = _run_progress(encode_cmd, job_id, total_s, ppath)
        if rc != 0:
            return rc, is_oom

        # Step 4: Extract Annex-B HEVC from temp_mov (loses container timestamps,
        # but that's OK — we will restore them via binary patching in Step 7).
        _job_msg(job_id, "Extracting encoded HEVC stream...", 97)
        r = subprocess.run([FFMPEG_BIN, "-y", "-i", str(temp_mov),
                            "-c:v", "copy", "-bsf:v", "hevc_mp4toannexb",
                            "-an", "-f", "hevc", str(enc_hevc)],
                           capture_output=True, timeout=120)
        if r.returncode != 0:
            raise RuntimeError(f"HEVC extract from temp failed: {r.stderr.decode(errors='replace')[-1000:]}")

        # Step 5: Inject RPU NAL units into the encoded HEVC bitstream
        _job_msg(job_id, "Injecting Dolby Vision metadata...", 98)
        r = subprocess.run([DOVI_TOOL_BIN, "inject-rpu",
                            "-i", str(enc_hevc), "--rpu-in", str(rpu_bin),
                            "-o", str(dv_hevc)],
                           capture_output=True, timeout=300)
        if r.returncode != 0:
            raise RuntimeError(f"RPU inject failed: {r.stderr.decode(errors='replace')[-1000:]}")
        if not dv_hevc.exists() or dv_hevc.stat().st_size < 1024:
            raise RuntimeError(f"RPU inject produced empty output")

        # Step 6: Parse dv.hevc (Annex-B) into per-frame MP4-format samples
        _job_msg(job_id, "Parsing DV bitstream...", 98)
        dv_samples = _annexb_to_mp4_samples(str(dv_hevc))
        if not dv_samples:
            raise RuntimeError("Failed to parse DV HEVC samples from dv.hevc")

        # Step 7: Binary-patch temp_mov: swap video samples → dv_samples while
        # keeping CTTS/stts/stss intact. Write result to dst.
        _job_msg(job_id, "Muxing Dolby Vision output (timestamp-preserving)...", 99)
        if not _dovi_mux_with_timestamps(str(temp_mov), dv_samples, str(dst)):
            raise RuntimeError("_dovi_mux_with_timestamps failed — falling back not available")

        # Step 8: Copy dvvC/dvcC atom from source → output
        _job_msg(job_id, "Restoring Dolby Vision container signaling...", 99)
        _inject_dvcc(str(src), str(dst))

        return 0, False

    except RuntimeError as e:
        with _jobs_lock:
            jobs[job_id].update({"status": "error", "message": str(e)})
        return -1, False
    finally:
        shutil.rmtree(tmp, ignore_errors=True)


@job_guard
def _compress_job(job_id, src: Path, enc: str, crf: int,
                  out_ext: str, mute: bool = False):
    with _jobs_lock:
        if job_id not in jobs:
            return
        job = jobs[job_id]
    family = _get_codec_family(enc)
    t0     = None

    src_name = job.get("src_name", "video")
    src_stem = re.sub(r"[^\w\-. ]", "_", Path(src_name).stem)

    av1_warning = ""
    if family == "av1" and out_ext == ".mov":
        out_ext = ".mp4"
        av1_warning = " (Tự đổi sang MP4)"

    supported = CONTAINER_CODEC_SUPPORT.get(out_ext, ["h264"])
    if family not in supported:
        if out_ext == ".webm":
            enc = "libaom-av1"
            job["message"] = "⚠️ .WebM only supports AV1 → auto-switching to libaom-AV1"
        else:
            enc = "libx264"
            job["message"] = f"⚠️ {out_ext.upper()} does not support {family.upper()} → switching to H.264"
        family = _get_codec_family(enc)

    dst = _unique_output_path(src_stem, out_ext)
    if not _check_path_len_job(dst, job_id):
        try: dst.unlink()
        except: pass
        try: src.unlink()
        except: pass
        return

    real_enc, enc_extra = _get_real_enc_extra(enc)

    pfd, ppath = tempfile.mkstemp(suffix=".txt", prefix="vs_")
    os.close(pfd)

    probe_data     = _probe(str(src))
    total_s        = float(probe_data.get("format", {}).get("duration") or 0)

    fps_str = "30000/1001"  # default 29.97 fps (iPhone)
    for s in probe_data.get("streams", []):
        if s.get("codec_type") == "video":
            avg = s.get("avg_frame_rate", "0/0")
            if avg and avg != "0/0":
                fps_str = avg
            else:
                fps_str = s.get("r_frame_rate", "30000/1001")
            break

    # Dolby Vision detection
    is_dv, dv_profile = _get_dv_info(probe_data)
    use_dovi = is_dv and _DOVI_AVAILABLE

    # Store source metadata snapshot
    src_exif = _get_exif_snapshot(str(src))
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["src_exif"] = src_exif

    best_audio_idx = 0
    best_audio_br  = 0
    audio_streams_raw = [
        s for s in probe_data.get("streams", [])
        if s.get("codec_type") == "audio" and s.get("codec_name") and s.get("codec_name") != "none"
    ]
    audio_streams = list(enumerate(audio_streams_raw))
    for stream_idx, s in audio_streams:
        # Prefer decodable AAC (mp4a) over Apple Lossless/spatial (apac).
        # apac is a 3-4ch spatial stream: re-encoding it to stereo AAC without
        # proper downmix causes iOS playback stutter (timing desync with mebx tracks).
        codec_tag = s.get("codec_tag_string", "").lower()
        if codec_tag == "apac":
            # Only use apac if no other audio stream exists
            if best_audio_br > 0:
                continue
            br = 1   # placeholder — will be overridden by any real decodable stream
        else:
            br = 0
            raw_br = s.get("bit_rate")
            if raw_br:
                try: br = int(raw_br)
                except: pass
            if br < 10000:
                tags = s.get("tags", {})
                bps  = tags.get("BPS") or tags.get("BPS-eng")
                if bps:
                    try: br = int(bps)
                    except: pass
            if br < 10000:
                ch = s.get("channels", 0)
                sr = s.get("sample_rate", 0)
                if ch and sr:
                    br = ch * int(sr) * 16
        if br > best_audio_br:
            best_audio_br  = br
            best_audio_idx = stream_idx

    cmd = [FFMPEG_BIN] + _hwaccel_args(real_enc) + ["-i", str(src), "-map", "0:v:0"]

    if not mute and audio_streams:
        audio_map = (f"0:a:{best_audio_idx}"
                     if len(audio_streams) > 1 else "0:a:0")
        cmd += ["-map", audio_map]

    cmd += ["-c:v", real_enc]
    cmd += _quality_args(enc, crf)
    cmd += enc_extra

    if family == "h265" and out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-tag:v", "hvc1"]

    target_kbps = 192
    if not mute and audio_streams:
        target_kbps = max(128, min(320, int(best_audio_br / 1000))) if best_audio_br > 0 else 192
        audio_br_str = f"{target_kbps}k"
        cmd += [
            "-c:a", "aac",
            "-b:a", audio_br_str,
            "-ar",  "48000",
            "-ac",  "2",
        ]

    cmd += ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"]
    if not mute and audio_streams:
        cmd += ["-map_metadata:s:a", "0:s:a"]

    if out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-movflags", "use_metadata_tags+faststart"]
    elif out_ext != ".mkv":
        cmd += ["-movflags", "+faststart"]

    cmd += ["-y", "-progress", ppath,
            "-stats_period", "0.5", "-nostats", str(dst)]

    oom_retries = 0
    while True:
        with _jobs_lock:
            job["status"]  = "waiting"
            job["message"] = "Waiting in queue..."
        with _pick_sem(real_enc):
            if job.get("cancelled"): return
            with _jobs_lock:
                job["status"]  = "processing"
                job["message"] = f"Encoding ({real_enc}){av1_warning}{' — no audio' if mute else ''}..."
            t0 = time.time()
            try:
                if use_dovi:
                    rc, is_oom = _compress_dovi_pipeline(
                        job_id, src, dst, enc, real_enc, enc_extra, crf,
                        mute, best_audio_idx, target_kbps, total_s, ppath, fps_str)
                else:
                    rc, is_oom = _run_progress(cmd, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            if job.get("cancelled"): return
            oom_retries += 1
            if oom_retries >= 3:
                if use_dovi:
                    # DV pipeline OOM → fall back to regular encode (no DV)
                    use_dovi = False
                    oom_retries = 0
                    with _jobs_lock:
                        job["status"]  = "waiting"
                        job["message"] = "⚠️ DV preservation failed (low RAM) — encoding as HLG..."
                    try: dst.unlink()
                    except: pass
                    time.sleep(2)
                else:
                    with _jobs_lock:
                        job.update({"status": "error", "message": "Insufficient VRAM — failed after 3 retries."})
                    try: dst.unlink()
                    except: pass
                    return
            else:
                with _jobs_lock:
                    job["status"]  = "waiting"
                    job["message"] = f"GPU full, retrying ({oom_retries}/3)..."
                time.sleep(3)

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        _sync_meta(str(src), str(dst))

        dst_exif = _get_exif_snapshot(str(dst))
        with _jobs_lock:
            if job_id in jobs:
                jobs[job_id]["dst_exif"] = dst_exif

        ela      = time.time() - t0 if t0 else 0
        enc_time = _fmt_time(ela)

        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = out_duration = "N/A"

        try:
            d = float(out_data.get("format", {}).get("duration", 0))
            if d > 0:
                out_duration = f"{int(d//60)}:{int(d%60):02d}"
        except: pass

        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                w, h = s.get('width', '?'), s.get('height', '?')
                rotation = 0
                if "rotate" in s.get("tags", {}):
                    try: rotation = abs(int(s["tags"]["rotate"]))
                    except: pass
                else:
                    for sd in s.get("side_data_list", []):
                        if sd.get("side_data_type") == "Display Matrix":
                            try: rotation = abs(int(sd.get("rotation", 0)))
                            except: pass
                if rotation in (90, 270) and w != '?' and h != '?':
                    out_res = f"{h}x{w}"
                else:
                    out_res = f"{w}x{h}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break

        if mute:
            out_audio = "Muted"
        else:
            for s in out_data.get("streams", []):
                if s.get("codec_type") == "audio":
                    ac     = s.get("codec_name", "N/A").upper()
                    br     = s.get("bit_rate")
                    br_int = int(br) if br else 0
                    if br_int < 10000:
                        tags   = s.get("tags", {})
                        br_tag = tags.get("BPS") or tags.get("BPS-eng")
                        if br_tag:
                            try: br_int = int(br_tag)
                            except: pass
                    if br_int < 10000:
                        br_int = target_kbps * 1000
                    brs       = f" {br_int//1000}kbps" if br_int > 10000 else ""
                    out_audio = f"{ac}{brs}"
                    break

        result = {
            "status":        "done",
            "progress":      100,
            "message":       "Completed!",
            "output":        str(dst),
            "sz_before":     _fmt_size(str(src)),
            "sz_after":      _fmt_size(str(dst)),
            "encode_time":   enc_time,
            "out_res":       out_res,
            "out_fps":       out_fps,
            "out_audio":     out_audio,
            "out_codec":     real_enc,
            "out_duration":  out_duration,
            "src_is_dv":     is_dv,
            "src_dv_profile":dv_profile,
            "dst_is_dv":     _detect_dolby_vision(out_data),
        }
        if family == "av1" and av1_warning:
            result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 does not support MOV format — auto-converted to MP4"
        with _jobs_lock:
            job.update(result)
            log_name = job.get("log_file")
        if log_name:
            try: (Path(OUTPUT_DIR) / log_name).unlink(missing_ok=True)
            except: pass
    else:
        with _jobs_lock:
            if not job.get("cancelled") and job.get("status") != "error":
                job.update({"status": "error", "message": "Compression failed!"})
        try: dst.unlink()
        except: pass

    try: src.unlink()
    except: pass
