class MEMORYSTATUSEX(ctypes.Structure):
    _fields_ = [
        ("dwLength", ctypes.c_ulong),
        ("dwMemoryLoad", ctypes.c_ulong),
        ("ullTotalPhys", ctypes.c_ulonglong),
        ("ullAvailPhys", ctypes.c_ulonglong),

def _free_sys_ram_mb():
    try:
        stat = MEMORYSTATUSEX()
        stat.dwLength = ctypes.sizeof(MEMORYSTATUSEX)
        ctypes.windll.kernel32.GlobalMemoryStatusEx(ctypes.byref(stat))
        return stat.ullAvailPhys / (1024 * 1024)
    except:
        return 4000

class _AdaptiveGPUSemaphore:
    """
    Giới hạn song song GPU dựa trên:
    1. Số slot cứng (mặc định 4 concurrent job)
    2. VRAM tự do còn lại (> 10% tổng VRAM)
    3. RAM hệ thống (> 2GB dự trữ)

    Lưu ý: KHÔNG check % GPU Encoder Util vì một video 4K NVENC
    đã chạy là luôn có Video Encode = 100%, kiểm tra đó sẽ block
    mãi mãi và không bao giờ cho slot thứ 2.
    """
    def __init__(self, max_slots=4, keep_vram_pct=0.10):
        self._max          = max_slots
        self._keep_vram_pct = keep_vram_pct
        self._lock         = threading.Lock()
        self._count        = 0
        self._last_acquire = 0.0

    def _vram_stats(self):
        """Trả về (vram_free_mb, vram_total_mb) hoặc (None, None)."""
        try:
            r = subprocess.run(
                ["nvidia-smi",
                 "--query-gpu=memory.free,memory.total",
                 "--format=csv,noheader,nounits"],
                capture_output=True, text=True, timeout=2
            )
            if r.returncode == 0:
                free_l, total_l = [], []
                for line in r.stdout.strip().splitlines():
                    parts = line.split(',')
                    if len(parts) >= 2 and parts[0].strip().isdigit() and parts[1].strip().isdigit():
                        free_l.append(int(parts[0].strip()))
                        total_l.append(int(parts[1].strip()))
                if free_l and total_l:
                    return min(free_l), min(total_l)
        except Exception:
            pass
        return None, None

    def acquire(self):
        first = True
        while True:
            if not first:
                time.sleep(1.5)
            first = False
            # Anti-burst: wait OUTSIDE lock so other threads aren't blocked
            with self._lock:
                elapsed = time.time() - self._last_acquire
            if elapsed < 0.5:
                time.sleep(0.5 - elapsed)

            # Query VRAM OUTSIDE lock (~100ms subprocess, up to 2s timeout)
            vram_free, vram_total = self._vram_stats()

            with self._lock:
                req_vram  = vram_total * self._keep_vram_pct if vram_total else 800
                has_vram  = vram_free is None or vram_free > req_vram
                has_sys_ram = _free_sys_ram_mb() > 2000

                if self._count < self._max and has_vram and has_sys_ram:
                    self._count += 1
                    self._last_acquire = time.time()
                    return

    def release(self):
        with self._lock:
            self._count = max(0, self._count - 1)

    def __enter__(self):    self.acquire()
    def __exit__(self, *_): self.release()

def _auto_gpu_slots(vram_per_job_mb: int = 1800, min_slots: int = 1, max_slots: int = 8) -> int:
    """
    Tự động tính số GPU job song song dựa trên tổng VRAM của card.
    Công thức: floor(total_vram * 85% / vram_per_job)
    Mặc định mỗi job 4K NVENC + hwaccel decode cần ~1800MB VRAM buffer.
    Fallback về 2 slot nếu không đọc được VRAM (không có nvidia-smi).
    """
    try:
        r = subprocess.run(
            ["nvidia-smi", "--query-gpu=memory.total", "--format=csv,noheader,nounits"],
            capture_output=True, text=True, timeout=2
        )
        if r.returncode == 0:
            totals = [int(x.strip()) for x in r.stdout.strip().splitlines() if x.strip().isdigit()]
            if totals:
                total_mb   = min(totals)           # card yếu nhất nếu multi-GPU
                usable_mb  = int(total_mb * 0.85)  # dự trữ 15% cho driver/OS
                slots      = max(min_slots, min(max_slots, usable_mb // vram_per_job_mb))
                print(f"  GPU VRAM detected: {total_mb} MB → {slots} concurrent encode slots")
                return slots
    except Exception:
        pass
    print("  GPU VRAM: cannot read, using default 2 slots")
    return 2

def _fmt_size_bytes(b):
    if b < 1024**2:   return f"{b/1024:.1f} KB"

def _parse_dl_size(text: str) -> int:
    """Parse kích thước file từ dòng progress yt-dlp → bytes, hoặc 0 nếu không đọc được.
    Ví dụ: '[download]  2.7% of ~  1.44GiB at ...' → 1547600896
    """
    m = re.search(r'of\s+~?\s*([\d.]+)\s*(GiB|MiB|KiB|B)\b', text)
    if not m: return 0
    mult = {"GiB": 1 << 30, "MiB": 1 << 20, "KiB": 1 << 10, "B": 1}
    return int(float(m.group(1)) * mult.get(m.group(2), 0))

def _probe(path):
    if not FFPROBE_BIN: return {}
    try:
        r = subprocess.run(
            [FFPROBE_BIN,"-v","quiet","-print_format","json",
             "-show_streams","-show_format", path],
            capture_output=True, text=True, timeout=30)
    except subprocess.TimeoutExpired:
        return {}
    try: return json.loads(r.stdout)
    except: return {}

def _extract_ui_meta(tmp_path):
    data = _probe(str(tmp_path))
    info = {"duration": "N/A", "codec": "N/A", "res": "N/A", "fps": "N/A", "audio": "N/A"}
    if not data: return info

    try:
        d = float(data["format"]["duration"])
        info["duration"] = f"{int(d//60)}:{int(d%60):02d}"
    except: pass

    for s in data.get("streams", []):
        if s.get("codec_type") == "video":
            raw = s.get("codec_name", "").lower()
            info["codec"] = CODEC_LABELS.get(raw, raw.upper())
            w, h = s.get('width', '?'), s.get('height', '?')
            rotation = 0
            if "rotate" in s.get("tags", {}):
                try: rotation = abs(int(s["tags"]["rotate"]))
                except: pass
            else:
                for sd in s.get("side_data_list", []):
                    if sd.get("side_data_type") == "Display Matrix":
                        try: rotation = abs(int(sd.get("rotation", 0)))
                        except: pass
            if rotation in (90, 270) and w != '?' and h != '?': info["res"] = f"{h}x{w}"
            else: info["res"] = f"{w}x{h}"
            r = s.get("r_frame_rate", "0/1")
            if "/" in r:
                n, d2 = r.split("/")
                if int(d2): info["fps"] = f"{round(int(n)/int(d2),1)}"
            break

    for s in data.get("streams", []):
        if s.get("codec_type") == "audio":
            raw = s.get("codec_name", "").lower()
            ac = AUDIO_LABELS.get(raw, raw.upper())
            br = s.get("bit_rate")
            br_int = int(br) if br else 0
            if br_int < 10000:
                br_tag = s.get("tags", {}).get("BPS") or s.get("tags", {}).get("BPS-eng")
                if br_tag:
                    try: br_int = int(br_tag)
                    except: pass
            brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
            info["audio"] = f"{ac}{brs}"
            break

    return info

def _read_progress(fpath):
    try:
        txt  = Path(fpath).read_text(encoding="utf-8", errors="ignore")
        done = "progress=end" in txt
        t    = 0.0
        for ln in reversed(txt.splitlines()):
            if ln.startswith("out_time_ms="):
                try:
                    ms = int(ln.split("=")[1])
                    if ms > 0: t = ms/1_000_000
                except: pass
                break
        return t, done
    except: return 0.0, False

def _test_enc(codec):
    if not FFMPEG_BIN: return False
    r = subprocess.run(
        [FFMPEG_BIN,
         "-f", "lavfi", "-i", "nullsrc=s=256x256",
         "-t", "0.1",

def _detect_gpu() -> dict:
    candidates = [
        ("av1_nvenc",  "NVIDIA", "AV1"),
        ("av1_amf",    "AMD",    "AV1"),
        ("av1_qsv",    "Intel",  "AV1"),
        ("hevc_nvenc", "NVIDIA", "H.265"),
        ("hevc_amf",   "AMD",    "H.265"),
        ("hevc_qsv",   "Intel",  "H.265"),
        ("h264_nvenc", "NVIDIA", "H.264"),
        ("h264_amf",   "AMD",    "H.264"),
        ("h264_qsv",   "Intel",  "H.264"),
    ]
    found, seen = {}, set()
    for codec, vendor, fmt in candidates:
        key = (vendor, fmt)
        if key not in seen and _test_enc(codec):
            seen.add(key)
            found[codec] = {"vendor": vendor, "format": fmt}
    return found

    return GLOBAL_GPU_INFO, GLOBAL_DEFAULT_ENC, GLOBAL_ENC_GROUPS

def _get_default_encoder(gpu: dict) -> str:
    """
    Thứ tự ưu tiên:
    1. NVIDIA/AMD AV1
    2. NVIDIA/AMD H.265  ← GPU trước CPU
    3. CPU H.265 fallback
    QSV không bao giờ là default
    """
    for enc in ["av1_nvenc", "av1_amf"]:
        if enc in gpu: return enc
    for enc in ["hevc_nvenc", "hevc_amf"]:
        if enc in gpu: return enc
    return "libx265"


def _build_encoder_options(gpu: dict) -> list:
    groups = []

    # ── GPU AV1 (NVIDIA/AMD only) ─────────────────────────
    for codec, info in gpu.items():

                "warning": False,

            })

    # ── GPU H.265 (NVIDIA/AMD only) ───────────────────────
    for codec, info in gpu.items():
        if info["format"] == "H.265" and "qsv" not in codec:
            groups.append({
                "group":   f"<i data-lucide=\"rocket\" style=\"width:16px;height:16px;vertical-align:-2px\"></i> {info['vendor']} GPU — H.265",
                "warning": False,
                "encoders": [{"codec": codec,
                    "label": f"{info['vendor']} H.265 — Fast, low CPU usage <i data-lucide=\"star\" style=\"width:14px;height:14px;vertical-align:-2px;color:var(--yellow)\"></i>"}]
            })

    # ── CPU H.265 / H.264 ─────────────────────────────────
    groups.append({
        "group":   "<i data-lucide=\"cpu\" style=\"width:16px;height:16px;vertical-align:-2px\"></i> CPU",
        "warning": False,
        "encoders": [
            {"codec": "libx265",
             "label": "H.265 / HEVC — Recommended <i data-lucide=\"check-circle-2\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--green)\"></i>"},
            {"codec": "libx264",
             "label": "H.264 — Highest compatibility"},
        ]
    })

    # ── CPU AV1 ───────────────────────────────────────────
    av1_enc = []
    if _test_enc("libsvtav1"):
        av1_enc += [
            {"codec": "libsvtav1",
             "label": "SVT-AV1 preset 6 — ~25% smaller than H.265"},
            {"codec": "libsvtav1_p4",
             "label": "SVT-AV1 preset 4 — Higher quality, slower"},

    groups.append({
        "group":   "<i data-lucide=\"cpu\" style=\"width:16px;height:16px;vertical-align:-2px\"></i> CPU — AV1 (smaller than H.265, slower)",
        "warning": False,
        "encoders": av1_enc
    })

    # ── GPU H.264 (NVIDIA/AMD only) ───────────────────────
    for codec, info in gpu.items():
        if info["format"] == "H.264" and "qsv" not in codec:
            groups.append({

                "warning": False,
                "encoders": [{"codec": codec,
                    "label": f"{info['vendor']} H.264"}]
            })

    # ── Intel QSV — CHỈ hiện khi KHÔNG có GPU rời ────────
    has_discrete_gpu = any("qsv" not in c for c in gpu)
    if not has_discrete_gpu:

             "label": f"Intel QSV {i['format']} — Faster than CPU"}

        if qsv_encs:
            groups.append({
                "group":   "<i data-lucide=\"zap\" style=\"width:16px;height:16px;vertical-align:-2px\"></i> Intel QSV — Faster than CPU, lower quality",
                "warning": True,
                "encoders": qsv_encs
            })

    return groups

def _get_real_enc_extra(enc: str) -> tuple:
    # SVT-AV1: dùng toàn bộ CPU cores vì luôn chỉ có 1 slot chạy
    if enc == "libsvtav1_p4": return "libsvtav1", ["-preset","4","-svtav1-params",f"lp={_cpu_cores}"]
    if enc == "libsvtav1":    return "libsvtav1", ["-preset","6","-svtav1-params",f"lp={_cpu_cores}"]
    if enc in ("libx265","libx264"):
        # Không set -threads — để encoder tự detect và tối ưu cho cả single lẫn multi-job
        return enc, ["-preset","medium"]
    if enc == "libaom-av1":
        # tiles 4x4 = 16 tiles — parallelism tốt hơn cho cả 1080p và 4K
        return enc, ["-cpu-used","6","-row-mt","1","-tiles","4x4"]
    return enc, []

def _quality_args(enc: str, crf: int) -> list:

def _sync_meta(src: str, dst: str) -> bool:
    """
    Bước 1: Apple QuickTime Keys (GPS, Camera, Device) — quan trọng nhất cho iPhone
    Bước 2: Standard metadata all:all
    """
    if not EXIFTOOL_BIN: return False

    # Bước 1: Copy Apple QuickTime Keys atoms
    cmd1 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        # Apple QuickTime Keys — GPS
        "-Keys:GPSCoordinates",
        "-Keys:LocationBody",
        "-Keys:LocationDate",
        "-Keys:LocationName",
        # Apple QuickTime Keys — Device
        "-Keys:Make",
        "-Keys:Model",
        "-Keys:Software",
        # Apple QuickTime Keys — Camera
        "-Keys:CameraLensModel",
        "-Keys:CreationDate",
        # Live Photo
        "-Keys:ContentIdentifier",

        # Thêm UserData fallback (iOS 17/18 Apple Photos fallback)
        "-UserData:Make<Keys:Make",
        "-UserData:Model<Keys:Model",
        "-UserData:GPSCoordinates<Keys:GPSCoordinates",
        "-UserData:CreationDate<Keys:CreationDate",
        "-overwrite_original",
        dst
    ]
    r1 = subprocess.run(cmd1, capture_output=True, text=True)

    # Bước 2: Copy standard metadata (không ghi đè Keys đã copy ở bước 1)
    cmd2 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-all:all",
        "--FileModifyDate",
        "--FileCreateDate",
        "--Keys:all",   # Không ghi đè Keys từ bước 1
        "-overwrite_original",
        dst
    ]
    r2 = subprocess.run(cmd2, capture_output=True, text=True, encoding="utf-8", errors="replace")

    return r1.returncode == 0 or r2.returncode == 0

def _run_progress(cmd, job_id, total_s, ppath, timeout_sec=7200):
    job = jobs.get(job_id, {})
    stem = "unknown"
    if "src_name" in job:
        stem = Path(job["src_name"]).stem
    elif "url" in job:
        stem = "download_" + re.sub(r"[^\w]", "", job["url"].split("/")[-1][:15])
    stem = re.sub(r"[^\w\-.]", "_", stem)
    ts = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    log_name = f"ffmpeg_err_{stem}_{ts}.log"
    log_path = Path(OUTPUT_DIR) / log_name
    with _jobs_lock:
        if job_id in jobs:
            jobs[job_id]["log_file"] = log_name
    deadline = time.time() + timeout_sec
    with open(log_path, "w", encoding="utf-8") as f:
        f.write("CMD: " + " ".join(cmd) + "\n\n")
        proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=f)
        procs[job_id] = proc
        try:
            while True:
                time.sleep(0.5)
                # Cancellation check — kill immediately and break
                if jobs.get(job_id, {}).get("cancelled"):
                    try:
                        if os.name == 'nt':
                            subprocess.run(["taskkill", "/F", "/T", "/PID", str(proc.pid)],
                                           stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, timeout=5)
                        else:
                            proc.terminate()
                            try: proc.wait(timeout=5)
                            except subprocess.TimeoutExpired: proc.kill()
                    except: pass
                    break
                # Timeout guard
                if time.time() > deadline:
                    try: proc.kill()
                    except: pass
                    with _jobs_lock:
                        jobs[job_id].update({"status": "error", "message": f"FFmpeg timeout after {timeout_sec//3600}h"})
                    break
                cur, done = _read_progress(ppath)
                if total_s > 0 and cur > 0:
                    with _jobs_lock:
                        jobs[job_id]["progress"] = min(int(cur/total_s*100), 99)
                if done or proc.poll() is not None:
                    break
        finally:
            proc.wait()
            procs.pop(job_id, None)
    try: os.remove(ppath)
    except: pass

    is_oom = False
    if proc.returncode != 0:
        try:
            log_content = log_path.read_text(encoding="utf-8", errors="ignore").lower()
            oom_keywords = ["out of memory", "cannot allocate memory", "insufficient video memory", "error allocating"]
            if any(k in log_content for k in oom_keywords):
                is_oom = True
        except:
            pass

    # Xóa log nếu nén thành công để đỡ rác ổ cứng
    if proc.returncode == 0:
        try: os.remove(log_path)
        except: pass

    return proc.returncode, is_oom

def _build_admin_response():
    with _clients_lock:
        now = time.time()
        expired = [cid for cid, info in active_clients.items() if now - info["last_seen"] > 5]
        for cid in expired:
            active_clients.pop(cid, None)
        clients_snapshot = list(active_clients.items())

    active_states = []
    has_loading = False

    for j in list(jobs.values()):
        status = j.get("status")
        if status in ["processing", "waiting"]:
            jtype = j.get("type")
            if jtype == "audio" and "EXTRACTING" not in active_states:
                active_states.append("EXTRACTING")
            elif jtype == "download" and "DOWNLOADING" not in active_states:
                active_states.append("DOWNLOADING")
            elif jtype not in ["audio", "download"] and "COMPRESSING" not in active_states:
                active_states.append("COMPRESSING")
        elif status in ["downloading", "probing", "uploading"]:
            has_loading = True

    if active_states:
        server_state = " / ".join(active_states)
    elif has_loading:
        server_state = "LOADING"
    else:
        server_state = "IDLE"

    return {
        "server_state": server_state,
        "clients": [
            {
                "id": cid,
                "ip": info["ip"],
                "os": info.get("os", "Unknown OS"),
                "browser": info.get("browser", "Unknown Browser"),
                "raw_ua": info.get("raw_ua", ""),
                "status": info["status"],
                "tab": info["tab"],
                "filename": info["filename"],
                "progress": info["progress"]
            }
            for cid, info in clients_snapshot
        ]
    }

        pass

            "created_at": time.time()
        }
    threading.Thread(
        target=_compress_job,
        args=(job_id, src_path, enc, crf, out_ext, mute),  # ← thêm mute

_MAX_PATH = 200  # safe margin under Windows MAX_PATH (260)

def _check_path_len(p: Path, job_id: str) -> bool:
    """Returns False and sets job error if path exceeds Windows MAX_PATH limit."""
    if os.name == "nt" and len(str(p)) > _MAX_PATH:
        with _jobs_lock:
            jobs[job_id].update({
                "status": "error",
                "message": f"⚠️ Output path too long ({len(str(p))} chars). Move the output folder closer to drive root."
            })
        return False
    return True

def _unique_output_path(stem: str, ext: str, base_dir: Path = None) -> Path:
    """Thread-safe output filename reservation via _output_lock + touch()."""
    base_dir = base_dir or OUTPUT_DIR
    with _output_lock:
        dst = base_dir / f"{stem}{ext}"
        counter = 2
        while True:
            try:
                dst.touch(exist_ok=False)   # atomic: raises FileExistsError if taken
                return dst
            except FileExistsError:
                dst = base_dir / f"{stem}_{counter}{ext}"
                counter += 1

def _job_guard(fn):
    """Bắt exception không xử lý trong job thread → job không bị stuck 'processing'."""
    def wrapper(job_id, *args, **kwargs):
        try:
            fn(job_id, *args, **kwargs)
        except Exception as e:
            import traceback
            with _jobs_lock:
                if job_id in jobs:
                    jobs[job_id].update({"status": "error", "message": f"Error: {e}"})
                else:
                    print(f"[job_guard] job {job_id} missing on error: {e}\n{traceback.format_exc()}",
                          file=sys.stderr)
        finally:
            # Signal sync download waiter (if any) when job finishes
            ev = _sync_events.pop(job_id, None)
            if ev:
                ev.set()
    return wrapper

@_job_guard
def _compress_job(job_id, src: Path, enc: str, crf: int,
                  out_ext: str, mute: bool = False):
    with _jobs_lock:
        if job_id not in jobs:
            return
        job = jobs[job_id]
    family = _get_codec_family(enc)
    # t0 sẽ được đặt sau khi được cấp slot GPU — không tính thời gian chờ hàng đợi
    t0     = None

    src_name = job.get("src_name", "video")
    src_stem = re.sub(r"[^\w\-. ]", "_", Path(src_name).stem)

    # ── Kiểm tra AV1 + MOV ───────────────────────────────
    av1_warning = ""
    if family == "av1" and out_ext == ".mov":
        out_ext = ".mp4"
        av1_warning = " (Tự đổi sang MP4)"

    # ── Kiểm tra container vs codec ───────────────────────
    supported = CONTAINER_CODEC_SUPPORT.get(out_ext, ["h264"])
    if family not in supported:
        if out_ext == ".webm":
            enc = "libaom-av1"
            job["message"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> .WebM only supports AV1 → auto-switching to libaom-AV1"
        else:
            enc = "libx264"
            job["message"] = f"<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> {out_ext.upper()} does not support {family.upper()} → switching to H.264"
        family = _get_codec_family(enc)

    # ── Output path (atomic reserve to prevent race on same filename) ─
    dst = _unique_output_path(src_stem, out_ext)
    if not _check_path_len(dst, job_id):
        try: src.unlink()
        except: pass
        return

    real_enc, enc_extra = _get_real_enc_extra(enc)

    pfd, ppath = tempfile.mkstemp(suffix=".txt", prefix="vs_")
    os.close(pfd)

    # ── Probe source một lần — lấy duration + audio streams ──
    probe_data     = _probe(str(src))
    total_s        = float(probe_data.get("format", {}).get("duration") or 0)
    best_audio_idx = 0
    best_audio_br  = 0
    audio_streams_raw = [
        s for s in probe_data.get("streams", [])
        if s.get("codec_type") == "audio" and s.get("codec_name") and s.get("codec_name") != "none"
    ]
    audio_streams = list(enumerate(audio_streams_raw))
    for stream_idx, s in audio_streams:
        br = 0
        raw_br = s.get("bit_rate")
        if raw_br:
            try: br = int(raw_br)
            except: pass
        if br < 10000:
            tags = s.get("tags", {})
            bps  = tags.get("BPS") or tags.get("BPS-eng")
            if bps:
                try: br = int(bps)
                except: pass
        if br < 10000:
            ch = s.get("channels", 0)
            sr = s.get("sample_rate", 0)
            if ch and sr:
                br = ch * int(sr) * 16
        if br > best_audio_br:
            best_audio_br  = br
            best_audio_idx = stream_idx

    # ── Build FFmpeg command ──────────────────────────────
    cmd = [FFMPEG_BIN] + _hwaccel_args(real_enc) + ["-i", str(src), "-map", "0:v:0"]

    # Audio map — chỉ khi không mute
    if not mute:
        audio_map = (f"0:a:{best_audio_idx}"
                     if len(audio_streams) > 1 else "0:a:0")
        cmd += ["-map", audio_map]

    # Data map
    # BỎ QUA -map 0:d? vì FFmpeg không thể copy chuẩn track 'mebx' của Apple
    # Việc copy sai định dạng sẽ khiến Apple Photos (iOS) từ chối đọc metadata!

    # Video encoder
    cmd += ["-c:v", real_enc]
    cmd += _quality_args(enc, crf)
    cmd += enc_extra

    # hvc1 tag cho H.265
    if family == "h265" and out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-tag:v", "hvc1"]

    # Audio encode — chỉ khi không mute
    if not mute:
        target_kbps = max(128, min(320, int(best_audio_br / 1000))) if best_audio_br > 0 else 192
        audio_br_str = f"{target_kbps}k"
        cmd += [
            "-c:a", "aac",   # nhẹ hơn aac_mf, không ngốn CPU
            "-b:a", audio_br_str,
            "-ar",  "48000",
            "-ac",  "2",
        ]

    # Metadata
    cmd += ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"]
    if not mute:
        cmd += ["-map_metadata:s:a", "0:s:a"]

    # Container flags
    if out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-movflags", "use_metadata_tags+faststart"]
    elif out_ext != ".mkv":
        cmd += ["-movflags", "+faststart"]

    cmd += ["-y", "-progress", ppath,
            "-stats_period", "0.5", "-nostats", str(dst)]

    oom_retries = 0
    while True:
        with _jobs_lock:
            job["status"]  = "waiting"
            job["message"] = "Waiting in queue..."
        with _pick_sem(real_enc):
            if job.get("cancelled"): return
            with _jobs_lock:
                job["status"]  = "processing"
                job["message"] = f"Encoding ({real_enc}){av1_warning}{' — no audio' if mute else ''}..."
            t0 = time.time()  # Bắt đầu đo từ đây — sau khi được cấp slot
            try:
                rc, is_oom = _run_progress(cmd, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                with _jobs_lock:
                    job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            if job.get("cancelled"): return
            oom_retries += 1
            if oom_retries >= 3:
                with _jobs_lock:
                    job.update({"status": "error", "message": "Insufficient VRAM — failed after 3 retries."})
                try: dst.unlink()
                except: pass
                return
            with _jobs_lock:
                job["status"]  = "waiting"
                job["message"] = f"GPU full, retrying ({oom_retries}/3)..."
            time.sleep(3)

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        _sync_meta(str(src), str(dst))

        ela      = time.time() - t0 if t0 else 0
        enc_time = _fmt_time(ela)

        # Probe output
        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = out_duration = "N/A"

        try:
            d = float(out_data.get("format", {}).get("duration", 0))
            if d > 0:
                out_duration = f"{int(d//60)}:{int(d%60):02d}"
        except: pass

        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break

        if mute:
            out_audio = "Muted"
        else:
            for s in out_data.get("streams", []):
                if s.get("codec_type") == "audio":
                    ac     = s.get("codec_name", "N/A").upper()
                    br     = s.get("bit_rate")
                    br_int = int(br) if br else 0
                    if br_int < 10000:
                        tags   = s.get("tags", {})
                        br_tag = tags.get("BPS") or tags.get("BPS-eng")
                        if br_tag:
                            try: br_int = int(br_tag)
                            except: pass
                    if br_int < 10000:          # MP4/AAC thường không có stream-level bitrate
                        br_int = target_kbps * 1000
                    brs       = f" {br_int//1000}kbps" if br_int > 10000 else ""
                    out_audio = f"{ac}{brs}"
                    break

        result = {
            "status":      "done",
            "progress":    100,
            "message":     "Completed!",
            "output":      str(dst),
            "sz_before":   _fmt_size(str(src)),
            "sz_after":    _fmt_size(str(dst)),
            "encode_time": enc_time,
            "out_res":     out_res,
            "out_fps":     out_fps,
            "out_audio":   out_audio,
            "out_codec":   real_enc,
            "out_duration":out_duration,
        }
        if family == "av1" and av1_warning:
            result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 does not support MOV format — auto-converted to MP4"
        with _jobs_lock:
            job.update(result)
    else:
        with _jobs_lock:
            if not job.get("cancelled"):
                job.update({"status": "error", "message": "Compression failed!"})
        try: dst.unlink()
        except: pass

    try: src.unlink()
    except: pass

@app.route("/api/social", methods=["POST"])
def api_social():
    ok, msg = _check_disk_space()
    if not ok: return jsonify({"error": msg}), 507
    preset = request.form.get("preset","instagram_story")
    ftype  = request.form.get("type","video")

    server_file = request.form.get("server_file")
    if server_file:
        orig_name = request.form.get("original_name")
        tmp_p = _resolve_upload_file(server_file)
        if tmp_p is None:
            return jsonify({"error":"Invalid file path!"}), 400
        if not tmp_p.exists():
            return jsonify({"error":"Uploaded file was lost! Please reselect."}), 400
        ext = Path(orig_name).suffix.lower()
        if ftype=="video" and ext not in VIDEO_EXTS:
            return jsonify({"error":"This preset is for videos!"}), 400
        if ftype=="photo" and ext not in IMAGE_EXTS:
            return jsonify({"error":"This preset is for photos!"}), 400

        job_id   = uuid.uuid4().hex[:8]
        src_path = tmp_p
    else:
        f = request.files.get("file")
        if not f: return jsonify({"error":"No file uploaded!"}), 400
        orig_name = re.sub(r'[\\/*?:"<>|]', "_", f.filename)
        ext = Path(orig_name).suffix.lower()
        if ftype=="video" and ext not in VIDEO_EXTS:
            return jsonify({"error":"This preset is for videos!"}), 400
        if ftype=="photo" and ext not in IMAGE_EXTS:
            return jsonify({"error":"This preset is for photos!"}), 400

        job_id   = uuid.uuid4().hex[:8]
        src_path = UPLOAD_DIR / f"{Path(orig_name).stem}_{job_id}{Path(orig_name).suffix}"
        f.save(str(src_path))

    jobs[job_id] = {
        "status":"processing","progress":0,
        "message":"Processing...",
        "output":None,"src_name":orig_name,"type":"social",
            "created_at": time.time()
    }
    if ftype == "video":
        if not SOCIAL_PRESETS.get(preset):
            jobs.pop(job_id, None)
            return jsonify({"error": "Invalid preset"}), 400
        threading.Thread(
            target=_social_video_job,
            args=(job_id, src_path, SOCIAL_PRESETS[preset], preset),
            daemon=True
        ).start()
    else:
        if not PHOTO_PRESETS.get(preset):
            jobs.pop(job_id, None)
            return jsonify({"error": "Invalid preset"}), 400
        threading.Thread(
            target=_social_photo_job,
            args=(job_id, src_path, PHOTO_PRESETS[preset], preset),
            daemon=True
        ).start()
    return jsonify({"job_id": job_id})

@_job_guard
def _social_video_job(job_id, src: Path, preset: dict, preset_key: str):
    job     = jobs[job_id]

    dst_name = f"{src.stem}_{preset_key}.mp4"
    dst      = OUTPUT_DIR / dst_name
    counter  = 1
    while dst.exists():
        dst = OUTPUT_DIR / f"{src.stem}_{preset_key}_{counter}.mp4"
        counter += 1

    if not _check_path_len(dst, job_id):
        try: src.unlink()
        except: pass
        return

    total_s = _duration(str(src))

    # Build scale filter
    filters = []
    scale, fps = preset.get("scale"), preset.get("fps")
    if scale == "9:16":
        filters.append(
            "scale=1080:1920:force_original_aspect_ratio=decrease,"
            "pad=1080:1920:(ow-iw)/2:(oh-ih)/2:black,format=yuv420p"
        )
    elif scale == "1080":
        filters.append("scale=1080:-2,format=yuv420p")
    if fps:
        filters.append(f"fps={fps}")

    # Chọn encoder — CHỈ NVENC và AMF, KHÔNG QSV (dùng cache từ startup)
    enc = preset["encoder"]
    gpu, _, _ = get_gpu_info()
    gpu_map = {
        "libx264": ["h264_nvenc","h264_amf"],
        "libx265": ["hevc_nvenc","hevc_amf"],
    }
    for gpu_enc in gpu_map.get(enc, []):
        if gpu_enc in gpu:
            enc = gpu_enc
            break

    crf = preset["crf"]

    pfd, ppath = tempfile.mkstemp(suffix=".txt"); os.close(pfd)

    # hwaccel_output_format cuda không tương thích với CPU filter (scale, fps)
    # → chỉ enable khi không có filter (YouTube preset)
    cmd = [FFMPEG_BIN] + (_hwaccel_args(enc) if not filters else []) + ["-i", str(src)]
    if filters: cmd += ["-vf", ",".join(filters)]
    cmd += ["-c:v", enc] + _quality_args(enc, crf)
    if "265" in enc or "hevc" in enc: cmd += ["-tag:v","hvc1"]
    ab   = preset["audio"]   # e.g. "192k"
    cmd += [
        "-c:a", "aac",   # nhẹ hơn aac_mf, không ngốn CPU
        "-b:a", ab,
        "-ar",  "48000",
        "-ac",  "2",
        "-map_metadata","0",
        "-movflags","use_metadata_tags+faststart",
        "-y","-progress",ppath,"-stats_period","0.5","-nostats", str(dst)
    ]

    oom_retries = 0
    while True:
        job["status"] = "waiting"
        job["message"] = "Waiting in queue..."

        t0 = None
        with _pick_sem(enc):
            if job.get("cancelled"): return
            job["status"] = "processing"
            job["message"] = f"Encoding ({enc})..."
            t0 = time.time()  # Bắt đầu đo từ đây — sau khi được cấp slot
            try:
                rc, is_oom = _run_progress(cmd, job_id, total_s, ppath)
            except Exception as e:
                rc = -1
                is_oom = False
                job.update({"status": "error", "message": f"Error: {e}"})

            if rc != 0 and is_oom:
                pass
            else:
                break

        if rc != 0 and is_oom:
            if job.get("cancelled"): return
            oom_retries += 1
            if oom_retries >= 3:
                job.update({"status": "error", "message": "Insufficient VRAM — failed after 3 retries."})
                try: dst.unlink()
                except: pass
                return
            job["status"] = "waiting"
            job["message"] = f"GPU full, retrying ({oom_retries}/3)..."
            time.sleep(3)

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024
    ela      = time.time() - t0 if t0 else 0
    enc_time = _fmt_time(ela)

    if ok:
        _sync_meta(str(src), str(dst))

        # Probe output
        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = "N/A"
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "audio":
                ac = s.get("codec_name", "N/A").upper()
                out_audio = f"{ac}"
                break

        job.update({
            "status":"done","progress":100,"message":"Completed!",
            "output":str(dst),
            "sz_before":_fmt_size(str(src)),
            "sz_after": _fmt_size(str(dst)),
            "encode_time": enc_time,
            "out_res": out_res,
            "out_fps": out_fps,
            "out_audio": out_audio,
            "out_enc": preset.get("label", "Preset")
        })
    else:
        job.update({"status":"error","message":"Export failed!"})
    try: src.unlink()
    except: pass

@_job_guard
def _social_photo_job(job_id, src: Path, preset: dict, preset_key: str):
    job = jobs[job_id]

    dst_name = f"{src.stem}_{preset_key}.jpg"
    dst      = OUTPUT_DIR / dst_name
    counter  = 1
    while dst.exists():
        dst = OUTPUT_DIR / f"{src.stem}_{preset_key}_{counter}.jpg"
        counter += 1

    tw, th = preset["w"], preset["h"]
    fit    = preset.get("fit","contain")

    # Build filter
    if fit == "contain":
        vf = (
            f"scale={tw}:{th}:force_original_aspect_ratio=decrease,"
            f"pad={tw}:{th}:(ow-iw)/2:(oh-ih)/2:black,"
            f"colorspace=all=bt709:iall=bt709,"
            f"format=yuvj420p"
        )
    else:
        vf = (
            f"scale={tw}:{th}:force_original_aspect_ratio=increase,"
            f"crop={tw}:{th},"
            f"colorspace=all=bt709:iall=bt709,"
            f"format=yuvj420p"
        )

    q   = max(1, int((100 - preset["q"]) / 100 * 31))
    cmd = [FFMPEG_BIN,"-i",str(src),"-vf",vf,"-q:v",str(q),"-frames:v","1","-y",str(dst)]
    job["status"] = "waiting"
    job["message"] = "Waiting in queue..."
    with light_semaphore:
        if job.get("cancelled"): return
        job["status"] = "processing"
        job["message"] = f"Exporting image..."
        try:
            r = subprocess.run(cmd, capture_output=True, text=True, encoding="utf-8", errors="replace")
            rc = r.returncode
        except Exception as e:
            rc = -1
            job.update({"status": "error", "message": f"Error: {e}"})
    ok  = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        # Sync GPS + EXIF từ ảnh gốc
        if EXIFTOOL_BIN:
            subprocess.run([
                EXIFTOOL_BIN,
                "-api","LargeFileSupport=1",
                "-tagsfromfile", str(src),
                "-GPSLatitude","-GPSLongitude","-GPSAltitude",
                "-GPSLatitudeRef","-GPSLongitudeRef",
                "-DateTimeOriginal","-CreateDate",
                "-Make","-Model","-LensModel",
                "--FileModifyDate","--FileCreateDate",
                "-overwrite_original", str(dst)
            ], capture_output=True, text=True, encoding="utf-8", errors="replace")

        out_data  = _probe(str(dst))
        out_res   = "N/A"
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                break

        job.update({
            "status":"done","progress":100,"message":"Completed!",
            "output":str(dst),
            "sz_before":_fmt_size(str(src)),
            "sz_after": _fmt_size(str(dst)),
            "encode_time": "N/A",
            "out_res": out_res,
            "out_fps": "N/A",
            "out_audio": "N/A",
            "out_enc": preset.get("label", "Preset")
        })
    else:
        job.update({"status":"error","message":"Photo export failed!"})
    try: src.unlink()
    except: pass

def _fmt_time(s):
    if s < 60: return f"{s:.0f}s"
    return f"{int(s//60)}:{int(s%60):02d}"

@_job_guard
def _audio_job(job_id, src: Path, af: dict):
    job = jobs[job_id]
    dst = OUTPUT_DIR / f"{job_id}_audio{af['ext']}"

    # Copy mode: detect codec và chọn đúng extension
    if af["codec"] == "copy":
        CODEC_EXT_MAP = {
            "aac":".m4a","mp3":".mp3","opus":".opus",
            "flac":".flac","vorbis":".ogg","ac3":".ac3","eac3":".eac3",
        }
        data = _probe(str(src))
        for s in data.get("streams",[]):
            if s.get("codec_type") == "audio":
                dst = OUTPUT_DIR / f"{job_id}_audio{CODEC_EXT_MAP.get(s.get('codec_name','aac'),'.m4a')}"
                break

    if not _check_path_len(dst, job_id):
        try: src.unlink()
        except: pass
        return

    cmd = [FFMPEG_BIN,"-i",str(src),"-vn","-c:a",af["codec"]] + af["opts"] + ["-y",str(dst)]

    job["status"] = "waiting"
    job["message"] = "Waiting in queue..."

    t0  = time.time()
    with light_semaphore:
        if job.get("cancelled"): return
        job["status"] = "processing"
        job["message"] = f"Extracting audio..."
        jobs[job_id]["progress"] = 50
        try:
            r = subprocess.run(
                cmd, capture_output=True, text=True,
                encoding="utf-8", errors="replace"
            )
            rc = r.returncode
        except Exception as e:
            rc = -1
            job.update({"status": "error", "message": f"Error: {e}"})
    ela = time.time() - t0
    ok  = rc == 0 and dst.exists() and dst.stat().st_size > 0

    if ok:
        enc_time = f"{ela:.0f}s" if ela < 60 else f"{int(ela//60)}:{int(ela%60):02d}"
        job.update({
            "status":"done","progress":100,
            "message":"Completed!","output":str(dst),
            "sz_after":_fmt_size(str(dst)),
            "encode_time":enc_time,
            "out_audio":af["label"],
        })
    else:
        job.update({"status":"error","message":"Audio extraction failed!"})
    try: src.unlink()
    except: pass

@app.route("/api/download", methods=["POST"])
def api_download():
    if not YTDLP_BIN: return jsonify({"error":"yt-dlp not found!"}), 400
    data        = request.get_json()
    sync        = data.get("sync", False)
    url         = data.get("url","").strip()
    quality     = data.get("quality","best")
    ao          = data.get("audio_only", False)
    audio_fmt   = data.get("audio_fmt","m4a")
    playlist    = data.get("playlist", False)
    playlist_titles = data.get("titles", [])
    browser     = data.get("browser","")
    cookies_txt = data.get("cookies_txt", False)
    output_dir  = data.get("output_dir", str(OUTPUT_DIR))
    subs        = data.get("subs", False)
    sub_langs   = data.get("sub_langs", "vi,en").strip() or "vi,en"
    trim_start  = data.get("trim_start", "").strip()
    trim_end    = data.get("trim_end",   "").strip()
    if not output_dir.strip():
        output_dir = str(OUTPUT_DIR)

    if not url or not url.startswith("http"):
        return jsonify({"error":"Invalid URL!"}), 400

    job_id = uuid.uuid4().hex[:8]
    if sync:
        _sync_events[job_id] = threading.Event()
    jobs[job_id] = {
        "status":"processing","progress":0,
        "message":"Fetching info...",
        "output":None,"type":"download","url":url,"playlist":playlist,
        "playlist_titles": playlist_titles,
        "out_dir": output_dir,
        "created_at": time.time(),
    }
    threading.Thread(
        target=_download_job,
        args=(job_id,url,quality,ao,audio_fmt,playlist,browser,cookies_txt,output_dir,subs,sub_langs,trim_start,trim_end),
        daemon=True
    ).start()

    if sync:
        _SYNC_TIMEOUT = 300  # 5 min — ample for LAN short videos; prevents thread exhaustion
        if not _sync_events[job_id].wait(timeout=_SYNC_TIMEOUT):
            return Response(f"Download timeout after {_SYNC_TIMEOUT}s",
                            status=408, mimetype="text/plain")

        job = jobs[job_id]
        if job["status"] == "error":
            return Response(job.get("message", "Download failed"), status=500, mimetype="text/plain")
        if job.get("is_dir"):
            return Response("Playlist not supported in sync mode", status=400, mimetype="text/plain")

        out_file = job.get("output")
        if out_file and os.path.isfile(out_file):
            return send_file(out_file, as_attachment=True, download_name=Path(out_file).name)
        return Response("File not found after download", status=500, mimetype="text/plain")

    return jsonify({"job_id": job_id})

@_job_guard
def _download_job(job_id, url, quality, audio_only,
                  audio_fmt, playlist, browser, cookies_txt, output_dir,
                  subs=False, sub_langs="vi,en",
                  trim_start="", trim_end=""):
    job    = jobs[job_id]

    dl_dir = Path(output_dir)
    try:
        dl_dir.mkdir(parents=True, exist_ok=True)
    except Exception as e:
        job.update({"status": "error", "message": f"Cannot create output folder: {e}"})
        return

    if os.name == "nt" and len(str(dl_dir / "x.mp4")) > _MAX_PATH:
        job.update({"status": "error", "message": f"⚠️ Output folder path too long ({len(str(dl_dir))} chars). Move folder closer to drive root."})
        return

    log_file = dl_dir / f"{job_id}.log"
    job["log_path"] = str(log_file)

    if playlist and not job.get("playlist_titles"):
        try:
            cookies_file = BASE_DIR / "data" / "cookies.txt"
            title_cmd = [YTDLP_BIN, "--flat-playlist", "--yes-playlist", "--print", "title", "--encoding", "utf-8"]
            if cookies_txt and cookies_file.exists():
                title_cmd += ["--cookies", str(cookies_file)]
            elif browser and browser not in ("", "none"):
                title_cmd += ["--cookies-from-browser", browser]
            title_cmd.append(url)
            r = subprocess.run(title_cmd, capture_output=True, text=True, encoding="utf-8", errors="replace")
            if r.returncode == 0:
                job["playlist_titles"] = [ln.strip() for ln in r.stdout.splitlines() if ln.strip()]
        except: pass

    cmd = [
        YTDLP_BIN,
        "--ffmpeg-location", str(Path(FFMPEG_BIN).parent),
        "--newline", "--no-warnings", "--ignore-errors",
        "--embed-metadata",
        "--embed-thumbnail",
        "--encoding", "utf-8",
    ]
    if playlist:
        cmd += ["-o", str(dl_dir / "%(playlist_title)s" / "%(title)s.%(ext)s")]
        cmd.append("--yes-playlist")
    else:
        cmd += ["-o", str(dl_dir / "%(title)s.%(ext)s")]
        cmd.append("--no-playlist")

    # Cookie
    cookies_file = BASE_DIR / "data" / "cookies.txt"
    if cookies_txt and cookies_file.exists():
        cmd += ["--cookies", str(cookies_file)]
    elif browser and browser not in ("", "none"):
        cmd += ["--cookies-from-browser", browser]

    # Audio only
    if audio_only:
        aq = DOWNLOAD_AUDIO_QUALITY.get(audio_fmt, DOWNLOAD_AUDIO_QUALITY["m4a"])
        # M4A: ưu tiên m4a gốc để tránh re-encode AAC
        # MP3/FLAC/WAV: lấy bestaudio (Opus 256k > AAC 128k) để source chất lượng hơn
        audio_src = "bestaudio[ext=m4a]/bestaudio/best" if aq["fmt"] == "m4a" else "bestaudio/best"
        cmd += [
            "-f", audio_src,
            "-x",
            "--audio-format",       aq["fmt"],
            "--audio-quality",      aq["quality"],
            "--convert-thumbnails", "jpg",
            "--ppa", "ThumbnailsConvertor+ffmpeg_o:-vf crop=ih:ih",
            "--add-metadata",
        ]
        job["message"] = f"Downloading audio ({aq['fmt'].upper()})..."
    else:
        fmt = DOWNLOAD_QUALITY.get(quality, DOWNLOAD_QUALITY["best"])
        cmd += ["-f", fmt, "--merge-output-format", "mp4"]

    # Subtitles (video only — không áp dụng cho audio extract)
    if subs and not audio_only:
        cmd += [
            "--write-subs", "--write-auto-subs",
            "--sub-langs", sub_langs,
            "--embed-subs",
            "--sub-format", "srt/vtt/best",
        ]

    # Time range trim — chỉ tải đúng đoạn cần, không tải cả video
    if trim_start or trim_end:
        start_ts = trim_start if trim_start else "0"
        end_ts   = trim_end   if trim_end   else "inf"
        cmd += [
            "--download-sections", f"*{start_ts}-{end_ts}",
            "--force-keyframes-at-cuts",
        ]

    cmd.append(url)

    f_log = open(log_file, "w", encoding="utf-8")
    try:
        proc = subprocess.Popen(
            cmd, stdout=f_log, stderr=subprocess.STDOUT,
            text=True, bufsize=1, encoding="utf-8"
        )
        procs[job_id] = proc
    finally:
        f_log.close()

    dl_files, total, current = [], 0, 0
    last_line = ""
    cookie_error = False
    stream_idx   = -1   # -1 = chưa có stream nào; 0 = video; 1 = audio
    stream_sizes = {}   # {stream_idx: bytes} — để tính tỉ lệ video/audio

    with open(log_file, "r", encoding="utf-8") as f_read:
        while proc.poll() is None:
            line = f_read.readline()
            if not line:
                time.sleep(0.5)
                continue
            line = line.rstrip()
            if "[download]" in line and "%" in line:
                try:
                    parts   = line.split()
                    pct_str = next((p for p in parts if "%" in p), None)
                    if pct_str:
                        pct = float(pct_str.replace("%", ""))

                        # Ghi nhận kích thước stream lần đầu thấy
                        if stream_idx >= 0 and stream_idx not in stream_sizes:
                            sz = _parse_dl_size(line)
                            if sz: stream_sizes[stream_idx] = sz

                        # Tính progress mượt: video stream chiếm ~90%, audio ~10%
                        if audio_only or stream_idx <= 0:
                            raw_pct = int(pct) if audio_only else int(pct * 0.9)
                        else:
                            s0 = stream_sizes.get(0, 0)
                            s1 = stream_sizes.get(stream_idx, 0)
                            w0 = s0 / (s0 + s1) if s0 and s1 else 0.9
                            raw_pct = int(w0 * 100 + pct * (1 - w0))

                        if total > 0:
                            jobs[job_id]["progress"] = min(int((current/total)*100 + (1/total)*raw_pct), 99)
                        else:
                            jobs[job_id]["progress"] = min(raw_pct, 99)

                    speed_str = next((p for p in parts if "/s" in p), None)
                    if speed_str:
                        jobs[job_id]["dl_speed"] = speed_str
                except: pass
            elif "[download] Destination:" in line:
                stream_idx += 1   # -1→0 (video), 0→1 (audio)
                f = Path(line.replace("[download] Destination:","").strip())
                if f not in dl_files: dl_files.append(f)
                title = re.sub(r'\.f\d+$', '', f.stem)
                if total > 0:
                    job["message"] = f"Downloading {current}/{total}: {title}"
                else:
                    job["message"] = f"Downloading: {title}"
            elif "has already been downloaded" in line:
                f = Path(line.replace("[download]", "").replace("has already been downloaded", "").replace("has already been downloaded and merged", "").strip())
                if f not in dl_files: dl_files.append(f)
                current += 1
                title = re.sub(r'\.f\d+$', '', f.stem)
                if total > 0:
                    job["message"] = f"Downloading {current}/{total}: {title}"
                    job["progress"] = min(int((current/total)*100), 99)
                else:
                    job["message"] = f"Downloading: {title}"
            elif "Merging formats into" in line:
                m = re.search(r'Merging formats into \"([^\"]+)\"', line)
                if m:
                    f = Path(m.group(1))
                    if f not in dl_files: dl_files.append(f)
            elif "[ExtractAudio] Destination:" in line:
                f = Path(line.replace("[ExtractAudio] Destination:", "").strip())
                if f not in dl_files: dl_files.append(f)
            elif "Downloading item" in line or "Downloading video" in line:
                try:
                    parts = line.split()
                    if "item" in parts:
                        idx = parts.index("item") + 1
                    elif "video" in parts:
                        idx = parts.index("video") + 1
                    current = int(parts[idx])
                    total   = int(parts[idx+2])
                    stream_idx   = -1   # reset stream tracking cho video mới
                    stream_sizes = {}
                    title_suffix = "..."
                    if playlist and job.get("playlist_titles"):
                        if 1 <= current <= len(job["playlist_titles"]):
                            title_suffix = f": {job['playlist_titles'][current-1]}"
                    job["message"] = f"Downloading {current}/{total}{title_suffix}"
                    if total > 0:
                        job["progress"] = min(int(((current-1)/total)*100), 99)
                except: pass
            elif ("cookie" in line.lower() and any(w in line.lower() for w in ("error","fail","database"))):
                cookie_error = True
            last_line = line

    proc.wait()

    # Fallback: tìm file mới nhất
    dl_files = [f for f in dl_files if f.exists() and f.stat().st_size > 0]
    if not dl_files:
        pool = []
        for e in ["*.mp4","*.mkv","*.webm","*.mp3","*.m4a","*.flac","*.opus"]:
            pool.extend(dl_dir.glob(e))
        valid_pool = [f for f in pool if f.exists() and f.stat().st_size > 0]
        if valid_pool:
            dl_files = [max(valid_pool, key=lambda x: x.stat().st_mtime)]

    success = (proc.returncode == 0 or "Finished downloading playlist" in last_line or bool(dl_files))

    if success:
        time.sleep(0.5) # Wait for file handles to be completely released
        try: log_file.unlink(missing_ok=True)
        except: pass

    if success and dl_files and not job.get("cancelled"):
        if len(dl_files) == 1 and dl_files[0].parent.parent == dl_dir:
            try:
                f = dl_files[0]
                new_f = dl_dir / f.name
                shutil.move(str(f), str(new_f))
                dl_files[0] = new_f
                f.parent.rmdir()
            except: pass

        out_res = out_fps = out_audio = out_enc = out_dur = "N/A"

        if playlist and len(dl_files) > 1:
            total_sz = sum(f.stat().st_size for f in dl_files)
            parent_dir = dl_files[0].parent
            job.update({
                "status":"done","progress":100,
                "message":f"Completed! {len(dl_files)} items",
                "output":str(parent_dir),"is_dir":True,
                "file_count":len(dl_files),
                "sz_after":_fmt_size_bytes(total_sz),
                "dl_title":parent_dir.name,
                "out_dir": str(parent_dir),
                "encode_time": _fmt_time(time.time() - job["created_at"])
            })
        else:
            if not dl_files:
                job.update({"status": "error", "message": "Download failed — no output file found."})
                return
            f = dl_files[-1]
            try:
                p_data = _probe(str(f))
                try:
                    d = float(p_data["format"]["duration"])
                    out_dur = f"{int(d//60)}:{int(d%60):02d}"
                except: pass

                for s in p_data.get("streams", []):
                    if s.get("codec_type") == "video":
                        raw = s.get("codec_name", "").lower()
                        out_enc = CODEC_LABELS.get(raw, raw.upper())
                        out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                        r = s.get("r_frame_rate", "0/1")
                        if "/" in r:
                            n, d2 = r.split("/")
                            if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                        break

                for s in p_data.get("streams", []):
                    if s.get("codec_type") == "audio":
                        ac = s.get("codec_name", "N/A").upper()
                        br = s.get("bit_rate")
                        br_int = int(br) if br else 0
                        if br_int < 10000:
                            tags = s.get("tags", {})
                            br_tag = tags.get("BPS") or tags.get("BPS-eng")
                            if br_tag:
                                try: br_int = int(br_tag)
                                except: pass
                        brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
                        out_audio = f"{ac}{brs}"
                        break
            except: pass

            job.update({
                "status":"done","progress":100,"message":"Completed!",
                "output":str(f),"is_dir":False,
                "sz_after":_fmt_size(str(f)),
                "dl_title":re.sub(r'\.f\d+$', '', f.stem),
                "out_res": out_res,
                "out_fps": out_fps,
                "out_audio": out_audio,
                "out_enc": out_enc,
                "duration": out_dur,
                "encode_time": _fmt_time(time.time() - job["created_at"])
            })
    else:
        if not job.get("cancelled"):
            err = "Download failed!"
            if cookie_error:
                if cookies_txt or (browser and browser not in ("none", "")):
                    err = "Cookie error! Fully close your browser or use cookies.txt"
                else:
                    err = "Download failed! Video may be age-restricted or private (try using cookies)."
            job.update({"status":"error","message":err})

@app.route("/api/download_file/<job_id>")
def api_download_file(job_id):
    job = jobs.get(job_id)
    if not job or not job.get("output"):
        return jsonify({"error":"File not ready!"}), 404

    p = Path(job["output"])
    if not p.exists():
        return jsonify({"error":"File has been auto-deleted (7-day limit). Please re-process."}), 404

    # ── Dùng tên file thực tế từ output path ─────────────
    # Vì backend đã đặt tên đúng (tên_gốc.ext)
    # → dùng thẳng tên file output, không cần build lại
    job_type = job.get("type", "")

    if job_type == "compress":
        # File đã có tên đúng: EVD-15323.mp4
        # Dùng thẳng tên file thực tế
        dl_name = p.name

    elif job_type in ("social", "audio"):
        # Các job khác vẫn build tên từ src_name
        src_name   = job.get("src_name", "video")
        src_stem   = Path(src_name).stem
        clean_stem = re.sub(r"[^\w\-. ]", "_", src_stem)
        suffix_map = {
            "social": "_social",
            "audio":  "_audio",
        }
        suffix  = suffix_map.get(job_type, "")
        dl_name = f"{clean_stem}{suffix}{p.suffix}"

    else:
        # Download — dùng tên file thực tế
        dl_name = p.name

    return send_file(str(p), as_attachment=True, download_name=dl_name)

def _auto_cleanup():
    while True:
        try:
            now = time.time()
            # Dọn dẹp jobs cũ > 7 ngày
            with _jobs_lock:
                to_remove = [jid for jid, j in jobs.items()
                             if now - j.get("created_at", now) > 7 * 24 * 3600]
                for jid in to_remove:
                    jobs.pop(jid, None)
                    procs.pop(jid, None)

            # Dọn dẹp uploads & outputs > 7 ngày
            for d in [UPLOAD_DIR, OUTPUT_DIR, TEMP_DIR]:
                if not d.exists(): continue
                for f in d.iterdir():
                    try:
                        if now - f.stat().st_mtime > 7 * 24 * 3600:
                            if f.is_file(): f.unlink()
                            elif f.is_dir(): shutil.rmtree(f)
                    except: pass
        except: pass
        time.sleep(3600)  # Chạy mỗi giờ

            border = int(w * 0.06)

            x1 = w - r - margin
            y1 = h - r - margin
            x2 = w - margin
            y2 = h - margin

            # Colors

                color = (234, 179, 8, 255) # yellow #eab308

            dc.ellipse((x1, y1, x2, y2), fill=color)

            return img

function resLabel(resStr) {
  if (!resStr || resStr === '?' || resStr === '—') return resStr;
  const m = resStr.match(/^(\d+)[x×](\d+)$/i);
  if (!m) return resStr;
  const w = parseInt(m[1]), h = parseInt(m[2]);
  const longSide = Math.max(w, h);
  const shortSide = Math.min(w, h);
  let lbl = '';
  if (longSide >= 3840 || shortSide >= 2160) lbl = '4K';
  else if (longSide >= 2560 || shortSide >= 1440) lbl = '2K';
  else if (longSide >= 1920 || shortSide >= 1080) lbl = 'FHD';
  else if (longSide >= 1280 || shortSide >= 720)  lbl = 'HD';
  return lbl ? `${resStr} (${lbl})` : resStr;
}

async function onFileChange(input, tab) {
  const file = input.files[0]; if (!file) return;
  S[tab].file = file;
  input.value = "";   // reset để có thể chọn lại cùng file

  const drop  = $("drop-"  + tab);
  const vinfo = $("vinfo-" + tab);
  if (drop)  drop.style.display  = "none";
  if (vinfo) vinfo.style.display = "block";

  sv("vn-name-" + tab, file.name);

  const sfxMap = { compress:"c", audio:"a", social:"s" };
  const sfx    = sfxMap[tab] || tab;
  sv("vi-size-" + sfx, fmtSize(file.size));

  if (tab === "social") {
    const ext = file.name.split(".").pop().toLowerCase();
    socialFileType = ["jpg","jpeg","heic","heif","png","webp","tiff","tif"]
                     .includes(ext) ? "photo" : "video";
    $("card-video-preset").classList.toggle("hidden", socialFileType !== "video");
    $("card-photo-preset").classList.toggle("hidden", socialFileType !== "photo");
    sv("vi-type-s", file.name.split(".").pop().toUpperCase() || (socialFileType === "photo" ? "Image" : "Video"));
    const vf = $("vinfo-video-s");
    if (vf) vf.classList.toggle("hidden", socialFileType !== "video");
    if (socialFileType !== "video") return;
  }

  setBusy(tab, true); // Disable button while uploading and probing
  await probeFile(file, tab);
  setBusy(tab, false); // Enable button after success
}

async function probeFile(file, tab) {
  const loadEl = $("vinfo-loading-" + tab);
  if (loadEl) {
    loadEl.classList.remove("hidden");
    loadEl.textContent = "Uploading (0%)...";
  }

  return new Promise((resolve, reject) => {
    const fd = new FormData();
    fd.append("file", file);

    const xhr = new XMLHttpRequest();
    xhr.open("POST", "/api/probe", true);

    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable && loadEl) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) {
          loadEl.textContent = `Uploading (${pct}%)...`;
        } else {
          loadEl.textContent = "Reading metadata...";
        }
      }
    };

    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          const sfx  = tab === "compress" ? "c" : tab === "social" ? "s" : "a";
          sv("vi-codec-" + sfx, probeCodecMap[data.codec] || data.codec || "—");
          sv("vi-res-"   + sfx, resLabel(data.res) || "—");
          sv("vi-fps-"   + sfx, data.fps ? data.fps + " fps" : "—");
          sv("vi-dur-"   + sfx, data.duration || "—");
          sv("vi-audio-" + sfx, data.audio    || "—");
          sv("vi-size-"  + sfx, data.size     || "—");
          if (data.server_file) S[tab].server_file = data.server_file;
        } catch(e) {}
      }
      if (loadEl) {
        loadEl.classList.add("hidden");
        loadEl.textContent = "Reading metadata..."; // reset text for next time
      }
      resolve();
    };

    xhr.onerror = () => {
      if (loadEl) loadEl.classList.add("hidden");
      resolve();
    };

    xhr.send(fd);
  });
}

function _parseTimeToSec(t) {
  const parts = (t || "").trim().split(":").map(Number).reverse();
  return (parts[0] || 0) + (parts[1] || 0) * 60 + (parts[2] || 0) * 3600;
}
function _secToHMS(s) {
  const h = Math.floor(s / 3600), m = Math.floor((s % 3600) / 60), sec = Math.floor(s % 60);
  return h ? `${h}:${String(m).padStart(2,'0')}:${String(sec).padStart(2,'0')}` : `${m}:${String(sec).padStart(2,'0')}`;
}

// ── Batch download ──────────────────────────────────────────
let _batchJobs = {};   // { job_id: { url, el } }
let _batchTimer = null;

async function startBatchDownload() {
  const text = $("dl-urls").value;
  const urls = text.split("\n").map(u => u.trim()).filter(u => u.startsWith("http"));
  if (!urls.length) return showErr("download", "⚠️ No valid URLs found!");

  hideErr("download");
  $("btn-area-download").classList.add("hidden");
  $("res-batch-dl").classList.remove("hidden");
  $("batch-dl-list").innerHTML = "";
  _batchJobs = {};
  sv("batch-dl-summary", `0 / ${urls.length} done`);

  const opts = {
    quality:     $("dl-quality").value,
    audio_only:  dlType === "audio",
    audio_fmt:   $("dl-audio-fmt").value,
    playlist:    $("dl-playlist").checked,
    titles:      [],
    browser:     dlBrowser === "cookies_txt" ? "" : dlBrowser,
    cookies_txt: dlBrowser === "cookies_txt",
    subs:        $("dl-subs").checked,
    sub_langs:   $("dl-subs-lang-val").value.trim() || "vi,en",
    trim_start:  $("dl-trim-enable").checked ? ($("dl-trim-start").value || "").trim() : "",
    trim_end:    $("dl-trim-enable").checked ? ($("dl-trim-end").value   || "").trim() : "",
    output_dir:  $("dl-output-dir").value.trim(),
  };

  for (const url of urls) {
    const el = _addBatchRow(url);
    try {
      const res  = await fetch("/api/download", {
        method: "POST", headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ ...opts, url }),
      });
      const data = await res.json();
      if (!res.ok || !data.job_id) { _setBatchRowError(el, data.error || "Failed"); continue; }
      _batchJobs[data.job_id] = { url, el };
    } catch(e) {
      _setBatchRowError(el, "Network error");
    }
  }

  if (Object.keys(_batchJobs).length) {
    _batchTimer = setInterval(_pollBatch, 1200);
  }
}

async function _pollBatch() {
  const ids = Object.keys(_batchJobs);
  if (!ids.length) { clearInterval(_batchTimer); _batchTimer = null; return; }

  let res;
  try {
    res = await fetch("/api/status_batch", {
      method: "POST", headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ job_ids: ids }),
    });
  } catch(e) { return; }
  let statuses;
  try { statuses = await res.json(); } catch(e) { return; }

  let doneCount = 0;
  for (const [jid, job] of Object.entries(_batchJobs)) {
    const data = statuses[jid];
    if (!data) continue;
    const { el } = job;
    const statusEl = el.querySelector(".brow-status");
    const barEl    = el.querySelector(".brow-bar");

    if (data.status === "done") {
      doneCount++;
      statusEl.style.cssText = "flex-shrink:0;padding:2px 8px;border-radius:100px;font-size:11px;font-weight:600;background:rgba(var(--green-rgb),.15);color:var(--green)";
      statusEl.textContent   = data.sz_after ? `✓ ${data.sz_after}` : "✓ Done";
      barEl.style.width      = "100%";
      barEl.style.background = "var(--green)";
      delete _batchJobs[jid];
    } else if (data.status === "error") {
      _setBatchRowError(el, data.message);
      delete _batchJobs[jid];
    } else if (data.status === "waiting") {
      statusEl.textContent = "Waiting…";
    } else {
      statusEl.textContent   = `${data.progress || 0}%`;
      barEl.style.width      = (data.progress || 0) + "%";
    }
  }

  const total = ids.length;
  const done  = total - Object.keys(_batchJobs).length;
  sv("batch-dl-summary", `${done} / ${total} done`);

  if (!Object.keys(_batchJobs).length) {
    clearInterval(_batchTimer); _batchTimer = null;
    sv("batch-dl-summary", `${total} / ${total} done`);
    if (window.lucide) lucide.createIcons();
  }
}

  $("dl-title").style.display = "none";

    _trimE.checked = false; _trimE.disabled = true;

    $("dl-trim-start").value = ""; $("dl-trim-end").value = "";

function updateDlTrimHint() {
  const startVal = ($("dl-trim-start").value || "").trim();
  const endVal   = ($("dl-trim-end").value   || "").trim();
  const hint     = $("dl-trim-hint");
  if (!hint) return;
  if (!startVal && !endVal) { hint.textContent = ""; return; }
  const start = _parseTimeToSec(startVal);
  const end   = _parseTimeToSec(endVal);
  if (startVal && endVal) {
    const dur = end - start;
    hint.textContent = dur > 0 ? `Will download ${_secToHMS(dur)} of video` : "⚠️ End must be after start";
    hint.style.color = dur > 0 ? "var(--blue)" : "var(--red)";
  } else if (startVal) {
    hint.textContent = `Will download from ${startVal} to end`;
    hint.style.color = "var(--blue)";
  } else {
    hint.textContent = `Will download from beginning to ${endVal}`;
    hint.style.color = "var(--blue)";
  }
}

// ── Submit ─────────────────────────────────────────────────
async function _submit(tab, endpoint, fd) {
  setBusy(tab, true); hideErr(tab);
  $("prog-"     + tab).classList.remove("hidden");
  $("res-"      + tab).classList.add("hidden");
  $("btn-area-" + tab).classList.add("hidden");

  const bar = $("bar-" + tab); if (bar) bar.style.width = "0%";
  const pct = $("pct-" + tab); if (pct) pct.textContent = "0%";

  S[tab].t0 = Date.now();
  const _msgEl = $("msg-" + tab); if (_msgEl) _msgEl.textContent = "Preparing...";
  try {
    const res  = await fetch(endpoint, { method: "POST", body: fd });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error || "Error");
    S[tab].jobId = data.job_id;
    if (S[tab].timer) clearInterval(S[tab].timer);
    S[tab].timer = setInterval(() => pollJob(tab), 800);
  } catch(e) {
    showErr(tab, e.message);
    $("prog-"     + tab).classList.add("hidden");
    $("btn-area-" + tab).classList.remove("hidden");
    setBusy(tab, false);
  }
}

// ── Poll ───────────────────────────────────────────────────
async function pollJob(tab) {
  const s = S[tab]; if (!s.jobId) return;
  if (s.polling) return;
  s.polling = true;
  try {
    let data;
    try {
      const res = await fetch("/api/status/" + s.jobId);
      if (!res.ok) return;
      data = await res.json();
    } catch(e) { return; }

    const msgEl = $("msg-" + tab); if (msgEl) msgEl.textContent = data.message || "";
    const pct   = data.progress || 0;
    const barEl = $("bar-" + tab); if (barEl) barEl.style.width = pct + "%";
    const pctEl = $("pct-" + tab); if (pctEl) pctEl.textContent = pct + "%";
    if (pct > 5 && s.t0) {
      const ela = (Date.now() - s.t0) / 1000;
      const eta = Math.round(ela / pct * (100 - pct));
      const etaEl = $("eta-" + tab);
      if (etaEl) {
        let metaText = (eta > 0 && isFinite(eta)) ? "~" + eta + "s left" : "";
        if (tab === "download" && data.dl_speed)
          metaText = (metaText ? metaText + " · " : "") + data.dl_speed;
        etaEl.textContent = metaText;
      }
    }

    if (data.status === "done") {
      clearInterval(s.timer);
      s.timer = null;
      $("prog-" + tab).classList.add("hidden");
      $("res-"  + tab).classList.remove("hidden");
      setBusy(tab, false);

      if (tab === "compress") {
        showCompressResult(data);
      } else if (tab === "social") {
        const outPath = data.output || "";
        const sb = parseSize(data.sz_before), sa = parseSize(data.sz_after);
        let pct = null; if (sb > 0 && sa > 0) pct = Math.round((1 - sa/sb) * 100);

        sv("res-s-fname",  outPath.split("\\").pop().split("/").pop() || "—");
        sv("res-s-before", data.sz_before   || "—");
        sv("res-s-after",  data.sz_after    || "—");
        sv("res-s-pct",    pct !== null ? pct + "%" : "—");
        sv("res-s-time",   data.encode_time || "—");
        sv("res-s-res",    resLabel(data.out_res) || "—");
        sv("res-s-fps",    data.out_fps     || "—");
        sv("res-s-audio",  data.out_audio   || "—");
        sv("res-s-enc",    data.out_enc     || "—");
      } else if (tab === "audio") {
        sv("res-a-fmt",  data.out_audio    || "—");
        sv("res-a-size", data.sz_after     || "—");
        sv("res-a-dur",  data.out_duration || "—");
        sv("res-a-time", data.encode_time  || "—");
      } else {
        sv("res-d-title", data.dl_title || "—");
        sv("res-d-size",  data.sz_after  || "—");
        sv("res-d-time",  data.encode_time || "—");
        sv("res-d-dur",   data.duration  || "—");
        sv("res-d-res",   resLabel(data.out_res) || "—");
        sv("res-d-fps",   data.out_fps   || "—");
        sv("res-d-audio", data.out_audio || "—");
        sv("res-d-enc",   data.out_enc   || "—");
        S.download.out_dir = data.out_dir;
        const countItem = $("res-d-count-item");
        if (data.file_count) {
          sv("res-d-count", data.file_count + " files");
          if (countItem) countItem.classList.remove("hidden");
        } else {
          if (countItem) countItem.classList.add("hidden");
        }
        const dlBtn = $("btn-dl-file");
        if (dlBtn) dlBtn.classList.toggle("hidden", !!data.is_dir);

        document.querySelectorAll(".res-d-detail").forEach(el => {
          el.classList.toggle("hidden", !!data.is_dir);
        });
      }
      if (data.warning) {
        const w = $("warn-" + tab);
        if (w) { w.textContent = data.warning; w.classList.remove("hidden"); }
      }

    } else if (data.status === "error") {
      clearInterval(s.timer);
      s.timer = null;
      if (data.message === "Cancelled") {
        $("prog-" + tab).classList.add("hidden");
        $("btn-area-" + tab).classList.remove("hidden");
        setBusy(tab, false);
        return;
      }
      $("prog-"     + tab).classList.add("hidden");
      $("btn-area-" + tab).classList.remove("hidden");
      showErr(tab, data.error || data.message || "Processing failed!", s.jobId);
      setBusy(tab, false);
    }
  } finally {
    s.polling = false;
  }
}

function getClientStatus() {
  const s = S[currentTab];
  if (currentTab === "compress" && s && s.jobs) {
    const isBusy = Object.values(s.jobs).some(j => !j.done);
    if (isBusy) return "compressing";
  } else if (s && s.timer) {
    if (currentTab === "download") return "downloading";
    if (currentTab === "audio") return "extracting";
    return "compressing";
  }
  if ($("vinfo-loading-" + currentTab) && !$("vinfo-loading-" + currentTab).classList.contains("hidden")) return "uploading";
  return "idle";
}

function getOS() {
  const ua = navigator.userAgent;
  if (ua.indexOf("Win") !== -1) return "Windows";
  if (ua.indexOf("like Mac") !== -1) return "iOS";   // must be before "Mac"
  if (ua.indexOf("Mac") !== -1) return "MacOS";
  if (ua.indexOf("Android") !== -1) return "Android";
  if (ua.indexOf("Linux") !== -1) return "Linux";
  return "Unknown OS";
}

function getBrowser() {
  const ua = navigator.userAgent;
  if (ua.indexOf("Edg") !== -1) return "Edge";
  if (ua.indexOf("Chrome") !== -1) return "Chrome";
  if (ua.indexOf("Firefox") !== -1) return "Firefox";
  if (ua.indexOf("Safari") !== -1) return "Safari";
  return "Unknown Browser";
}

function _updateAdminUI(data) {
  const sItem = $("server-status-item");
  if (sItem) sItem.style.display = "flex";

  const dot = $("server-status-dot");
  const txt = $("server-status-txt");
  if (dot && txt) {
    let sc = "var(--yellow)";
    if (data.server_state === "IDLE") sc = "var(--green)";
    else if (data.server_state === "DOWNLOADING" || data.server_state === "UPLOADING") sc = "var(--blue)";
    else if (data.server_state === "EXTRACTING") sc = "var(--purple-l)";
    else if (data.server_state === "LOADING" || data.server_state === "CONNECTING") sc = "var(--text-2)";

    if (data.server_state === "IDLE") {
      dot.className = "status-dot dot-ok";
      dot.style.background = "";
      dot.style.boxShadow = "";
    } else {
      dot.className = "status-dot";
      dot.style.background = sc;
      dot.style.boxShadow = `0 0 8px ${sc}`;
    }
    txt.style.color = sc;
    txt.style.textShadow = `0 0 8px ${sc}`;
    txt.textContent = `${data.server_state}`;
  }

  const clients = data.clients || [];
  const remoteClients = clients.filter(c => c.ip !== "127.0.0.1" && c.ip !== "::1");

  const countText = remoteClients.length + " Client" + (remoteClients.length !== 1 ? "s" : "");
  if ($("dash-client-count")) $("dash-client-count").textContent = countText;
  if ($("dash-client-count-2")) $("dash-client-count-2").textContent = countText;

  const list = $("dash-client-list");
  if (!list) return;

  if (remoteClients.length === 0) {
    list.innerHTML = `<div style="color:var(--text-3); text-align:center; padding:10px 0;">No external clients connected</div>`;
    return;
  }

  list.innerHTML = remoteClients.map(c => {
    const osValue      = c.os      && c.os      !== "Unknown OS"      ? c.os      : "Unknown OS";
    const browserValue = c.browser && c.browser !== "Unknown Browser" ? c.browser : "Unknown Browser";
    const osBrowserText = `${osValue} (${browserValue})`;

    let statusColor = "var(--green)", statusText = "ACTIVE";
    if      (c.status === "compressing") { statusColor = "var(--yellow)";   statusText = `COMPRESSING ${c.progress}`; }
    else if (c.status === "downloading") { statusColor = "var(--blue)";     statusText = `DOWNLOADING ${c.progress}`; }
    else if (c.status === "uploading")   { statusColor = "var(--yellow)";   statusText = "UPLOADING"; }
    else if (c.status === "extracting")  { statusColor = "var(--purple-l)"; statusText = `EXTRACTING ${c.progress}`; }
    else if (c.status === "loading")     { statusColor = "var(--text-2)";   statusText = "LOADING"; }
    else if (c.status === "connecting")  { statusColor = "var(--text-2)";   statusText = "CONNECTING"; }

    const fileText = c.filename
      ? `<div style="font-size:10px; color:var(--text-2); margin-top:3px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis;"><svg width="11" height="11" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" style="vertical-align:-1px"><path d="M13 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V9z"/><polyline points="13 2 13 9 20 9"/></svg> ${c.filename}</div>`
      : "";
    return `
      <div style="background:rgba(255,255,255,.03); border:1px solid var(--card-border); padding:8px 12px; border-radius:var(--radius-sm);">
        <div style="display:flex; justify-content:space-between; margin-bottom:4px;">

            <span style="font-size:11px; color:var(--text-2); margin-left:6px; padding:2px 6px; background:rgba(255,255,255,0.1); border-radius:100px;">${osBrowserText}</span>
          </div>
          <span style="color:${statusColor}; font-weight:600; font-size:11px; text-transform:uppercase; display:flex; align-items:center; gap:4px; text-shadow:0 0 8px ${statusColor};"><span class="status-dot" style="background:${statusColor}; box-shadow:0 0 8px ${statusColor};"></span>${statusText}</span>

let _pingBusy = false;
async function sendPing() {
  if (_pingBusy) return;
  _pingBusy = true;
  try {
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), 3000);
    const res = await fetch("/api/ping", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      signal: controller.signal,
      body: JSON.stringify({
        client_id: clientId,
        status: getClientStatus(),
        tab: currentTab,
        filename: getClientFilename(),
        progress: getClientProgress(),
        os: getOS(),
        browser: getBrowser()
      })
    });
    clearTimeout(timeoutId);
    if (res.ok) {
      _updateAdminUI(await res.json());
    } else {
      throw new Error("Failed");
    }
  } catch (e) {
    _setServerOffline();
  } finally {
    _pingBusy = false;
  }
}

    if (!_pingTimer) _pingTimer = setInterval(sendPing, 1000);
  }
});

          </div>
        `;
      }).join('');

      let minCreate = Date.now();
      let hasCreate = false;
      const serverTime = data.server_time ? data.server_time * 1000 : Date.now();
      
      jids.forEach((jid, i) => {
        const id = 'prog-item-' + i;
        const cBtn = $("btn-cancel-" + id);
        if (cBtn && S.compress.jobs[jid].done === false) {
          cBtn.style.display = "block";
          cBtn.onclick = () => cancelSingleJob(jid, id);
        }
        if (data.jobs[jid].created_at) {
          // Calculate how long ago the job was created on the server
          const elapsedOnServer = serverTime - (data.jobs[jid].created_at * 1000);
          // Map that absolute elapsed time to the client's current Date.now()
          const t = Date.now() - elapsedOnServer;
          if (t < minCreate) { minCreate = t; hasCreate = true; }
        }
      });
      if (hasCreate) S.compress.batchStartTime = minCreate;

      if (S.compress.timer) clearInterval(S.compress.timer);
      S.compress.timer = setInterval(pollBatchProgress, 1000);
      pollBatchProgress();
    }
  } catch (e) {
    console.error("Failed to restore batch:", e);
  }
}

async function onFileChange(input, tab) {
  if (tab !== 'compress') {
    // Old logic for social/audio
    const file = input.files[0]; if (!file) return;
    S[tab].file = file;
    input.value = "";
    const drop = $("drop-" + tab); const vinfo = $("vinfo-" + tab);
    if (drop) drop.style.display = "none";
    if (vinfo) vinfo.style.display = "block";
    sv("vn-name-" + tab, file.name);
    if (tab === "social") {
      const ext = file.name.split(".").pop().toLowerCase();
      socialFileType = ["jpg","jpeg","heic","heif","png","webp","tiff","tif"].includes(ext) ? "photo" : "video";
      $("card-video-preset").classList.toggle("hidden", socialFileType !== "video");
      $("card-photo-preset").classList.toggle("hidden", socialFileType !== "photo");
      sv("vi-type-s", socialFileType === "photo" ? "Photo" : "Video");
      const vf = $("vinfo-video-s");
      if (vf) vf.classList.toggle("hidden", socialFileType !== "video");
      if (socialFileType !== "video") return;
    }
    setBusy(tab, true);
    await probeFile(file, tab);
    setBusy(tab, false);
    return;
  }

  // BATCH LOGIC FOR COMPRESS
  if (!input.files || input.files.length === 0) return;
  const files = Array.from(input.files);
  // input.value = ""; // KHÔNG RESET Ở ĐÂY, TRÁNH LỖI MẤT FILE TRÊN SAFARI/CHROME

  S.compress.files = files.map((f, i) => ({
    id: i,
    file: f,
    server_file: null,
    status: 'pending', // pending, probing, ready, error
    size: f.size
  }));

  sv("vn-name-compress", `Selected ${files.length} videos`);

  const drop = $("drop-compress"); const vinfo = $("vinfo-compress");
  if (drop) drop.style.display = "none";
  if (vinfo) vinfo.style.display = "block";

  renderBatchList();

  setBusy('compress', true);
  const fileCount = S.compress.files.length;
  for (let i = 0; i < fileCount; i++) {
    if (i >= S.compress.files.length) break;
    await probeBatchFile(i);
  }
  setBusy('compress', false);
}

async function probeBatchFile(index) {
  const fObj = S.compress.files[index];
  fObj.status = 'probing';
  renderBatchList();

  const loadEl = $("vinfo-loading-compress");
  loadEl.classList.remove("hidden");
  loadEl.textContent = `Uploading file ${index+1}/${S.compress.files.length} (0%)...`;

  return new Promise((resolve) => {
    const fd = new FormData();
    fd.append("file", fObj.file);
    const xhr = new XMLHttpRequest();
    xhr.open("POST", "/api/probe", true);
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) loadEl.textContent = `Uploading file ${index+1} (${pct}%)...`;
        else loadEl.textContent = `Reading metadata for file ${index+1}...`;
      }
    };
    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          if (data.server_file) {
            fObj.server_file = data.server_file;
            fObj.status = 'ready';
            fObj.meta = data;
          } else {
            fObj.status = 'error';
            fObj.errMsg = data.error || "No server_file";
          }
        } catch(e) {
          fObj.status = 'error';
          fObj.errMsg = "JSON Parse Error";
        }
      } else {
        fObj.status = 'error';
        try {
          const errData = JSON.parse(xhr.responseText);
          fObj.errMsg = errData.error || `HTTP ${xhr.status}`;
        } catch(e) {
          fObj.errMsg = `HTTP ${xhr.status}: ${xhr.statusText}`;
        }
      }
      loadEl.classList.add("hidden");
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.onerror = () => {
      fObj.status = 'error';
      fObj.errMsg = "Network disconnected";
      loadEl.classList.add("hidden");
      renderBatchList();
      _updateBatchStartBtn();
      resolve();
    };
    xhr.send(fd);
  });
}

        <div class="prog-meta" style="margin-top:4px"><span id="pct-${id}">0%</span><span id="eta-${id}"></span></div>
      </div>
    `;
  }).join('');

  // Submit all requests to backend (they will queue up via lock)
  const clientBatchId = Date.now().toString(36) + Math.random().toString(36).substr(2, 5);
  for (let i = 0; i < readyFiles.length; i++) {
    const fObj = readyFiles[i];
    const id = 'prog-item-' + i;

    const fd = new FormData();
    fd.append("client_batch_id", clientBatchId);
    fd.append("server_file", fObj.server_file);
    fd.append("original_name", fObj.file.name);
    fd.append("encoder", enc);
    fd.append("format", fmt);
    fd.append("crf", crf);
    if (muteAudio) fd.append("mute", "1");

    const ui_meta = fObj.meta ? {
      codec: fObj.meta.codec,
      res: fObj.meta.res,
      duration: fObj.meta.duration,
      fps: fObj.meta.fps,
      audio: fObj.meta.audio,
      size: fObj.size
    } : {};
    fd.append("ui_meta", JSON.stringify(ui_meta));

    try {
      const res = await fetch("/api/compress", { method: "POST", body: fd });
      const data = await res.json();
      const fileIdx = S.compress.files.indexOf(fObj);
      if (res.ok && data.job_id) {
        S.compress.jobs[data.job_id] = { ui_id: id, file: fObj.file.name, fileIdx, done: false };
        const cBtn = $("btn-cancel-" + id);
        if (cBtn) {
          cBtn.style.display = "block";
          cBtn.onclick = () => cancelSingleJob(data.job_id, id);
        }
      } else {
        S.compress.jobs[id] = { ui_id: id, file: fObj.file.name, fileIdx, done: true, result: { status: "error", message: data.error || "Rejected" } };
        const progItem = $(id);
        if (progItem) {
          progItem.innerHTML = _buildJobResultHTML(id, fObj.file.name, S.compress.jobs[id].result, `<span style="margin-right:4px;">${i + 1}.</span>`);
        }
      }
    } catch (e) {
      sv('msg-' + id, "Connection error");
    }
  }

  // Start polling
  if (S.compress.timer) clearInterval(S.compress.timer);
  S.compress.timer = setInterval(pollBatchProgress, 1000);
}

function parseSize(szStr) {
  if (!szStr) return 0;
  const m = szStr.match(/([\d.]+)\s*([KMGT]?B)/i);
  if (!m) return parseFloat(szStr) || 0;
  const v = parseFloat(m[1]), u = m[2].toUpperCase();
  if (u === 'KB') return v * 1024;
  if (u === 'MB') return v * 1024 * 1024;
  if (u === 'GB') return v * 1024 * 1024 * 1024;
  return v;
}

let _batchPollBusy = false;
async function pollBatchProgress() {
  if (_batchPollBusy) return;
  _batchPollBusy = true;
  try {
  const allJobIds = Object.keys(S.compress.jobs);
  const activeJobIds = allJobIds.filter(jid => !S.compress.jobs[jid].done);
  const doneCount = allJobIds.length - activeJobIds.length;

  const progCountEl = $("batch-progress-count");
  if (progCountEl && allJobIds.length > 0) {
    progCountEl.textContent = `(${doneCount}/${allJobIds.length} videos)`;
  }

  if (activeJobIds.length === 0) {
    clearInterval(S.compress.timer);
    S.compress.timer = null;
    setBusy('compress', false);
    // Dừng timer hiển thị khi xong
    const timerEl = $("batch-elapsed-timer");
    if (timerEl && S.compress.batchStartTime) {
      const totalSec = Math.round((Date.now() - S.compress.batchStartTime) / 1000);
      const m = Math.floor(totalSec / 60), s = totalSec % 60;
      timerEl.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
    }
    renderBatchResult();
    return;
  }

  // Cập nhật bộ đếm giờ live mỗi giây
  const _timerEl = $("batch-elapsed-timer");
  if (_timerEl && S.compress.batchStartTime) {
    const _elapsed = Math.round((Date.now() - S.compress.batchStartTime) / 1000);
    const _m = Math.floor(_elapsed / 60), _s = _elapsed % 60;
    _timerEl.textContent = `${String(_m).padStart(2,'0')}:${String(_s).padStart(2,'0')}`;
  }

  const _abortCtrl = new AbortController();
  const _abortTid  = setTimeout(() => _abortCtrl.abort(), 3000);
  try {
    const res = await fetch("/api/status_batch", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ job_ids: activeJobIds }),
      signal: _abortCtrl.signal
    });
    clearTimeout(_abortTid);
    if (!res.ok) return;
    const batchData = await res.json();

    activeJobIds.forEach(jid => {
      const data = batchData[jid];
      if (!data) return;

      const jobMeta = S.compress.jobs[jid];
      const uid = jobMeta.ui_id;

      const pct = data.progress || 0;
      const elPct = $('pct-' + uid);
      const elMsg = $('msg-' + uid);
      const elBar = $('bar-' + uid);
      const elEta = $('eta-' + uid);

      if (elPct) elPct.innerText = pct + "%";
      if (elMsg) elMsg.innerText = data.message || "Processing...";
      if (elBar) elBar.style.width = pct + "%";

      if (pct > 0 && pct < 100) {
        if (!jobMeta.t0) jobMeta.t0 = Date.now() / 1000;
        const ela = (Date.now() / 1000) - jobMeta.t0;
        const eta = Math.round((ela / pct) * (100 - pct));
        if (elEta) elEta.innerText = eta > 0 ? "~" + eta + "s left" : "";
      } else {
        if (elEta) elEta.innerText = "";
      }

      if (data.status === "done" || data.status === "error") {
        jobMeta.done = true;
        jobMeta.result = data;
        const btnCancel = $("btn-cancel-" + uid);
        if (btnCancel) btnCancel.style.display = "none";
        const progItem = $(uid);
        if (progItem) {
          progItem.style.background = "transparent";
          progItem.style.padding = "12px 0";
          progItem.style.borderBottom = "1px solid rgba(255,255,255,0.05)";
          const idx = jobMeta.fileIdx !== undefined ? jobMeta.fileIdx
                    : S.compress.files ? S.compress.files.findIndex(f => f.file.name === jobMeta.file) : -1;
          const idxStr = idx >= 0 ? `<span style="margin-right:4px;">${idx + 1}.</span>` : '';
          progItem.innerHTML = _buildJobResultHTML(jid, jobMeta.file, data, idxStr);
        }
      }
    });
  } catch (e) {
    clearTimeout(_abortTid);
    if (e.name !== 'AbortError') console.error("Batch polling error:", e);
  }
  } finally {
    _batchPollBusy = false;
  }
}

function renderBatchResult() {
  $("prog-compress").classList.add("hidden");
  $("res-compress").classList.remove("hidden");

  const jobCount  = Object.keys(S.compress.jobs).length;
  const doneCount = Object.values(S.compress.jobs).filter(j => j.result && j.result.status === 'done').length;
  const errCount = jobCount - doneCount;

  let theme = "green";
  let themeL = "green-l";
  let titleHtml = `<i data-lucide="check-circle-2" style="width:16px;height:16px;vertical-align:-2px;color:var(--green)"></i> All tasks completed!`;

  if (doneCount === 0 && errCount > 0) {
    theme = "red";
    themeL = "red-l";
    titleHtml = `<i data-lucide="x-circle" style="width:16px;height:16px;vertical-align:-2px;color:var(--red)"></i> Batch failed!`;
  } else if (errCount > 0) {
    theme = "yellow";
    themeL = "yellow-l";
    titleHtml = `<i data-lucide="alert-triangle" style="width:16px;height:16px;vertical-align:-2px;color:var(--yellow)"></i> Batch finished with errors`;
  }

  const badge = $("res-compress").querySelector(".res-badge");
  badge.innerHTML = titleHtml;
  badge.style.fontSize = '20px';
  badge.style.color = `var(--${themeL})`;
  badge.style.textShadow = `0 0 12px rgba(var(--${theme}-rgb), 0.4)`;

  const resBox = $("res-compress").querySelector(".result-box");
  resBox.style.background = `linear-gradient(180deg, rgba(var(--${theme}-rgb), 0.18) 0%, rgba(var(--${theme}-rgb), 0.00) 100%)`;
  resBox.style.border = `1px solid rgba(var(--${theme}-rgb), 0.35)`;
  resBox.style.boxShadow = `0 16px 40px rgba(0, 0, 0, 0.45)`;

  // Tính tổng thời gian từ lúc bấm Bắt đầu nén đến lúc xong tất cả
  let totalTimeHtml = '';
  if (S.compress.batchStartTime) {
    const totalMs   = Date.now() - S.compress.batchStartTime;
    const totalSec  = Math.round(totalMs / 1000);
    const mins      = Math.floor(totalSec / 60);
    const secs      = totalSec % 60;
    const timeStr   = mins > 0 ? `${mins}m ${secs}s` : `${totalSec}s`;

    totalTimeHtml = `
      <div style="margin-top:10px; padding:10px 14px; background:rgba(var(--${theme}-rgb),0.06); border:1px solid rgba(var(--${theme}-rgb),0.2); border-radius:var(--radius-sm); display:flex; align-items:center; justify-content:space-between; flex-wrap:wrap; gap:8px;">
        <span style="font-size:12px; color:var(--text-2); display:flex; align-items:center; gap:6px;">
          <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" style="vertical-align:-2px;color:var(--${theme})"><circle cx="12" cy="12" r="10"/><polyline points="12 6 12 12 16 14"/></svg>
          Total processing time for ${jobCount} files
        </span>
        <span style="font-size:14px; font-weight:700; color:var(--${theme});">${timeStr}</span>
      </div>`;
  }

  // Gắn tổng thời gian vào sau badge "Đã xử lý xong toàn bộ!"
  let totalEl = $("batch-total-time");
  if (!totalEl) {
    totalEl = document.createElement('div');
    totalEl.id = "batch-total-time";
    badge.insertAdjacentElement('afterend', totalEl);
  }
  totalEl.innerHTML = totalTimeHtml;
  lucide.createIcons();

  const itemStyle = "padding:12px 0; border-bottom:1px solid rgba(255,255,255,0.05); font-size:13px;";
  $("batch-result-list").innerHTML = (S.compress.files || [])
    .map((f, idx) => {
      const jobEntry = Object.entries(S.compress.jobs).find(([k, v]) => v.fileIdx === idx);
      if (!jobEntry) return '';
      const [jid, job] = jobEntry;
      const idxStr = `<span style="margin-right:4px;">${idx + 1}.</span>`;
      return `<div style="${itemStyle}">${_buildJobResultHTML(jid, job.file, job.result, idxStr)}</div>`;
    })
    .join('');
}
