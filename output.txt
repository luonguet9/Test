def _setup_ffmpeg():
    for fb, pb in [
        (BASE_DIR/"bin/ffmpeg/bin/ffmpeg.exe", BASE_DIR/"bin/ffmpeg/bin/ffprobe.exe"),
        (BASE_DIR/"bin/ffmpeg/bin/ffmpeg",     BASE_DIR/"bin/ffmpeg/bin/ffprobe"),

    return None, None

def _fmt_size(path):

def _fmt_size_bytes(b):
    if b < 1024**2:   return f"{b/1024:.1f} KB"
    if b < 1024**3:   return f"{b/1024**2:.1f} MB"
    return f"{b/1024**3:.2f} GB"

def _make_output_path(src: Path, ext: str, suffix: str = "_compress") -> Path:
    """Tạo tên output trong OUTPUT_DIR, tự tăng số nếu file đã tồn tại."""
    base = re.sub(r"_(compress|social|audio)(_\d+)?$", "", src.stem)
    cand = OUTPUT_DIR / f"{base}{suffix}{ext}"
    if not cand.exists():
        return cand
    i = 2
    while True:
        cand = OUTPUT_DIR / f"{base}{suffix}_{i}{ext}"
        if not cand.exists():
            return cand
        i += 1

def _probe(path):
    if not FFPROBE_BIN: return {}
    r = subprocess.run(
        [FFPROBE_BIN,"-v","quiet","-print_format","json",
         "-show_streams","-show_format", path],
        capture_output=True, text=True)
    try: return json.loads(r.stdout)
    except: return {}

def _duration(path):
    if not FFPROBE_BIN: return 0.0
    r = subprocess.run(
        [FFPROBE_BIN,"-v","quiet","-show_entries","format=duration",
         "-of","default=noprint_wrappers=1:nokey=1", path],
        capture_output=True, text=True)
    try: return float(r.stdout.strip())
    except: return 0.0

def _read_progress(fpath):
    try:
        txt  = Path(fpath).read_text(encoding="utf-8", errors="ignore")
        done = "progress=end" in txt
        t    = 0.0
        for ln in reversed(txt.splitlines()):
            if ln.startswith("out_time_ms="):
                try:
                    ms = int(ln.split("=")[1])
                    if ms > 0: t = ms/1_000_000
                except: pass
                break
        return t, done
    except: return 0.0, False

def _test_enc(codec):
    if not FFMPEG_BIN: return False
    r = subprocess.run(
        [FFMPEG_BIN,
         "-f", "lavfi", "-i", "nullsrc=s=256x256",
         "-t", "0.1",
         "-pix_fmt", "yuv420p",   # NVENC/AMF yêu cầu yuv420p
         "-c:v", codec,
         "-f", "null", "-"],
        capture_output=True, text=True
    )
    return r.returncode == 0

def _detect_gpu() -> dict:
    candidates = [
        ("av1_nvenc",  "NVIDIA", "AV1"),
        ("av1_amf",    "AMD",    "AV1"),
        ("av1_qsv",    "Intel",  "AV1"),
        ("hevc_nvenc", "NVIDIA", "H.265"),
        ("hevc_amf",   "AMD",    "H.265"),
        ("hevc_qsv",   "Intel",  "H.265"),
        ("h264_nvenc", "NVIDIA", "H.264"),
        ("h264_amf",   "AMD",    "H.264"),
        ("h264_qsv",   "Intel",  "H.264"),
    ]
    found, seen = {}, set()
    for codec, vendor, fmt in candidates:
        key = (vendor, fmt)
        if key not in seen and _test_enc(codec):
            seen.add(key)
            found[codec] = {"vendor": vendor, "format": fmt}
    return found

def get_gpu_info():
    global GLOBAL_GPU_INFO, GLOBAL_DEFAULT_ENC, GLOBAL_ENC_GROUPS
    if GLOBAL_GPU_INFO is None:
        GLOBAL_GPU_INFO = _detect_gpu()

        GLOBAL_ENC_GROUPS = _build_encoder_options(GLOBAL_GPU_INFO)
    return GLOBAL_GPU_INFO, GLOBAL_DEFAULT_ENC, GLOBAL_ENC_GROUPS

def _get_default_encoder(gpu: dict) -> str:
    """
    Thứ tự ưu tiên:
    1. NVIDIA/AMD AV1
    2. NVIDIA/AMD H.265  ← GPU trước CPU
    3. CPU H.265 fallback
    QSV không bao giờ là default
    """
    for enc in ["av1_nvenc", "av1_amf"]:
        if enc in gpu: return enc
    for enc in ["hevc_nvenc", "hevc_amf"]:
        if enc in gpu: return enc
    return "libx265"

def _sync_meta(src: str, dst: str) -> bool:
    """
    Bước 1: Apple QuickTime Keys (GPS, Camera, Device) — quan trọng nhất cho iPhone
    Bước 2: Standard metadata all:all
    """
    if not EXIFTOOL_BIN: return False

    # Bước 1: Copy Apple QuickTime Keys atoms
    cmd1 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        # Apple QuickTime Keys — GPS
        "-Keys:GPSCoordinates",
        "-Keys:LocationBody",
        "-Keys:LocationDate",
        "-Keys:LocationName",
        # Apple QuickTime Keys — Device
        "-Keys:Make",
        "-Keys:Model",
        "-Keys:Software",
        # Apple QuickTime Keys — Camera
        "-Keys:CameraLensModel",
        "-Keys:CreationDate",
        # Live Photo
        "-Keys:ContentIdentifier",
        
        # Thêm UserData fallback (iOS 17/18 Apple Photos fallback)
        "-UserData:Make<Keys:Make",
        "-UserData:Model<Keys:Model",
        "-UserData:GPSCoordinates<Keys:GPSCoordinates",
        "-UserData:CreationDate<Keys:CreationDate",
        "-overwrite_original",
        dst
    ]
    r1 = subprocess.run(cmd1, capture_output=True, text=True)

    # Bước 2: Copy standard metadata (không ghi đè Keys đã copy ở bước 1)
    cmd2 = [
        EXIFTOOL_BIN,
        "-api", "LargeFileSupport=1",
        "-api", "QuickTimeUTC=1",
        "-tagsfromfile", src,
        "-all:all",
        "--FileModifyDate",
        "--FileCreateDate",
        "--Keys:all",   # Không ghi đè Keys từ bước 1
        "-overwrite_original",
        dst
    ]
    r2 = subprocess.run(cmd2, capture_output=True, text=True, encoding="utf-8", errors="replace")

    return r1.returncode == 0 or r2.returncode == 0

# PROGRESS RUNNER
# ══════════════════════════════════════════════════════════════
def _run_progress(cmd, job_id, total_s, ppath):
    job = jobs.get(job_id, {})
    stem = "unknown"
    if "src_name" in job:
        stem = Path(job["src_name"]).stem
    elif "url" in job:
        stem = "download_" + re.sub(r"[^\w]", "", job["url"].split("/")[-1][:15])
    stem = re.sub(r"[^\w\-.]", "_", stem)
    ts = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    log_name = f"ffmpeg_err_{stem}_{ts}.log"
    log_path = Path(OUTPUT_DIR) / log_name
    job["log_file"] = log_name
    with open(log_path, "w", encoding="utf-8") as f:
        f.write("CMD: " + " ".join(cmd) + "\n\n")
        proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=f)
        procs[job_id] = proc
        while True:
            time.sleep(0.5)
            cur, done = _read_progress(ppath)
            if total_s > 0 and cur > 0:
                jobs[job_id]["progress"] = min(int(cur/total_s*100), 99)
            if done or proc.poll() is not None: break
        proc.wait()
    procs.pop(job_id, None)
    try: os.remove(ppath)
    except: pass
    
    # Xóa log nếu nén thành công để đỡ rác ổ cứng
    if proc.returncode == 0:
        try: os.remove(log_path)
        except: pass
        
    return proc.returncode

def _build_admin_response():
    now = time.time()
    expired = [cid for cid, info in active_clients.items() if now - info["last_seen"] > 5]
    for cid in expired:
        active_clients.pop(cid, None)

    active_states = []
    has_loading = False
    
    for j in jobs.values():
        status = j.get("status")
        if status in ["processing", "waiting"]:
            jtype = j.get("type")
            if jtype == "audio" and "EXTRACTING" not in active_states:
                active_states.append("EXTRACTING")
            elif jtype == "download" and "DOWNLOADING" not in active_states:
                active_states.append("DOWNLOADING")
            elif jtype not in ["audio", "download"] and "COMPRESSING" not in active_states:
                active_states.append("COMPRESSING")
        elif status in ["downloading", "probing", "uploading"]:
            has_loading = True

    if active_states:
        server_state = " / ".join(active_states)
    elif has_loading:
        server_state = "LOADING"
    else:
        server_state = "IDLE"

    return {
        "server_state": server_state,
        "clients": [
            {
                "id": cid,
                "ip": info["ip"],
                "os": info.get("os", "Unknown OS"),
                "browser": info.get("browser", "Unknown Browser"),
                "raw_ua": info.get("raw_ua", ""),
                "status": info["status"],
                "tab": info["tab"],
                "filename": info["filename"],
                "progress": info["progress"]
            }
            for cid, info in active_clients.items()
        ]
    }

def _job_guard(fn):
    """Bắt exception không xử lý trong job thread → job không bị stuck 'processing'."""
    def wrapper(job_id, *args, **kwargs):
        try:
            fn(job_id, *args, **kwargs)
        except Exception as e:
            if job_id in jobs:
                jobs[job_id].update({"status": "error", "message": f"Lỗi: {e}"})
    return wrapper

@_job_guard
def _compress_job(job_id, src: Path, enc: str, crf: int,
                  out_ext: str, mute: bool = False):
    job    = jobs[job_id]
    family = _get_codec_family(enc)
    t0     = time.time()

    src_name = job.get("src_name", "video")
    src_stem = re.sub(r"[^\w\-. ]", "_", Path(src_name).stem)

    # ── Kiểm tra AV1 + MOV ───────────────────────────────

    # ── Kiểm tra AV1 + MOV ────────────────────────────────────
    av1_warning = ""
    if family == "av1" and out_ext == ".mov":
        out_ext = ".mp4"
        av1_warning = " (Tự đổi sang MP4)"

    # ── Kiểm tra container vs codec ───────────────────────
    supported = CONTAINER_CODEC_SUPPORT.get(out_ext, ["h264"])
    if family not in supported:
        if out_ext == ".webm":
            enc = "libaom-av1"
            job["message"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> .WebM chỉ hỗ trợ AV1 → tự động chuyển libaom-AV1"
        else:
            enc = "libx264"
            job["message"] = f"<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> {out_ext.upper()} không hỗ trợ {family.upper()} → chuyển H.264"
        family = _get_codec_family(enc)

    # ── Output path ───────────────────────────────────────
    dst = OUTPUT_DIR / f"{src_stem}{out_ext}"
    counter = 1
    while dst.exists():
        dst = OUTPUT_DIR / f"{src_stem}_{counter}{out_ext}"
        counter += 1

    total_s             = _duration(str(src))
    real_enc, enc_extra = _get_real_enc_extra(enc)

    pfd, ppath = tempfile.mkstemp(suffix=".txt", prefix="vs_")
    os.close(pfd)

    # ── Tìm audio stream tốt nhất ────────────────────────
    best_audio_idx = 0
    best_audio_br  = 0
    probe_data     = _probe(str(src))
    audio_streams_raw = [
        s for s in probe_data.get("streams", [])
        if s.get("codec_type") == "audio" and s.get("codec_name") and s.get("codec_name") != "none"
    ]
    audio_streams = list(enumerate(audio_streams_raw))
    for stream_idx, s in audio_streams:
        br = 0
        raw_br = s.get("bit_rate")
        if raw_br:
            try: br = int(raw_br)
            except: pass
        if br < 10000:
            tags = s.get("tags", {})
            bps  = tags.get("BPS") or tags.get("BPS-eng")
            if bps:
                try: br = int(bps)
                except: pass
        if br < 10000:
            ch = s.get("channels", 0)
            sr = s.get("sample_rate", 0)
            if ch and sr:
                br = ch * int(sr) * 16
        if br > best_audio_br:
            best_audio_br  = br
            best_audio_idx = stream_idx

    # ── Build FFmpeg command ──────────────────────────────
    cmd = [FFMPEG_BIN, "-i", str(src), "-map", "0:v:0"]

    # Audio map — chỉ khi không mute
    if not mute:
        audio_map = (f"0:a:{best_audio_idx}"
                     if len(audio_streams) > 1 else "0:a:0")
        cmd += ["-map", audio_map]

    # Data map
    # BỎ QUA -map 0:d? vì FFmpeg không thể copy chuẩn track 'mebx' của Apple
    # Việc copy sai định dạng sẽ khiến Apple Photos (iOS) từ chối đọc metadata!

    # Video encoder
    cmd += ["-c:v", real_enc]
    cmd += _quality_args(enc, crf)
    cmd += enc_extra

    # hvc1 tag cho H.265
    if family == "h265" and out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-tag:v", "hvc1"]

    # Audio encode — chỉ khi không mute
    if not mute:
        audio_br_str = "192k"
        if best_audio_br > 0:
            target_kbps = max(128, min(320, int(best_audio_br / 1000)))
            audio_br_str = f"{target_kbps}k"
            
        cmd += [
            "-c:a", "aac_mf",
            "-b:a", audio_br_str,
            "-ar",  "48000",
            "-ac",  "2",
        ]

    # Metadata
    cmd += ["-map_metadata", "0", "-map_metadata:s:v", "0:s:v"]
    if not mute:
        cmd += ["-map_metadata:s:a", "0:s:a"]

    # Container flags
    if out_ext in (".mp4", ".mov", ".m4v"):
        cmd += ["-movflags", "use_metadata_tags+faststart"]
    elif out_ext != ".mkv":
        cmd += ["-movflags", "+faststart"]

    cmd += ["-y", "-progress", ppath,
            "-stats_period", "0.5", "-nostats", str(dst)]

    job["status"] = "waiting"
    job["message"] = "⏳ Đang đợi tới lượt..."
    if family == "av1":
        with av1_semaphore:
            if job.get("cancelled"): return
            with ffmpeg_semaphore:
                if job.get("cancelled"): return
                job["status"] = "processing"
                job["message"] = f"Đang encode ({real_enc}){av1_warning}{' — không âm thanh' if mute else ''}..."
                rc = _run_progress(cmd, job_id, total_s, ppath)
    else:
        with ffmpeg_semaphore:
            if job.get("cancelled"): return
            job["status"] = "processing"
            job["message"] = f"Đang encode ({real_enc}){' — không âm thanh' if mute else ''}..."
            rc = _run_progress(cmd, job_id, total_s, ppath)

    ok = rc == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        _sync_meta(str(src), str(dst))

        ela      = time.time() - t0
        enc_time = (f"{ela:.0f}s" if ela < 60
                    else f"{int(ela//60)}:{int(ela%60):02d}")

        # Probe output
        out_data  = _probe(str(dst))
        out_res   = out_fps = out_audio = "N/A"

        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                break

        if mute:
            out_audio = "🔇 Đã tắt"
        else:
            for s in out_data.get("streams", []):
                if s.get("codec_type") == "audio":
                    ac     = s.get("codec_name", "N/A").upper()
                    br     = s.get("bit_rate")
                    br_int = int(br) if br else 0
                    if br_int < 10000:
                        tags   = s.get("tags", {})
                        br_tag = tags.get("BPS") or tags.get("BPS-eng")
                        if br_tag:
                            try: br_int = int(br_tag)
                            except: pass
                    brs       = f" {br_int//1000}kbps" if br_int > 0 else ""
                    out_audio = f"{ac}{brs}"
                    break

        result = {
            "status":      "done",
            "progress":    100,
            "message":     "Hoàn tất!",
            "output":      str(dst),
            "sz_before":   _fmt_size(str(src)),
            "sz_after":    _fmt_size(str(dst)),
            "encode_time": enc_time,
            "out_res":     out_res,
            "out_fps":     out_fps,
            "out_audio":   out_audio,
            "out_codec":   real_enc,
        }
        if family == "av1":
            if av1_warning:
                result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 không hỗ trợ định dạng MOV, tự động chuyển sang MP4"
            else:
                result["warning"] = "<i data-lucide=\"alert-triangle\" style=\"width:16px;height:16px;vertical-align:-2px;color:var(--red)\"></i> AV1 không import được vào iPhone Photos!"
        job.update(result)
    else:
        if not job.get("cancelled"):
            job.update({"status": "error", "message": "Nén thất bại!"})
        try: dst.unlink()
        except: pass

    try: src.unlink()
    except: pass

    dst      = OUTPUT_DIR / dst_name
    counter  = 1
    while dst.exists():
        dst = OUTPUT_DIR / f"{src.stem}_{preset_key}_{counter}.jpg"
        counter += 1

    tw, th = preset["w"], preset["h"]
    fit    = preset.get("fit","contain")

    # Build filter
    if fit == "contain":
        vf = (
            f"scale={tw}:{th}:force_original_aspect_ratio=decrease,"
            f"pad={tw}:{th}:(ow-iw)/2:(oh-ih)/2:black,"
            f"colorspace=all=bt709:iall=bt709,"
            f"format=yuvj420p"
        )
    else:
        vf = (
            f"scale={tw}:{th}:force_original_aspect_ratio=increase,"
            f"crop={tw}:{th},"
            f"colorspace=all=bt709:iall=bt709,"
            f"format=yuvj420p"
        )

    q   = max(1, int((100 - preset["q"]) / 100 * 31))
    cmd = [FFMPEG_BIN,"-i",str(src),"-vf",vf,"-q:v",str(q),"-frames:v","1","-y",str(dst)]
    job["status"] = "waiting"
    job["message"] = f"⏳ Đang đợi tới lượt..."
    with ffmpeg_semaphore:
        if job.get("cancelled"): return
        job["status"] = "processing"
        job["message"] = f"Đang xuất ảnh..."
        try:
            r   = subprocess.run(cmd, capture_output=True, text=True, encoding="utf-8", errors="replace")
        except Exception as e:
            job.update({"status": "error", "message": f"Lỗi: {e}"})
            return
    ok  = r.returncode == 0 and dst.exists() and dst.stat().st_size > 1024

    if ok:
        # Sync GPS + EXIF từ ảnh gốc
        if EXIFTOOL_BIN:
            subprocess.run([
                EXIFTOOL_BIN,
                "-api","LargeFileSupport=1",
                "-tagsfromfile", str(src),
                "-GPSLatitude","-GPSLongitude","-GPSAltitude",
                "-GPSLatitudeRef","-GPSLongitudeRef",
                "-DateTimeOriginal","-CreateDate",
                "-Make","-Model","-LensModel",
                "--FileModifyDate","--FileCreateDate",
                "-overwrite_original", str(dst)
            ], capture_output=True, text=True, encoding="utf-8", errors="replace")

        out_data  = _probe(str(dst))
        out_res   = "N/A"
        for s in out_data.get("streams", []):
            if s.get("codec_type") == "video":
                out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                break
                
        job.update({
            "status":"done","progress":100,"message":"Hoàn tất!",
            "output":str(dst),
            "sz_before":_fmt_size(str(src)),
            "sz_after": _fmt_size(str(dst)),
            "encode_time": "N/A",
            "out_res": out_res,
            "out_fps": "N/A",
            "out_audio": "N/A",
            "out_enc": preset.get("label", "Preset")
        })
    else:
        job.update({"status":"error","message":"Xử lý ảnh thất bại!"})
    try: src.unlink()
    except: pass

def _fmt_time(s):
    if s < 60: return f"{s:.0f}s"
    return f"{int(s//60)}:{int(s%60):02d}"

@_job_guard
def _audio_job(job_id, src: Path, af: dict):
    job = jobs[job_id]
    dst = OUTPUT_DIR / f"{job_id}_audio{af['ext']}"

    # Copy mode: detect codec và chọn đúng extension
    if af["codec"] == "copy":
        CODEC_EXT_MAP = {
            "aac":".m4a","mp3":".mp3","opus":".opus",
            "flac":".flac","vorbis":".ogg","ac3":".ac3","eac3":".eac3",
        }
        data = _probe(str(src))
        for s in data.get("streams",[]):
            if s.get("codec_type") == "audio":
                dst = OUTPUT_DIR / f"{job_id}_audio{CODEC_EXT_MAP.get(s.get('codec_name','aac'),'.m4a')}"
                break

    cmd = [FFMPEG_BIN,"-i",str(src),"-vn","-c:a",af["codec"]] + af["opts"] + ["-y",str(dst)]
    
    job["status"] = "waiting"
    job["message"] = f"⏳ Đang đợi tới lượt..."
    
    t0  = time.time()
    with ffmpeg_semaphore:
        if job.get("cancelled"): return
        job["status"] = "processing"
        job["message"] = f"Đang tách audio..."
        jobs[job_id]["progress"] = 50
        try:
            r = subprocess.run(
                cmd, capture_output=True, text=True,
                encoding="utf-8", errors="replace"
            )
        except Exception as e:
            job.update({"status": "error", "message": f"Lỗi: {e}"})
            return
    ela = time.time() - t0
    ok  = r.returncode == 0 and dst.exists() and dst.stat().st_size > 0

    if ok:
        enc_time = f"{ela:.0f}s" if ela < 60 else f"{int(ela//60)}:{int(ela%60):02d}"
        job.update({
            "status":"done","progress":100,
            "message":"Xong!","output":str(dst),
            "sz_after":_fmt_size(str(dst)),
            "encode_time":enc_time,
            "out_audio":af["label"],
        })
    else:
        job.update({"status":"error","message":"Tách audio thất bại!"})
    try: src.unlink()
    except: pass

@_job_guard
def _download_job(job_id, url, quality, audio_only,
                  audio_fmt, playlist, browser, cookies_txt, output_dir):
    job    = jobs[job_id]
    
    dl_dir = Path(output_dir)
    try:
        dl_dir.mkdir(parents=True, exist_ok=True)
    except Exception as e:
        job.update({"status": "error", "message": f"Không thể tạo thư mục lưu: {e}"})
        return
        
    log_file = dl_dir / f"{job_id}.log"
    job["log_path"] = str(log_file)
    f_log = open(log_file, "w", encoding="utf-8")

    if playlist and not job.get("playlist_titles"):
        try:
            r = subprocess.run([YTDLP_BIN, "--flat-playlist", "--yes-playlist", "--print", "title", "--encoding", "utf-8", url], capture_output=True, text=True, encoding="utf-8", errors="replace")
            if r.returncode == 0:
                job["playlist_titles"] = [ln.strip() for ln in r.stdout.splitlines() if ln.strip()]
        except: pass

    cmd = [
        YTDLP_BIN,
        "--ffmpeg-location", str(Path(FFMPEG_BIN).parent),
        "--newline", "--no-warnings", "--ignore-errors",
        "--embed-metadata",
        "--embed-thumbnail",
        "--encoding", "utf-8",
    ]
    if playlist:
        cmd += ["-o", str(dl_dir / "%(playlist_title)s" / "%(title)s.%(ext)s")]
        cmd.append("--yes-playlist")
    else:
        cmd += ["-o", str(dl_dir / "%(title)s.%(ext)s")]
        cmd.append("--no-playlist")

    # Cookie
    cookies_file = BASE_DIR / "data" / "cookies.txt"
    if cookies_txt and cookies_file.exists():
        cmd += ["--cookies", str(cookies_file)]
    elif browser and browser not in ("", "none"):
        cmd += ["--cookies-from-browser", browser]

    # Audio only
    if audio_only:
        aq = DOWNLOAD_AUDIO_QUALITY.get(audio_fmt, DOWNLOAD_AUDIO_QUALITY["m4a"])
        cmd += [
            "-f", "bestaudio[ext=m4a]/bestaudio/best",
            "-x",
            "--audio-format",       aq["fmt"],
            "--audio-quality",      aq["quality"],
            "--convert-thumbnails", "jpg",
            "--ppa", "ThumbnailsConvertor+ffmpeg_o:-vf crop=ih:ih",
            "--add-metadata",
        ]
        job["message"] = f"Đang download audio ({aq['fmt'].upper()})..."
    else:
        fmt = DOWNLOAD_QUALITY.get(quality, DOWNLOAD_QUALITY["1080p"])
        cmd += ["-f", fmt, "--merge-output-format", "mp4"]

    cmd.append(url)

    proc = subprocess.Popen(
        cmd, stdout=f_log, stderr=subprocess.STDOUT,
        text=True, bufsize=1, encoding="utf-8"
    )
    dl_files, total, current = [], 0, 0
    last_line = ""
    cookie_error = False

    # Track progress by reading file
    f_log.close()
    with open(log_file, "r", encoding="utf-8") as f_read:
        while proc.poll() is None:
            line = f_read.readline()
            if not line:
                time.sleep(0.5)
                continue
            line = line.rstrip()
            if "[download]" in line and "%" in line:
                try:
                    parts   = line.split()
                    pct_str = next((p for p in parts if "%" in p), None)
                    if pct_str:
                        pct = float(pct_str.replace("%",""))
                        if total > 0:
                            jobs[job_id]["progress"] = min(int((current/total)*100 + (1/total)*pct), 99)
                        else:
                            jobs[job_id]["progress"] = min(int(pct), 99)
                    speed_str = next((p for p in parts if "/s" in p), None)
                    if speed_str:
                        jobs[job_id]["dl_speed"] = speed_str
                except: pass
            elif "[download] Destination:" in line:
                f = Path(line.replace("[download] Destination:","").strip())
                if f not in dl_files: dl_files.append(f)
                title = re.sub(r'\.f\d+$', '', f.stem)
                if total > 0:
                    job["message"] = f"Đang tải bài {current}/{total}: {title}"
                else:
                    job["message"] = f"Đang tải: {title}"
            elif "has already been downloaded" in line:
                f = Path(line.replace("[download]", "").replace("has already been downloaded", "").replace("has already been downloaded and merged", "").strip())
                if f not in dl_files: dl_files.append(f)
                title = re.sub(r'\.f\d+$', '', f.stem)
                if total > 0:
                    job["message"] = f"Đang tải bài {current}/{total}: {title}"
                    job["progress"] = min(int((current/total)*100), 99)
                else:
                    job["message"] = f"Đang tải: {title}"
            elif "Merging formats into" in line:
                m = re.search(r'Merging formats into \"([^\"]+)\"', line)
                if m:
                    f = Path(m.group(1))
                    if f not in dl_files: dl_files.append(f)
            elif "[ExtractAudio] Destination:" in line:
                f = Path(line.replace("[ExtractAudio] Destination:", "").strip())
                if f not in dl_files: dl_files.append(f)
            elif "Downloading item" in line or "Downloading video" in line:
                try:
                    parts = line.split()
                    if "item" in parts:
                        idx = parts.index("item") + 1
                    elif "video" in parts:
                        idx = parts.index("video") + 1
                    current = int(parts[idx])
                    total   = int(parts[idx+2])
                    title_suffix = "..."
                    if playlist and job.get("playlist_titles"):
                        if 1 <= current <= len(job["playlist_titles"]):
                            title_suffix = f": {job['playlist_titles'][current-1]}"
                    job["message"] = f"Đang tải bài {current}/{total}{title_suffix}"
                    if total > 0:
                        job["progress"] = min(int(((current-1)/total)*100), 99)
                except: pass
            elif ("cookie" in line.lower() and any(w in line.lower() for w in ("error","fail","database"))):
                cookie_error = True
            last_line = line

    proc.wait()

    # Fallback: tìm file mới nhất
    dl_files = [f for f in dl_files if f.exists() and f.stat().st_size > 0]
    if not dl_files:
        pool = []
        for e in ["*.mp4","*.mkv","*.webm","*.mp3","*.m4a","*.flac","*.opus"]:
            pool.extend(dl_dir.glob(e))
        valid_pool = [f for f in pool if f.exists() and f.stat().st_size > 0]
        if valid_pool:
            dl_files = [max(valid_pool, key=lambda x: x.stat().st_mtime)]

    success = (proc.returncode == 0 or "Finished downloading playlist" in last_line or bool(dl_files))

    if success:
        time.sleep(0.5) # Wait for file handles to be completely released
        try: log_file.unlink(missing_ok=True)
        except: pass

    if success and dl_files:
        if len(dl_files) == 1 and dl_files[0].parent.parent == dl_dir:
            try:
                import shutil
                f = dl_files[0]
                new_f = dl_dir / f.name
                shutil.move(str(f), str(new_f))
                dl_files[0] = new_f
                f.parent.rmdir()
            except: pass

        out_res = out_fps = out_audio = out_enc = out_dur = "N/A"
        
        if playlist and len(dl_files) > 1:
            total_sz = sum(f.stat().st_size for f in dl_files)
            parent_dir = dl_files[0].parent
            job.update({
                "status":"done","progress":100,
                "message":f"Xong! {len(dl_files)} bài",
                "output":str(parent_dir),"is_dir":True,
                "file_count":len(dl_files),
                "sz_after":_fmt_size_bytes(total_sz),
                "dl_title":parent_dir.name,
                "out_dir": str(parent_dir),
                "encode_time": _fmt_time(time.time() - job["created_at"])
            })
        else:
            f = dl_files[-1]
            try:
                p_data = _probe(str(f))
                try:
                    d = float(p_data["format"]["duration"])
                    out_dur = f"{int(d//60)}:{int(d%60):02d}"
                except: pass
                
                for s in p_data.get("streams", []):
                    if s.get("codec_type") == "video":
                        raw = s.get("codec_name", "").lower()
                        out_enc = CODEC_LABELS.get(raw, raw.upper())
                        out_res = f"{s.get('width','?')}x{s.get('height','?')}"
                        r = s.get("r_frame_rate", "0/1")
                        if "/" in r:
                            n, d2 = r.split("/")
                            if int(d2): out_fps = f"{round(int(n)/int(d2),1)} fps"
                        break
                        
                for s in p_data.get("streams", []):
                    if s.get("codec_type") == "audio":
                        ac = s.get("codec_name", "N/A").upper()
                        br = s.get("bit_rate")
                        br_int = int(br) if br else 0
                        if br_int < 10000:
                            tags = s.get("tags", {})
                            br_tag = tags.get("BPS") or tags.get("BPS-eng")
                            if br_tag:
                                try: br_int = int(br_tag)
                                except: pass
                        brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
                        out_audio = f"{ac}{brs}"
                        break
            except: pass
            
            job.update({
                "status":"done","progress":100,"message":"Xong!",
                "output":str(f),"is_dir":False,
                "sz_after":_fmt_size(str(f)),
                "dl_title":re.sub(r'\.f\d+$', '', f.stem),
                "out_res": out_res,
                "out_fps": out_fps,
                "out_audio": out_audio,
                "out_enc": out_enc,
                "duration": out_dur,
                "encode_time": _fmt_time(time.time() - job["created_at"])
            })
    else:
        err = "Download thất bại!"
        if cookie_error:
            if cookies_txt or (browser and browser not in ("none", "")):
                err = "Lỗi Cookie! Tắt hoàn toàn trình duyệt hoặc dùng cookies.txt"
            else:
                err = "Lỗi tải! Video có thể bị giới hạn độ tuổi hoặc riêng tư (có thể thử dùng cookie)."
        shutil.rmtree(dl_dir, ignore_errors=True)
        job.update({"status":"error","message":err})

@app.route("/api/download_file/<job_id>")

    job = jobs.get(job_id)
    if not job or not job.get("output"):
        return jsonify({"error":"File chưa sẵn sàng!"}), 404

    p = Path(job["output"])
    if not p.exists():
        return jsonify({"error":"File không tìm thấy!"}), 404

    # ── Dùng tên file thực tế từ output path ─────────────
    # Vì backend đã đặt tên đúng (tên_gốc.ext)
    # → dùng thẳng tên file output, không cần build lại
    job_type = job.get("type", "")

    if job_type == "compress":
        # File đã có tên đúng: EVD-15323.mp4
        # Dùng thẳng tên file thực tế
        dl_name = p.name

    elif job_type in ("social", "audio"):
        # Các job khác vẫn build tên từ src_name
        src_name   = job.get("src_name", "video")
        src_stem   = Path(src_name).stem
        clean_stem = re.sub(r"[^\w\-. ]", "_", src_stem)
        suffix_map = {
            "social": "_social",
            "audio":  "_audio",
        }
        suffix  = suffix_map.get(job_type, "")
        dl_name = f"{clean_stem}{suffix}{p.suffix}"

    else:
        # Download — dùng tên file thực tế
        dl_name = p.name

    return send_file(str(p), as_attachment=True, download_name=dl_name)

        else:
            return jsonify({"title": url[:50]})

        return jsonify({"title": url[:50]})
        
@app.route("/api/probe", methods=["POST"])
def api_probe():
    ok, msg = _check_disk_space()
    if not ok: return jsonify({"error": msg}), 507
    f = request.files.get("file")
    if not f: return jsonify({"error": "No file"}), 400

    orig_name = secure_filename(f.filename)
    tmp_path = UPLOAD_DIR / orig_name
    counter = 1
    while tmp_path.exists():
        tmp_path = UPLOAD_DIR / f"{Path(orig_name).stem}_{counter}{Path(orig_name).suffix}"
        counter += 1
    f.save(str(tmp_path))

    try:
        data = _probe(str(tmp_path))
        info = {
            "name": orig_name, "size": _fmt_size(str(tmp_path)),
            "duration": "N/A", "codec": "N/A",
            "res": "N/A", "fps": "N/A", "audio": "N/A",
        }

        # Duration
        try:
            d = float(data["format"]["duration"])
            info["duration"] = f"{int(d//60)}:{int(d%60):02d}"
        except: pass

        # Video stream
        for s in data.get("streams", []):
            if s.get("codec_type") == "video":
                raw = s.get("codec_name", "").lower()
                info["codec"] = CODEC_LABELS.get(raw, raw.upper())
                
                w = s.get('width', '?')
                h = s.get('height', '?')
                
                # Xử lý Rotation
                rotation = 0
                tags = s.get("tags", {})
                if "rotate" in tags:
                    try: rotation = abs(int(tags["rotate"]))
                    except: pass
                else:
                    for sd in s.get("side_data_list", []):
                        if sd.get("side_data_type") == "Display Matrix":
                            try: rotation = abs(int(sd.get("rotation", 0)))
                            except: pass
                
                if rotation in (90, 270) and w != '?' and h != '?':
                    info["res"] = f"{h}x{w}"
                else:
                    info["res"] = f"{w}x{h}"
                r = s.get("r_frame_rate", "0/1")
                if "/" in r:
                    n, d2 = r.split("/")
                    if int(d2): info["fps"] = f"{round(int(n)/int(d2),1)}"
                break

        # Audio stream — lấy bitrate đúng
        for s in data.get("streams", []):
            if s.get("codec_type") == "audio":
                raw    = s.get("codec_name", "").lower()
                ac     = AUDIO_LABELS.get(raw, raw.upper())
                br     = s.get("bit_rate")
                br_int = int(br) if br else 0

                # Thử stream bit_rate trước
                # Nếu < 10kbps → không tin → thử tags
                if br_int < 10000:
                    tags   = s.get("tags", {})
                    br_tag = tags.get("BPS") or tags.get("BPS-eng")
                    if br_tag:
                        try: br_int = int(br_tag)
                        except: pass

                # Vẫn không có → thử format level
                # (chia đều cho số stream để ước tính)
                if br_int < 10000:
                    try:
                        fmt_br     = int(data["format"].get("bit_rate", 0))
                        n_streams  = len([
                            x for x in data.get("streams", [])
                            if x.get("codec_type") in ("video","audio")
                        ])
                        if fmt_br > 0 and n_streams > 0:
                            # Ước tính audio ~192kbps nếu có video
                            # Không dùng fmt_br/n_streams vì video chiếm đa số
                            pass
                    except: pass

                brs = f" {br_int//1000}kbps" if br_int > 10000 else ""
                info["audio"] = f"{ac}{brs}"
                break

        # TRẢ VỀ TÊN FILE ĐÃ UPLOAD LÊN SERVER (KHÔNG XÓA NỮA)
        info["server_file"] = tmp_path.name
        return jsonify(info)
        
    except Exception as e:
        # Nếu có lỗi trong lúc đọc, xóa file rác
        try: tmp_path.unlink()
        except: pass
        return jsonify({"error": str(e)}), 500

def _auto_cleanup():
    while True:
        try:
            now = time.time()
            # Dọn dẹp jobs cũ > 12h
            to_remove = []
            for jid, j in list(jobs.items()):
                if now - j.get("created_at", now) > 12 * 3600:
                    to_remove.append(jid)
            for jid in to_remove:
                jobs.pop(jid, None)
                procs.pop(jid, None)

            # Dọn dẹp uploads & outputs > 12h
            for d in [UPLOAD_DIR, OUTPUT_DIR, TEMP_DIR]:
                if not d.exists(): continue
                for f in d.iterdir():
                    try:
                        if now - f.stat().st_mtime > 12 * 3600:
                            if f.is_file(): f.unlink()
                            elif f.is_dir(): shutil.rmtree(f)
                    except: pass
        except: pass
        time.sleep(3600)  # Chạy mỗi giờ

  try { document.execCommand("copy"); cb(); } catch(x) {}
  document.body.removeChild(e);
}

async function onFileChange(input, tab) {
  const file = input.files[0]; if (!file) return;
  S[tab].file = file;
  input.value = "";   // reset để có thể chọn lại cùng file

  const drop  = $("drop-"  + tab);
  const vinfo = $("vinfo-" + tab);
  if (drop)  drop.style.display  = "none";
  if (vinfo) vinfo.style.display = "block";

  sv("vn-name-" + tab, file.name);

  const sfxMap = { compress:"c", audio:"a", social:"s" };
  const sfx    = sfxMap[tab] || tab;
  sv("vi-size-" + sfx, fmtSize(file.size));

  if (tab === "social") {
    const ext = file.name.split(".").pop().toLowerCase();
    socialFileType = ["jpg","jpeg","heic","heif","png","webp","tiff","tif"]
                     .includes(ext) ? "photo" : "video";
    $("card-video-preset").classList.toggle("hidden", socialFileType !== "video");
    $("card-photo-preset").classList.toggle("hidden", socialFileType !== "photo");
    sv("vi-type-s", socialFileType === "photo" ? "🖼️ Ảnh" : "🎬 Video");
    const vf = $("vinfo-video-s");
    if (vf) vf.classList.toggle("hidden", socialFileType !== "video");
    if (socialFileType !== "video") return;
  }
  
  setBusy(tab, true); // Disable button while uploading and probing
  await probeFile(file, tab);
  setBusy(tab, false); // Enable button after success
}

async function probeFile(file, tab) {
  const loadEl = $("vinfo-loading-" + tab);
  if (loadEl) {
    loadEl.classList.remove("hidden");
    loadEl.textContent = "⏳ Đang upload (0%)...";
  }
  
  return new Promise((resolve, reject) => {
    const fd = new FormData();
    fd.append("file", file);
    
    const xhr = new XMLHttpRequest();
    xhr.open("POST", "/api/probe", true);
    
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable && loadEl) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) {
          loadEl.textContent = `⏳ Đang upload (${pct}%)...`;
        } else {
          loadEl.textContent = "⏳ Đang đọc thông tin...";
        }
      }
    };
    
    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          const sfx  = tab === "compress" ? "c" : tab === "social" ? "s" : "a";
          sv("vi-codec-" + sfx, data.codec    || "—");
          sv("vi-res-"   + sfx, data.res      || "—");
          sv("vi-fps-"   + sfx, data.fps ? data.fps + " fps" : "—");
          sv("vi-dur-"   + sfx, data.duration || "—");
          sv("vi-audio-" + sfx, data.audio    || "—");
          sv("vi-size-"  + sfx, data.size     || "—");
          if (data.server_file) S[tab].server_file = data.server_file;
        } catch(e) {}
      }
      if (loadEl) {
        loadEl.classList.add("hidden");
        loadEl.textContent = "⏳ Đang đọc thông tin..."; // reset text for next time
      }
      resolve();
    };
    
    xhr.onerror = () => {
      if (loadEl) loadEl.classList.add("hidden");
      resolve();
    };
    
    xhr.send(fd);
  });
}

// ── Submit ─────────────────────────────────────────────────
async function _submit(tab, endpoint, fd) {
  setBusy(tab, true); hideErr(tab);
  $("prog-"     + tab).classList.remove("hidden");
  $("res-"      + tab).classList.add("hidden");
  $("btn-area-" + tab).classList.add("hidden");
  
  const bar = $("bar-" + tab); if (bar) bar.style.width = "0%";
  const pct = $("pct-" + tab); if (pct) pct.textContent = "0%";
  
  S[tab].t0 = Date.now();
  const _msgEl = $("msg-" + tab); if (_msgEl) _msgEl.textContent = "Đang chuẩn bị...";
  try {
    const res  = await fetch(endpoint, { method: "POST", body: fd });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error || "Lỗi");
    S[tab].jobId = data.job_id;
    S[tab].timer = setInterval(() => pollJob(tab), 800);
  } catch(e) {
    showErr(tab, e.message);
    $("prog-"     + tab).classList.add("hidden");
    $("btn-area-" + tab).classList.remove("hidden");
    setBusy(tab, false);
  }
}

// ── Poll ───────────────────────────────────────────────────
async function pollJob(tab) {
  const s = S[tab]; if (!s.jobId) return;
  const res  = await fetch("/api/status/" + s.jobId);
  const data = await res.json();

  const msgEl = $("msg-" + tab); if (msgEl) msgEl.textContent = data.message || "";
  const pct   = data.progress || 0;
  const barEl = $("bar-" + tab); if (barEl) barEl.style.width = pct + "%";
  const pctEl = $("pct-" + tab); if (pctEl) pctEl.textContent = pct + "%";
  if (pct > 5 && s.t0) {
    const ela = (Date.now() - s.t0) / 1000;
    const eta = Math.round(ela / pct * (100 - pct));
    const etaEl = $("eta-" + tab);
    if (etaEl) {
      let metaText = eta > 0 ? "còn ~" + eta + "s" : "";
      if (tab === "download" && data.dl_speed)
        metaText = (metaText ? metaText + " · " : "") + data.dl_speed;
      etaEl.textContent = metaText;
    }
  }

  if (data.status === "done") {
    clearInterval(s.timer);
    s.timer = null;
    $("prog-" + tab).classList.add("hidden");
    $("res-"  + tab).classList.remove("hidden");
    setBusy(tab, false);

    if (tab === "compress") {
      showCompressResult(data);
    } else if (tab === "social") {
      const outPath = data.output || "";
      const sb = parseSize(data.sz_before), sa = parseSize(data.sz_after);
      let pct = null; if (sb > 0 && sa > 0) pct = Math.round((1 - sa/sb) * 100);
      
      sv("res-s-fname",  outPath.split("\\").pop().split("/").pop() || "—");
      sv("res-s-before", data.sz_before   || "—");
      sv("res-s-after",  data.sz_after    || "—");
      sv("res-s-pct",    pct !== null ? pct + "%" : "—");
      sv("res-s-time",   data.encode_time || "—");
      sv("res-s-res",    data.out_res     || "—");
      sv("res-s-fps",    data.out_fps     || "—");
      sv("res-s-audio",  data.out_audio   || "—");
      sv("res-s-enc",    data.out_enc     || "—");
    } else if (tab === "audio") {
      sv("res-a-fmt",  data.out_audio  || "—");
      sv("res-a-size", data.sz_after   || "—");
      sv("res-a-time", data.encode_time || "—");
    } else {
      sv("res-d-title", data.dl_title || "—");
      sv("res-d-size",  data.sz_after  || "—");
      sv("res-d-time",  data.encode_time || "—");
      sv("res-d-dur",   data.duration  || "—");
      sv("res-d-res",   data.out_res   || "—");
      sv("res-d-fps",   data.out_fps   || "—");
      sv("res-d-audio", data.out_audio || "—");
      sv("res-d-enc",   data.out_enc   || "—");
      S.download.out_dir = data.out_dir;
      const countItem = $("res-d-count-item");
      if (data.file_count) {
        sv("res-d-count", data.file_count + " tệp");
        if (countItem) countItem.classList.remove("hidden");
      } else {
        if (countItem) countItem.classList.add("hidden");
      }
      const dlBtn = $("btn-dl-file");
      if (dlBtn) dlBtn.classList.toggle("hidden", !!data.is_dir);
      
      document.querySelectorAll(".res-d-detail").forEach(el => {
        el.classList.toggle("hidden", !!data.is_dir);
      });
    }
    if (data.warning) {
      const w = $("warn-" + tab);
      if (w) { w.textContent = data.warning; w.classList.remove("hidden"); }
    }
  } else if (data.status === "error") {
    clearInterval(s.timer);
    s.timer = null;
    $("prog-"     + tab).classList.add("hidden");
    $("btn-area-" + tab).classList.remove("hidden");
    showErr(tab, data.error || data.message || "Xử lý thất bại!", s.jobId);
    setBusy(tab, false);
  }
}

// ── Compress Result ────────────────────────────────────────
function showCompressResult(data) {
  const pct = (data.sz_before && data.sz_after)
    ? Math.round((1 - parseSize(data.sz_after) / parseSize(data.sz_before)) * 100)
    : (data.reduction || null);

  const outPath = data.output || "";
  const outName = outPath.split("\\").pop().split("/").pop() || "—";
  sv("res-fname",  outName);
  sv("res-before", data.sz_before   || "—");
  sv("res-after",  data.sz_after    || "—");
  sv("res-pct",    pct !== null ? pct + "%" : "—");
  sv("res-time",   data.encode_time || "—");
  sv("res-res",    data.out_res     || "—");
  sv("res-fps",    data.out_fps     || "—");
  sv("res-audio",  data.out_audio   || "—");
  sv("res-enc",    S.compress.encLabel || "—");
  sv("res-crf",    "CRF " + (S.compress.crf || "—"));
}

function getOS() {
  const ua = navigator.userAgent;
  if (ua.indexOf("Win") !== -1) return "Windows";
  if (ua.indexOf("Mac") !== -1) return "MacOS";
  if (ua.indexOf("Android") !== -1) return "Android";
  if (ua.indexOf("Linux") !== -1) return "Linux";
  if (ua.indexOf("like Mac") !== -1) return "iOS";
  return "Unknown OS";
}

function getBrowser() {

  if (ua.indexOf("Firefox") !== -1) return "Firefox";
  if (ua.indexOf("Safari") !== -1) return "Safari";

}

function _updateAdminUI(data) {
  const sItem = $("server-status-item");
  if (sItem) sItem.style.display = "flex";

  const dot = $("server-status-dot");
  const txt = $("server-status-txt");
  if (dot && txt) {
    let sc = "var(--yellow)";
    if (data.server_state === "IDLE") sc = "var(--green)";
    else if (data.server_state === "DOWNLOADING" || data.server_state === "UPLOADING") sc = "var(--blue)";
    else if (data.server_state === "EXTRACTING") sc = "var(--purple-l)";
    else if (data.server_state === "LOADING" || data.server_state === "CONNECTING") sc = "var(--text-2)";

    if (data.server_state === "IDLE") {
      dot.className = "status-dot dot-ok";

      dot.style.boxShadow = "";
    } else {
      dot.className = "status-dot";

      dot.style.boxShadow = `0 0 8px ${sc}`;
    }
    txt.style.color = sc;
    txt.style.textShadow = `0 0 8px ${sc}`;
    txt.textContent = `SERVER: ${data.server_state}`;
  }

  const clients = data.clients || [];
  const remoteClients = clients.filter(c => c.ip !== "127.0.0.1" && c.ip !== "::1");

    return;
  }

    else if (c.status === "loading")     { statusColor = "var(--text-2)";   statusText = "LOADING"; }
    else if (c.status === "connecting")  { statusColor = "var(--text-2)";   statusText = "CONNECTING"; }

async function probeBatchFile(index) {
  const fObj = S.compress.files[index];
  fObj.status = 'probing';
  renderBatchList();
  
  const loadEl = $("vinfo-loading-compress");
  loadEl.classList.remove("hidden");
  loadEl.textContent = `⏳ Uploading file ${index+1}/${S.compress.files.length} (0%)...`;
  
  return new Promise((resolve) => {
    const fd = new FormData();
    fd.append("file", fObj.file);
    const xhr = new XMLHttpRequest();
    xhr.open("POST", "/api/probe", true);
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable) {
        const pct = Math.round((e.loaded / e.total) * 100);
        if (pct < 100) loadEl.textContent = `⏳ Uploading file ${index+1} (${pct}%)...`;
        else loadEl.textContent = `⏳ Đang đọc thông tin file ${index+1}...`;
      }
    };
    xhr.onload = () => {
      if (xhr.status === 200) {
        try {
          const data = JSON.parse(xhr.responseText);
          if (data.server_file) {
            fObj.server_file = data.server_file;
            fObj.status = 'ready';
            fObj.meta = data;
          } else {
            fObj.status = 'error';
            fObj.errMsg = data.error || "Không có server_file";
          }
        } catch(e) { 
          fObj.status = 'error'; 
          fObj.errMsg = "Lỗi Parse JSON";
        }
      } else {
        fObj.status = 'error';
        try {
          const errData = JSON.parse(xhr.responseText);
          fObj.errMsg = errData.error || `HTTP ${xhr.status}`;
        } catch(e) {
          fObj.errMsg = `HTTP ${xhr.status}: ${xhr.statusText}`;
        }
      }
      loadEl.classList.add("hidden");
      renderBatchList();
      resolve();
    };
    xhr.onerror = () => {
      fObj.status = 'error';
      fObj.errMsg = "Mất kết nối mạng";
      loadEl.classList.add("hidden");
      renderBatchList();
      resolve();
    };
    xhr.send(fd);
  });
}

  }).join('');
  
  // Submit all requests to backend (they will queue up via lock)
  for (let i = 0; i < readyFiles.length; i++) {
    const fObj = readyFiles[i];
    const id = 'prog-item-' + i;
    
    const fd = new FormData();
    fd.append("server_file", fObj.server_file);
    fd.append("original_name", fObj.file.name);
    fd.append("encoder", enc);
    fd.append("format", fmt);
    fd.append("crf", crf);
    if (muteAudio) fd.append("mute", "1");
    
    try {
      const res = await fetch("/api/compress", { method: "POST", body: fd });
      const data = await res.json();
      if (res.ok && data.job_id) {
        S.compress.jobs[data.job_id] = { ui_id: id, file: fObj.file.name, done: false };
        const cBtn = $("btn-cancel-" + id);
        if (cBtn) {
          cBtn.style.display = "block";
          cBtn.onclick = () => cancelSingleJob(data.job_id, id);
        }
      } else {
        sv('msg-' + id, "❌ Lỗi: " + (data.error || "Xử lý thất bại"));
      }
    } catch (e) {
      sv('msg-' + id, "❌ Lỗi kết nối");
    }
  }
  
  // Start polling
  if (S.compress.timer) clearInterval(S.compress.timer);
  S.compress.timer = setInterval(pollBatchProgress, 1000);
}

function parseSize(szStr) {
  if (!szStr) return 0;
  const m = szStr.match(/([\d.]+)\s*([KMGT]?B)/i);
  if (!m) return parseFloat(szStr) || 0;
  const v = parseFloat(m[1]), u = m[2].toUpperCase();
  if (u === 'KB') return v * 1024;
  if (u === 'MB') return v * 1024 * 1024;
  if (u === 'GB') return v * 1024 * 1024 * 1024;
  return v;
}

async function pollBatchProgress() {
  const activeJobIds = Object.keys(S.compress.jobs).filter(jid => !S.compress.jobs[jid].done);
  
  if (activeJobIds.length === 0) {
    clearInterval(S.compress.timer);
    S.compress.timer = null;
    setBusy('compress', false);
    renderBatchResult();
    return;
  }
  
  try {
    const res = await fetch("/api/status_batch", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ job_ids: activeJobIds })
    });
    if (!res.ok) return;
    const batchData = await res.json();

    activeJobIds.forEach(jid => {
      const data = batchData[jid];
      if (!data) return;
      
      const jobMeta = S.compress.jobs[jid];
      const uid = jobMeta.ui_id;

      const pct = data.progress || 0;
      const elPct = $('pct-' + uid);
      const elMsg = $('msg-' + uid);
      const elBar = $('bar-' + uid);
      const elEta = $('eta-' + uid);

      if (elPct) elPct.innerText = pct + "%";
      if (elMsg) elMsg.innerText = data.message || "Đang xử lý...";
      if (elBar) elBar.style.width = pct + "%";

      if (pct > 0 && pct < 100) {
        if (!jobMeta.t0) jobMeta.t0 = Date.now() / 1000;
        const ela = (Date.now() / 1000) - jobMeta.t0;
        const eta = Math.round((ela / pct) * (100 - pct));
        if (elEta) elEta.innerText = eta > 0 ? "còn ~" + eta + "s" : "";
      } else {
        if (elEta) elEta.innerText = "";
      }

      if (data.status === "done" || data.status === "error") {
        jobMeta.done = true;
        jobMeta.result = data;
        const btnCancel = $("btn-cancel-" + uid);
        if (btnCancel) btnCancel.style.display = "none";
        const progItem = $(uid);
        if (progItem) {
          progItem.style.background = "transparent";
          progItem.style.padding = "12px 0";
          progItem.style.borderBottom = "1px solid rgba(255,255,255,0.05)";
          progItem.innerHTML = _buildJobResultHTML(jid, jobMeta.file, data);
        }
      }
    });
  } catch (e) {
    console.error("Batch polling error:", e);
  }
}

function renderBatchResult() {
  $("prog-compress").classList.add("hidden");
  $("res-compress").classList.remove("hidden");

  const itemStyle = "padding:12px 0; border-bottom:1px solid rgba(255,255,255,0.05); font-size:13px;";
  $("batch-result-list").innerHTML = Object.entries(S.compress.jobs)
    .map(([jid, job]) => `<div style="${itemStyle}">${_buildJobResultHTML(jid, job.file, job.result)}</div>`)
    .join('');
}
